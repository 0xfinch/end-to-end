36/**
37 * An OTRv3 DSA PUBKEY.
38 * @constructor
39 * @extends {e2e.otr.pubkey.Pubkey}
40 * @param {!e2e.signer.key.DsaPublicKey} key DSA public key object.
41 */
42e2e.otr.pubkey.Dsa = function(key) {
43 goog.base(this);
44 this.p_ = goog.array.clone(key.p);
45 this.q_ = goog.array.clone(key.q);
46 this.g_ = goog.array.clone(key.g);
47 this.y_ = goog.array.clone(key.y);
48};
49goog.inherits(e2e.otr.pubkey.Dsa, e2e.otr.pubkey.Pubkey);

59/** @inheritDoc */
60e2e.otr.pubkey.Dsa.prototype.serializePubkey = function() {
61 return e2e.otr.serializeBytes([
62 new e2e.otr.Mpi(new Uint8Array(this.p_)),
63 new e2e.otr.Mpi(new Uint8Array(this.q_)),
64 new e2e.otr.Mpi(new Uint8Array(this.g_)),
65 new e2e.otr.Mpi(new Uint8Array(this.y_))
66 ]);

70/**
71 * Deconstructs DSA PUBKEY into p, q, g, and y.
72 * @return {!e2e.signer.key.DsaPublicKey} The object containing p, q, g, and y.
73 */
74e2e.otr.pubkey.Dsa.prototype.deconstruct = function() {
75 return {
76 p: goog.array.clone(this.p_),
77 q: goog.array.clone(this.q_),
78 g: goog.array.clone(this.g_),
79 y: goog.array.clone(this.y_)
80 };

84/**
85 * Extracts a DSA PUBKEY from the body, and returns the DSA PUBKEY.
86 * @param {!Uint8Array} body The body from where to extract the data.
87 * @return {!e2e.otr.pubkey.Dsa} The generated packet.
88 */
89e2e.otr.pubkey.Dsa.parse = function(body) {
90 var iter = new e2e.otr.util.Iterator(body);
91 var key = {
92 p: e2e.otr.Mpi.parse(iter.nextEncoded()),
93 q: e2e.otr.Mpi.parse(iter.nextEncoded()),
94 g: e2e.otr.Mpi.parse(iter.nextEncoded()),
95 y: e2e.otr.Mpi.parse(iter.nextEncoded())
96 };
97 if (iter.hasNext()) {
98 throw new e2e.otr.error.ParseError('Malformed DSA PUBKEY.');
99 }
100 return new e2e.otr.pubkey.Dsa(key);
101};

104/** @inheritDoc */
105e2e.otr.pubkey.Dsa.prototype.pack = function() {
106 return {p: this.p_, q: this.q_, g: this.g_, y: this.y_};
107};

110/** @inheritDoc */
111e2e.otr.pubkey.Dsa.unpack = function(data) {
112 assert(e2e.isByteArray(data.p) && e2e.isByteArray(data.q) &&
113 e2e.isByteArray(data.g) && e2e.isByteArray(data.y));
114 return new e2e.otr.pubkey.Dsa(/** @type {!e2e.signer.key.DsaPublicKey} */
115 (data));
116};
