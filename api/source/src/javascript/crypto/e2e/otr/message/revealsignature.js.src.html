44var constants = e2e.otr.constants;

49/**
50 * An OTRv3 REVEAL SIGNATURE.
51 * @constructor
52 * @extends {e2e.otr.message.Encoded}
53 * @param {!e2e.otr.Session} session The enclosing session.

55e2e.otr.message.RevealSignature = function(session) {

58goog.inherits(e2e.otr.message.RevealSignature, e2e.otr.message.Encoded);

76/**
77 * Serialize the REVEAL SIGNATURE into a Uint8Array.
78 * @return {!Uint8Array} The serialized REVEAL SIGNATURE.
79 */
80e2e.otr.message.RevealSignature.prototype.serializeMessageContent = function() {
81 var keys = this.session_.deriveKeyValues();
82 var keyB = this.session_.keymanager.getKey();
83 var mb = new goog.crypt.Hmac(new e2e.hash.Sha256(), keys.m1)
84 .getHmac(Array.apply([], e2e.otr.serializeBytes([
85 keyB.key.generate(),
86 this.session_.keymanager.getRemoteKey(new Uint8Array(4)).key,
87 this.session_.getPublicKey(),
88 keyB.keyid
89 ])));
90
91 var xb = Array.apply([], e2e.otr.serializeBytes([
92 this.session_.getPublicKey(),
93 keyB.keyid,
94 new (this.session_.getSigner())(mb)
95 ]));
96
97 var sig = new e2e.otr.Data(e2e.otr.util.aes128ctr.encrypt(keys.c, xb));
98
99 var mac = new goog.crypt.Hmac(new e2e.hash.Sha256(), keys.m2)
100 .getHmac(Array.apply([], sig.serialize()));
101 mac = mac.slice(0, 160 / 8);
102
103 return e2e.otr.serializeBytes([
104 new e2e.otr.Data(new Uint8Array(this.session_.authData.r)), sig, mac]);
105};

108/**
109 * Processes a REVEAL SIGNATURE message.
110 * @param {!e2e.otr.Session} session The enclosing session.
111 * @param {!Uint8Array} data The data to be processed.
112 */
113e2e.otr.message.RevealSignature.process = function(session, data) {
114 switch (session.getAuthState()) {
115 case AUTHSTATE.AWAITING_REVEALSIG:
116 var iter = new e2e.otr.util.Iterator(data);
117 var r = e2e.otr.Data.parse(iter.nextEncoded()).deconstruct();
118 var aesxb = iter.nextEncoded();
119 var mac = iter.next(20);
120
121 if (iter.hasNext() || mac.length != 20) {
122 throw new e2e.otr.error.ParseError('Malformed REVEAL SIGNATURE.');
123 }
124
125 // TODO(rcc): Remove annotation when closure-compiler #260 is fixed.
126 var gxmpi = e2e.otr.util.aes128ctr.decrypt(r, /** @type {!Uint8Array} */ (
127 e2e.otr.assertState(session.authData.aesgx, 'AES(gx) not defined')));
128 var gxmpiHash = new e2e.hash.Sha256().hash(Array.apply([], gxmpi));
129
130 var gx = Array.apply([], e2e.otr.Mpi.parse(gxmpi).deconstruct());
131 var dh = session.keymanager.getKey().key;
132
133 if (
134 // h(g^x) mismatch.
135 e2e.otr.compareByteArray(/** @type {!Uint8Array} */ (
136 e2e.otr.assertState(session.authData.hgx, 'hgx not defined')),
137 gxmpiHash) ||
138
139 // Or Invalid g^x.
140 !dh.isValidBase(gx)
141 ) {
142 // TODO(rcc): Log the error and/or warn the user.
143 return;
144 }
145
146 var s = dh.generate(gx);
147 var keys = session.deriveKeyValues(s);
148
149 var calculatedMac = new goog.crypt.Hmac(new e2e.hash.Sha256(), keys.m2)
150 .getHmac(Array.apply([], aesxb));
151
152 calculatedMac = calculatedMac.slice(0, 160 / 8);
153
154 if (e2e.otr.compareByteArray(Array.apply([], mac), calculatedMac)) {
155 // TODO(rcc): Log the error and/or warn the user.
156 return;
157 }
158
159 var xb = e2e.otr.util.aes128ctr.decrypt(keys.c,
160 e2e.otr.Data.parse(aesxb).deconstruct());
161
162 iter = new e2e.otr.util.Iterator(xb);
163 var pubBType = iter.next(2);
164 // TODO(rcc): Make Type.parse accept Iterator to pull appropriate data.
165 var pubB = new e2e.otr.pubkey.Dsa({
166 p: Array.apply([], e2e.otr.Mpi.parse(iter.nextEncoded()).deconstruct()),
167 q: Array.apply([], e2e.otr.Mpi.parse(iter.nextEncoded()).deconstruct()),
168 g: Array.apply([], e2e.otr.Mpi.parse(iter.nextEncoded()).deconstruct()),
169 y: Array.apply([], e2e.otr.Mpi.parse(iter.nextEncoded()).deconstruct())
170 });
171 var keyidB = iter.next(4);
172 var sigmb = e2e.otr.Sig.parse(iter.next(40));
173
174 if (iter.hasNext()) {
175 throw new e2e.otr.error.ParseError('Malformed xb in REVEAL SIGNATURE.');
176 }
177
178 var mb = new goog.crypt.Hmac(new e2e.hash.Sha256(), keys.m1)
179 .getHmac(Array.apply([], e2e.otr.serializeBytes([
180 gx,
181 session.keymanager.getKey().key.generate(),
182 pubB,
183 keyidB
184 ])));
185
186 if (!e2e.otr.Sig.verify(pubB.deconstruct(), mb, sigmb)) {
187 // TODO(rcc): Log the error and/or warn the user.
188 return;
189 }
190
191 session.keymanager.storeRemoteKey(keyidB, gx);
192 session.authData.s = s;
193 session.send(new e2e.otr.message.Signature(session));
194 session.setAuthState(AUTHSTATE.NONE);
195 session.setMsgState(constants.MSGSTATE.ENCRYPTED);
196
197 // TODO(rcc): Send any stored messages.
198
199 break;
200
201 case AUTHSTATE.NONE:
202 case AUTHSTATE.AWAITING_DHKEY:
203 case AUTHSTATE.AWAITING_SIG:
204 case AUTHSTATE.V1_SETUP:
205 return;
206
207 default:
208 e2e.otr.assertState(false, 'Invalid auth state.');
209 }
210};
