47/**
48 * An OTRv3 SIGNATURE.
49 * @constructor
50 * @extends {e2e.otr.message.Encoded}
51 * @param {!e2e.otr.Session} session The enclosing session.
52 */
53e2e.otr.message.Signature = function(session) {
54 goog.base(this, session);
55};
56goog.inherits(e2e.otr.message.Signature, e2e.otr.message.Encoded);
57
58
59/**
60 * Specifies the type of the message.
61 * @type {!e2e.otr.Byte}
62 */
63e2e.otr.message.Signature.MESSAGE_TYPE = constants.MessageType.SIGNATURE;

66/**
67 * Serialize the SIGNATURE into a Uint8Array.
68 * @return {!Uint8Array} The serialized SIGNATURE.
69 */
70e2e.otr.message.Signature.prototype.serializeMessageContent = function() {
71 var keys = this.session_.deriveKeyValues();
72 var keyA = this.session_.keymanager.getKey();
73 var ma = new goog.crypt.Hmac(new e2e.hash.Sha256(), keys.m1prime)
74 .getHmac(Array.apply([], e2e.otr.serializeBytes([
75 keyA.key.generate(),
76 this.session_.keymanager.getRemoteKey().key,
77 this.session_.getPublicKey(),
78 keyA.keyid
79 ])));
80
81 var xa = Array.apply([], e2e.otr.serializeBytes([
82 this.session_.getPublicKey(),
83 keyA.keyid,
84 new (this.session_.getSigner())(ma)
85 ]));
86
87 var sig = new e2e.otr.Data(
88 e2e.otr.util.aes128ctr.encrypt(keys.cprime, xa));
89
90 var mac = new goog.crypt.Hmac(new e2e.hash.Sha256(), keys.m2prime)
91 .getHmac(Array.apply([], sig.serialize()));
92
93 mac = mac.slice(0, 160 / 8);
94
95 return e2e.otr.serializeBytes([sig, mac]);
96};
97
98
99/**
100 * Processes a SIGNATURE message.
101 * @param {!e2e.otr.Session} session The enclosing session.
102 * @param {!Uint8Array} data The data to be processed.
103 */
104e2e.otr.message.Signature.process = function(session, data) {
105 switch (session.getAuthState()) {
106 case AUTHSTATE.AWAITING_SIG:
107 var iter = new e2e.otr.util.Iterator(data);
108 var aesxa = iter.nextEncoded();
109 var mac = iter.next(20);
110
111 var keys = session.deriveKeyValues();
112
113 var calculatedMac = new goog.crypt.Hmac(new e2e.hash.Sha256(),
114 keys.m2prime).getHmac(Array.apply([], aesxa));
115
116 calculatedMac = calculatedMac.slice(0, 160 / 8);
117
118 if (e2e.otr.compareByteArray(mac, calculatedMac)) {
119 // TODO(rcc): Log the error and/or warn the user.
120 return;
121 }
122
123 var xa = e2e.otr.util.aes128ctr.decrypt(keys.cprime,
124 e2e.otr.Data.parse(aesxa).deconstruct());
125
126 iter = new e2e.otr.util.Iterator(xa);
127
128 // TODO(rcc): Make Type.parse accept Iterator to pull appropriate data.
129 var pubAType = iter.next(2);
130 var pubA = new e2e.otr.pubkey.Dsa({
131 p: Array.apply([], e2e.otr.Mpi.parse(iter.nextEncoded()).deconstruct()),
132 q: Array.apply([], e2e.otr.Mpi.parse(iter.nextEncoded()).deconstruct()),
133 g: Array.apply([], e2e.otr.Mpi.parse(iter.nextEncoded()).deconstruct()),
134 y: Array.apply([], e2e.otr.Mpi.parse(iter.nextEncoded()).deconstruct())
135 });
136 var keyidA = iter.next(4);
137 var sigma = e2e.otr.Sig.parse(iter.next(40));
138
139 var gy = session.keymanager.getRemoteKey(new Uint8Array(4)).key;
140 var ma = new goog.crypt.Hmac(new e2e.hash.Sha256(), keys.m1prime)
141 .getHmac(Array.apply([], e2e.otr.serializeBytes([
142 gy,
143 session.keymanager.getKey().key.generate(),
144 pubA,
145 keyidA
146 ])));
147
148 if (!e2e.otr.Sig.verify(pubA.deconstruct(), ma, sigma)) {
149 // TODO(rcc): Log the error and/or warn the user.
150 return;
151 }
152
153 session.keymanager.storeRemoteKey(keyidA, gy);
154 session.setAuthState(AUTHSTATE.NONE);
155 session.setMsgState(constants.MSGSTATE.ENCRYPTED);
156
157 // TODO(rcc): Send any stored messages.
158
159 break;
160
161 case AUTHSTATE.NONE:
162 case AUTHSTATE.AWAITING_DHKEY:
163 case AUTHSTATE.AWAITING_REVEALSIG:
164 return;
165
166 default:
167 e2e.otr.assertState(false, 'Invalid auth state.');
168 }
169};
170
171e2e.otr.message.handler.add(e2e.otr.message.Signature);
172}); // goog.scope
