51/**
52 * An OTRv3 DATA message.
53 * @constructor
54 * @extends {e2e.otr.message.Encoded}
55 * @param {!e2e.otr.Session} session The enclosing session.
56 * @param {string} plaintext The plaintext portion of the message.
57 * @param {!Array.<e2e.otr.message.Tlv>=} opt_tlvs 0+ TLV records to include.
58 * @param {number=} opt_flags The flags to set on the message.
59 */
60e2e.otr.message.Data = function(session, plaintext, opt_tlvs, opt_flags) {
61 goog.base(this, session);
62
63 if (plaintext.indexOf('\0') > -1) {
64 throw new e2e.otr.error.InvalidArgumentsError(
65 'Plaintext may not contain \\0');
66 }
67
68 this.plaintext_ = plaintext;
69 this.tlvs_ = opt_tlvs || [];
70 this.flags_ = new Uint8Array([0x00 | opt_flags]);
71};
72goog.inherits(e2e.otr.message.Data, e2e.otr.message.Encoded);

82/**
83 * Generates the keys used for AES and SHA1-HMAC of data messages.
84 * @param {!e2e.cipher.DiffieHellman} localDh The DH of the local key component.
85 * @param {!e2e.ByteArray} remoteKey The remote key component.
86 * @return {!{sendingAes: !e2e.ByteArray, receivingAes: !e2e.ByteArray,
87 * sendingMac: !e2e.ByteArray, receivingMac: !e2e.ByteArray}}
88 */
89e2e.otr.message.Data.computeKeys = function(localDh, remoteKey) {
90 var highEnd = e2e.otr.compareByteArray(localDh.generate(), remoteKey) > 0;
91 var sendbyte = highEnd ? 0x01 : 0x02;
92 var recvbyte = highEnd ? 0x02 : 0x01;
93 var secret = localDh.generate(remoteKey);
94 var secbytes = new e2e.otr.Mpi(new Uint8Array(secret)).serialize();
95
96 /**
97 * h1 function as defined in OTR spec.
98 * @param {number} b The input byte.
99 * @return {!e2e.ByteArray} The 160-bit output of the SHA-1 hash of the
100 * (5+len) bytes consisting of the byte b, followed by secbytes.
101 */
102 var h1 = function(b) {
103 return new e2e.hash.Sha1().hash([b].concat(secbytes));
104 };
105
106 var sendingAes = h1(sendbyte).slice(0, 16);
107 var receivingAes = h1(recvbyte).slice(0, 16);
108
109 return {
110 sendingAes: sendingAes,
111 sendingMac: new e2e.hash.Sha1().hash(sendingAes),
112 receivingAes: receivingAes,
113 receivingMac: new e2e.hash.Sha1().hash(receivingAes)
114 };
115};

118/**
119 * Serialize the DATA message into a Uint8Array.
120 * @return {!Uint8Array} The serialized DATA message.
121 */
122e2e.otr.message.Data.prototype.serializeMessageContent = function() {
123 var ctrTop = e2e.incrementByteArray(this.session_.sendCtr);
124 var remoteKey = this.session_.keymanager.getRemoteKey();
125 var dh = this.session_.keymanager.getKey();
126
127 var message = e2e.otr.serializeBytes([
128 e2e.stringToByteArray(this.plaintext_),
129 [0x00] // NULL BYTE
130 ].concat(this.tlvs_));
131
132 var keys = e2e.otr.message.Data.computeKeys(dh.key, remoteKey.key);
133
134 var encryptedMessage = e2e.otr.util.aes128ctr.encrypt(keys.sendingAes,
135 message, ctrTop.concat(goog.array.repeat(0, 8)));
136
137 var nextDh = new e2e.cipher.DiffieHellman(constants.DH_MODULUS,
138 [constants.DH_GENERATOR]);
139 this.session_.keymanager.storeKey(nextDh);
140
141 var partialData = e2e.otr.serializeBytes([
142 this.flags_,
143 dh.keyid,
144 remoteKey.keyid,
145 new e2e.otr.Mpi(new Uint8Array(nextDh.generate())),
146 ctrTop,
147 new e2e.otr.Data(new Uint8Array(encryptedMessage))
148 ]);
149
150 var mac = new goog.crypt.Hmac(new e2e.hash.Sha1(), keys.sendingMac).getHmac(
151 this.addHeader(partialData));
152
153 // TODO(rcc): old MAC keys to be revealed.
154 var oldMacKeys = [];
155
156 return e2e.otr.serializeBytes([partialData, mac, oldMacKeys]);
157};

160/**
161 * Processes a DATA message message.
162 * @param {!e2e.otr.Session} session The enclosing session.
163 * @param {!Uint8Array} data The data to be processed.
164 */
165e2e.otr.message.Data.process = function(session, data) {
166 switch (session.getMsgState()) {
167 case MSGSTATE.ENCRYPTED:
168 var tee = new e2e.otr.util.Tee();
169 var iter = new e2e.otr.util.Iterator(data);
170
171 var flags = e2e.otr.byteToNum(tee.tee(iter.next()));
172 var ignoreUnreadable = flags & 0x01;
173 var remoteKeyid = tee.tee(iter.next(4));
174 var keyid = tee.tee(iter.next(4));
175 var nextDh = e2e.otr.Mpi.parse(tee.tee(iter.nextEncoded())).deconstruct();
176 var ctrTop = Array.apply([], tee.tee(iter.next(8)));
177 var encryptedMessage = e2e.otr.Data.parse(tee.tee(iter.nextEncoded()))
178 .deconstruct();
179 var mac = iter.next(20);
180 // TODO(rcc): handle revealing old MAC keys.
181
182 var key = session.keymanager.getKey(keyid);
183 var remoteKey = session.keymanager.getRemoteKey(remoteKeyid);
184 var keys = e2e.otr.message.Data.computeKeys(key.key, remoteKey.key);
185
186 var macData = e2e.otr.message.Encoded.addHeader(session,
187 constants.MessageType.DATA, tee.dump(), true);
188 var calculatedMac = new goog.crypt.Hmac(new e2e.hash.Sha1(),
189 keys.receivingMac).getHmac(macData);
190
191 if (e2e.otr.compareByteArray(mac, calculatedMac)) {
192 if (!ignoreUnreadable) {
193 // TODO(rcc): Log the error and/or warn the user. Remove extra return.
194 return;
195 }
196 return;
197 }
198
199 session.keymanager.storeRemoteKey(e2e.incrementByteArray(Array.apply([],
200 remoteKeyid)), Array.apply([], nextDh));
201
202 var message = e2e.otr.util.aes128ctr.decrypt(keys.receivingAes,
203 encryptedMessage, ctrTop.concat(goog.array.repeat(0, 8)));
204
205 var split = Array.prototype.indexOf.call(message, '\0');
206 session.display(e2e.byteArrayToString(message.subarray(0, split)));
207 // TODO(rcc): Process TLVs.
208 break;
209
210 case MSGSTATE.PLAINTEXT:
211 case MSGSTATE.FINISHED:
212 // TODO(rcc): Inform the user that an unreadable encrypted message was
213 // received, and reply with an Error Message.
214 throw new e2e.otr.error.NotImplementedError('Not yet implemented.');
215 break;
216
217 default:
218 e2e.otr.assertState(false, 'Invalid message state.');
219 }
220};
