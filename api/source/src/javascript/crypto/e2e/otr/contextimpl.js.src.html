35goog.scope(function() {

38/** @const */
39var STORAGE_PUBLIC_KEY = 'PUBKEY';
40
41
42/** @const */
43var STORAGE_PRIVATE_KEY = 'PRIVKEY';
44
45
46
47/**
48 * Constructor for the context manager.
49 * @constructor
50 * @implements {e2e.otr.Context}
51 */
52e2e.otr.ContextImpl = function() {
53 this.localStorage_ = new goog.storage.mechanism.HTML5LocalStorage();
54
55 // The properties defined here are backed by localStorage.
56 this.defineLocalStorageProperties_({
57 pubkey: {
58 key: STORAGE_PUBLIC_KEY,
59 get: e2e.otr.pubkey.Dsa.unpack,
60 set: function(dsa) { return dsa.pack(); }
61 },
62 privkey_: STORAGE_PRIVATE_KEY
63 });
64
65 if (!this.privkey_) {
66 // TODO(rcc): Generate a DSA public/private key pair.
67 throw new e2e.otr.error.NotImplementedError(
68 'Generating DSA keys is not yet supported');
69 }
70
71 if (!this.pubkey) {
72 // TODO(rcc): Generate a public key from the private key.
73 // This case only occurs when a private key is present without a public key.
74 throw new e2e.otr.error.NotImplementedError(
75 'Generating DSA keys is not yet supported');
76 }
77
78 e2e.otr.implements(e2e.otr.ContextImpl, e2e.otr.Context);
79};

97/**
98 * Defines properties backed by a local storage key.
99 * {prop: PROP_KEY} creates this.prop backed by localStorage key PROP_KEY.
100 * {prop: {key: PROP_KEY, get: getter, set: setter}} creates this.prop backed by
101 * localStorage key PROP_KEY and the specified getter and setter.
102 *
103 * @private
104 * @template J
105 * J is a a JSON-compatible type.
106 * @template T
107 * @param {!Object.<string,
108 * (string|!{
109 * key: string,
110 * get: (function(J):T|undefined),
111 * set: (function(T):J|undefined)
112 * })>} props The property to localStorage key map with optional get/setter.
113 */
114e2e.otr.ContextImpl.prototype.defineLocalStorageProperties_ = function(props) {
115 Object.keys(props).forEach(goog.bind(function(prop) {
116 var val = props[prop];
117 var key = val.key || val;
118 var getFilter = val.get || goog.functions.identity;
119 var setFilter = val.set || goog.functions.identity;
120 // TODO(rcc): Consider adding caching to prevent excessive parse/stringify.
121 Object.defineProperty(this, prop, {
122 enumerable: true,
123 get: function() {
124 return getFilter(JSON.parse(this.localStorage_.get(key)));
125 },
126 set: function(val) {
127 return this.localStorage_.set(key, JSON.stringify(setFilter(val)));
128 }
129 });
130 }, this));
131};

135e2e.otr.ContextImpl.prototype.getSigner = function() {
136 return e2e.otr.Sig.bind(null, this.privkey_);
137};
