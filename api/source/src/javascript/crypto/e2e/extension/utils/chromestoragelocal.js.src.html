28/**
29 * Creates an iterable mechanism using chrome.storage.local.
30 * @param {function(!e2e.ext.util.ChromeStorageLocal)=} opt_callback Callback
31 * function to call once storage mechanism has been initialized.
32 * @constructor
33 * @extends {goog.storage.mechanism.IterableMechanism}
34 * @final
35 */
36e2e.ext.util.ChromeStorageLocal = function(opt_callback) {
37 this.storage_ = {};
38 this.updateStorage_(opt_callback);
39 chrome.storage.onChanged.addListener(goog.bind(function(changes, areaName) {
40 var keyName = e2e.ext.util.ChromeStorageLocal.STORAGE_KEY_NAME_;
41 if (areaName == 'local' && keyName in changes) {
42 this.storage_ = changes[keyName].newValue || {};
43 }
44 }, this));
45};
46goog.inherits(e2e.ext.util.ChromeStorageLocal,
47 goog.storage.mechanism.IterableMechanism);
48

54e2e.ext.util.ChromeStorageLocal.STORAGE_KEY_NAME_ =
55 'e2e.ext.util.ChromeStorageLocal';

58/**
59 * @param {function(!e2e.ext.util.ChromeStorageLocal)=} opt_callback
60 * @private
61 */
62e2e.ext.util.ChromeStorageLocal.prototype.updateStorage_ = function(
63 opt_callback) {
64 var keyName = e2e.ext.util.ChromeStorageLocal.STORAGE_KEY_NAME_;
65 chrome.storage.local.get(
66 keyName,
67 goog.bind(function(values) {
68 this.storage_ = values[keyName] || {};
69 opt_callback && opt_callback(this);
70 }, this));
71};

74/** @private */
75e2e.ext.util.ChromeStorageLocal.prototype.saveChanges_ = function() {
76 var newStorage = {};
77 newStorage[e2e.ext.util.ChromeStorageLocal.STORAGE_KEY_NAME_] = this.storage_;
78 chrome.storage.local.set(newStorage);
79};
