112/**
113 * Sends the payload to the helper.
114 * @param {*} payload Message payload.
115 * @param {!function(...)} callback The function to invoke with the response.
116 * @param {!function(Error)} errorCallback Function to invoke on error
117 * conditions.
118 * @private
119 */
120e2e.ext.utils.HelperProxy.prototype.sendMessage_ = function(payload, callback,
121 errorCallback) {
122 this.setupConnection(goog.bind(function() {
123 this.sendMessageImpl(payload, goog.bind(function(response) {
124 if (arguments.length == 0 && chrome.runtime.lastError) {
125 errorCallback(new Error(chrome.runtime.lastError.message));
126 } else if (response instanceof Error) {
127 errorCallback(response);
128 } else {
129 callback(response);
130 }
131 }, this), errorCallback);
132 }, this), errorCallback);
133};

169/**
170 * Helper proxy that connects with a helper injected via chrome.tabs API.
171 * To be used in a Chrome Extension.
172 * @param {boolean} isLookingGlassEnabled Indicates whether the looking glass is
173 * enabled.
174 * @param {number=} opt_tabId Tab ID to communicate with. Defaults
175 * to ID of the current tab in a current window.
176 * @constructor
177 * @extends {e2e.ext.utils.HelperProxy}
178 */
179e2e.ext.utils.TabsHelperProxy = function(isLookingGlassEnabled,
180 opt_tabId) {
181 goog.base(this, isLookingGlassEnabled);
182 /**
183 * @type {number|undefined} Tab ID to connect to.
184 * @private
185 */
186 this.tabId_ = opt_tabId;
187};
188goog.inherits(e2e.ext.utils.TabsHelperProxy, e2e.ext.utils.HelperProxy);

191/** @override */
192e2e.ext.utils.TabsHelperProxy.prototype.setupConnection = function(
193 callback, errorCallback) {
194 if (goog.isDefAndNotNull(this.tabId_)) {
195 this.connectToHelper_(callback, errorCallback);
196 } else {
197 // Query active tab.
198 chrome.tabs.query({
199 active: true,
200 currentWindow: true
201 }, goog.bind(function(tabs) {
202 if (!tabs || tabs.length !== 1) {
203 errorCallback(new Error('Cannot determine the active tab.'));
204 return;
205 }
206 var tabId = tabs[0] ? tabs[0].id : undefined;
207 this.tabId_ = tabId;
208 this.connectToHelper_(callback, errorCallback);
209 }, this));
210 }
211};

217 try {

226/**
227 * Connects to the helper, injecting it if the connection has not yet been
228 * setup.
229 * @param {!function()} callback The function to invoke once the
230 * connection is established.
231 * @param {!function(Error)} errorCallback Function to invoke if error occurred.
232 * @private
233 */
234e2e.ext.utils.TabsHelperProxy.prototype.connectToHelper_ = function(
235 callback, errorCallback) {
236 // chrome.tabs.executeScript in a view page doesn't call the callback if
237 // the injection did not succeed (while it does in the background page).
238 // As a workaround, move the actual code injection task into the background
239 // page.
240 var tabId = /** @type {number} */ (this.tabId_);
241 chrome.runtime.getBackgroundPage(function(ignore) {
242 chrome.tabs.executeScript(goog.asserts.assertNumber(tabId),
243 {code: 'typeof window.helper'},
244 function(results) {
245 // Assure that code injection was successful.
246 if (chrome.runtime.lastError) {
247 errorCallback(new Error(chrome.runtime.lastError.message));
248 return;
249 } else if (!goog.isDef(results)) {
250 errorCallback(new Error('Content script injection failed.'));
251 return;
252 }
253 if (results[0] === 'object') { // Helper already present.
254 callback();
255 } else {
256 chrome.tabs.executeScript(tabId, {
257 file: 'helper_binary.js'
258 }, function() {
259 setTimeout(function() {
260 callback();
261 }, e2e.ext.utils.HelperProxy.HELPER_INIT_DELAY);
262 });
263 }
264 });
265 });
266
267};

277/**
278 * Helper proxy that connects to a helper injected into a given Webview
279 * element.
280 * To be used in a Chrome App.
281 * @param {boolean} isLookingGlassEnabled Indicates whether the looking glass is
282 * enabled.
283 * @param {!Webview} webview Webview to communicate with.
284 * @constructor
285 * @extends {e2e.ext.utils.HelperProxy}
286 */
287e2e.ext.utils.WebviewHelperProxy = function(isLookingGlassEnabled,
288 webview) {
289 goog.base(this, isLookingGlassEnabled);
290 /**
291 * The Webview element to communicate with.
292 * @type {!Webview}
293 * @private
294 */
295 this.webview_ = webview;
296
297 /**
298 * Object storing response callbacks for API calls in progress
299 * @type {!e2e.ext.utils.CallbacksMap}
300 * @private
301 */
302 this.pendingCallbacks_ = new e2e.ext.utils.CallbacksMap();
303
304 /**
305 * @private
306 * @type {function(*,MessageSender):undefined}
307 */
308 this.boundListener_ = goog.bind(this.helperMessageListener_, this);
309 /**
310 * Handler processing requests from the web application. Set to null to
311 * disable handling web application requests.
312 * @type {?function(!e2e.ext.messages.BridgeMessageRequest)}
313 * @private
314 */
315 this.websiteRequestHandler_ = null;
316};
317goog.inherits(e2e.ext.utils.WebviewHelperProxy, e2e.ext.utils.HelperProxy);

320/** @override */

323 if (goog.isDefAndNotNull(this.webview_)) {

325 }

329/** @override */
330e2e.ext.utils.WebviewHelperProxy.prototype.sendMessageImpl = function(

332 var requestId = this.pendingCallbacks_.addCallbacks(callback, errorCallback);

349/**
350 * Processes message sent by the helper, calling appropriate response callbacks.
351 * @param {*} message Received message.
352 * @param {MessageSender} sender The sender.
353 * @private
354 */
355e2e.ext.utils.WebviewHelperProxy.prototype.helperMessageListener_ = function(
356 message, sender) {
357 if (sender.id !== chrome.runtime.id) {
358 return;
359 }
360 // Request from the web application.
361 if (message.request && goog.isFunction(this.websiteRequestHandler_)) {
362 this.websiteRequestHandler_(
363 /** @type {!e2e.ext.messages.BridgeMessageRequest} */ (message));
364 return;
365 }
366 // Response from the web application.
367 if (!message.requestId ||
368 !this.pendingCallbacks_.containsKey(message.requestId)) {
369 return;
370 }
371 var callbacks = this.pendingCallbacks_.getAndRemove(message.requestId);
372 callbacks.callback(message.response);
373};

376/**
377 * Connects to the helper, injecting it if the connection has not yet been
378 * setup.
379 * @param {!function()} callback The function to invoke once the
380 * connection is established.
381 * @private
382 */
383e2e.ext.utils.WebviewHelperProxy.prototype.connectToHelper_ = function(
384 callback) {
385 this.webview_.executeScript({code: 'typeof window.helper'}, goog.bind(
386 function(results) {
387 if (!chrome.runtime.onMessage.hasListener(this.boundListener_)) {
388 chrome.runtime.onMessage.addListener(this.boundListener_);
389 }
390 if (results[0] === 'object') { // Helper already present.
391 callback();
392 } else {
393 this.webview_.executeScript({
394 file: 'helper_binary.js'
395 }, goog.bind(function() {
396 // NOTE(koto): A workaround - send gmonkeystub to the helper,
397 // as in a Chrome App the helper cannot access chrome-extension://
398 // URLs.
399 setTimeout(goog.bind(this.sendStubToHelper_, this, 'gmonkeystub.js',
400 callback), e2e.ext.utils.HelperProxy.HELPER_INIT_DELAY);
401 }, this));
402 }
403 }, this));
404};

407/**
408 * Send to the Helper the contents of a given stub file to inject into the web
409 * application.
410 * @param {string} stubUrl URL of the stub file.
411 * @param {function()} callback The function to invoke when stub
412 * contents has been sent.
413 * @private
414 */
415e2e.ext.utils.WebviewHelperProxy.prototype.sendStubToHelper_ = function(stubUrl,
416 callback) {
417 goog.net.XhrIo.send(stubUrl, goog.bind(function(ev) {
418 var contents = ev.target.getResponseText();
419 this.webview_.executeScript({code: 'helper.setApiStub(' +
420 JSON.stringify(stubUrl) + ',' + JSON.stringify(contents) + ')'},
421 callback);
422 }, this));
423};
