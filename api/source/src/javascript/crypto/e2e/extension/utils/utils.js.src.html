33/**
34 * Creates a blob URL to download a file.
35 * @param {string} content The content to write to the new file.
36 * @param {!function(string)} callback The callback to invoke with the URL of
37 * the created file.
38 */
39utils.writeToFile = function(content, callback) {
40 var blob = new Blob(
41 [content], {type: 'application/pgp-keys; format=text;'});
42 var url = URL.createObjectURL(blob);
43 callback(url);
44};
45
46
47/**
48 * Reads the contents of the provided file returns it via the provided callback.
49 * Automatically handles both binary OpenPGP packets and text files.
50 * @param {!File} file The file to read.
51 * @param {!function(string)} callback The callback to invoke with the file's
52 * contents.
53 */
54utils.readFile = function(file, callback) {
55 utils.readFile_(false, file, function(contents) {
56 // The 0x80 bit is always set for the Packet Tag for OpenPGP packets.
57 if (contents.charCodeAt(0) >= 0x80) {
58 callback(contents);
59 } else {
60 utils.readFile_(true, file, callback);
61 }
62 });
63};
64
65
66/**
67 * Reads the contents of the provided file as text and returns them via the
68 * provided callback.
69 * @param {boolean} asText If true, then read as text.
70 * @param {!File} file The file to read.
71 * @param {!function(string)} callback The callback to invoke with the file's
72 * contents.
73 * @private
74 */
75utils.readFile_ = function(asText, file, callback) {
76 var reader = new FileReader();
77 reader.onload = function() {
78 if (reader.readyState != reader.LOADING) {
79 reader.onload = null;
80 callback(/** @type {string} */ (reader.result));
81 }
82 };
83 if (asText) {
84 reader.readAsText(file);
85 } else {
86 reader.readAsBinaryString(file);
87 }
88};

115/**
116 * Displays Chrome notifications to the user.
117 * @param {string} msg The message to display to the user.
118 * @param {!function()} callback A callback to invoke when the notification
119 * has been displayed.
120 */
121utils.showNotification = function(msg, callback) {
122 chrome.notifications.create(constants.ElementId.NOTIFICATION_SUCCESS, {
123 type: 'basic',
124 iconUrl: '/images/icon-48.png',
125 title: chrome.i18n.getMessage('extName'),
126 message: msg
127 }, function() {
128 window.setTimeout(function() {
129 chrome.notifications.clear(
130 constants.ElementId.NOTIFICATION_SUCCESS,
131 goog.nullFunction); // Dummy callback to keep Chrome happy.
132 }, constants.NOTIFICATIONS_DELAY);
133 callback();
134 });
135};

138/**
139 * Checks if the given window runs as part of a Chrome App execution
140 * environment.
141 * @param {Window=} opt_window Window object to test. Defaults to current.
142 * @return {boolean} True iff the code runs in a Chrome App.
143 */
144utils.runsInChromeApp = function(opt_window) {
145 var win = opt_window ? opt_window : window;
146 return Boolean(win.chrome.runtime) &&
147 goog.isFunction(win.chrome.runtime.getManifest) &&
148 goog.object.containsKey(win.chrome.runtime.getManifest(), 'app');
149};
150
151
152/**
153 * Checks if the given window runs as part of a Chrome Extension execution
154 * environment.
155 * @param {Window=} opt_window Window object to test. Defaults to current.
156 * @return {boolean} True iff the code runs in a Chrome Extension.
157 */
158utils.runsInChromeExtension = function(opt_window) {
159 var win = opt_window ? opt_window : window;
160 return Boolean(win.chrome.runtime) &&
161 goog.isFunction(win.chrome.runtime.getManifest) &&
162 !goog.object.containsKey(win.chrome.runtime.getManifest(), 'app');
163};
164
165
166/**
167 * Checks if the given window is a Chrome App window. Defaults to checking
168 * current window.
169 * @param {Window=} opt_window Window object to test.
170 * @return {boolean}
171 */
172utils.isChromeAppWindow = function(opt_window) {
173 var win = opt_window ? opt_window : window;
174 return (utils.runsInChromeApp(opt_window) &&
175 win.location.protocol == 'chrome-extension:');
176};
177
178
179/**
180 * Checks if the given window is a view/background page of a Chrome extension.
181 * Defaults to checking current window. Will return false in a content script.
182 * @param {Window=} opt_window Window object to test.
183 * @return {boolean}
184 */
185utils.isChromeExtensionWindow = function(opt_window) {
186 var win = opt_window ? opt_window : window;
187 return (utils.runsInChromeExtension(opt_window) &&
188 win.location.protocol == 'chrome-extension:');
189};

192/**
193 * Returns true if the current execution context is a content script.
194 * @return {boolean}

196utils.isContentScript = function() {
197 return Boolean(chrome.runtime) &&
198 Boolean(chrome.runtime.getURL) && // Running as Chrome extension/app
199 !Boolean(chrome.runtime.getBackgroundPage); // Running in a content script
