35goog.scope(function() {
36var constants = e2e.ext.constants;
37var panels = e2e.ext.ui.panels;
38var templates = e2e.ext.ui.templates.panels.generatekey;
39
40
41
42/**
43 * Constructor for the UI component that provides the form to generate new PGP
44 * keys.
45 * @param {!function(...)} callback The callback to invoke when a new PGP key is
46 * to be generated.
47 * @param {boolean=} opt_hideTitle Optional. A flag to control the display of
48 * the section title. If true, the section title will not be displayed.
49 * Defaults to false.
50 * @param {string=} opt_actionBtnTitle Optional. The title for the action
51 * button. Uses extension defaults if not specified.
52 * @constructor
53 * @extends {goog.ui.Component}
54 */
55panels.GenerateKey = function(callback, opt_hideTitle, opt_actionBtnTitle) {
56 goog.base(this);
57
58 /**
59 * The callback to invoke when a new PGP key is to be generated.
60 * @type {!function(...)}
61 * @private
62 */
63 this.callback_ = callback;
64
65 /**
66 * The title for the generate key section. If empty, it will not be displayed.
67 * @type {string}
68 * @private
69 */
70 this.sectionTitle_ = Boolean(opt_hideTitle) ?
71 '' : chrome.i18n.getMessage('genKeyTitle');
72
73 /**
74 * The title for the action button.
75 * @type {string}
76 * @private
77 */
78 this.actionButtonTitle_ =
79 opt_actionBtnTitle || chrome.i18n.getMessage('genKeyGenerateButtonLabel');
80};
81goog.inherits(panels.GenerateKey, goog.ui.Component);
82
83
84/** @override */
85panels.GenerateKey.prototype.createDom = function() {
86 goog.base(this, 'createDom');
87 this.decorateInternal(this.getElement());
88};
89
90
91/** @override */
92panels.GenerateKey.prototype.decorateInternal = function(elem) {
93 goog.base(this, 'decorateInternal', elem);
94 elem.id = constants.ElementId.GENERATE_KEY_FORM;
95
96 soy.renderElement(elem, templates.generateKeyForm, {
97 sectionTitle: this.sectionTitle_,
98 emailLabel: chrome.i18n.getMessage('genKeyEmailLabel'),
99 commentsLabel: chrome.i18n.getMessage('genKeyCommentsLabel'),
100 actionButtonTitle: this.actionButtonTitle_,
101 signupCancelButtonTitle: chrome.i18n.getMessage('actionCancelPgpAction')
102 });
103};
104
105
106/** @override */
107panels.GenerateKey.prototype.enterDocument = function() {
108 goog.base(this, 'enterDocument');
109
110 var keyboardHandler = new goog.ui.KeyboardShortcutHandler(
111 this.getElementByClass(constants.CssClass.EMAIL));
112 keyboardHandler.registerShortcut('enter', goog.events.KeyCodes.ENTER);
113 this.getHandler().
114 listen(
115 this.getElementByClass(constants.CssClass.ACTION),
116 goog.events.EventType.CLICK,
117 this.generate_).
118 listen(
119 this.getElementByClass(constants.CssClass.CANCEL),
120 goog.events.EventType.CLICK,
121 this.hideSignupForm_).
122 listen(
123 keyboardHandler,
124 goog.ui.KeyboardShortcutHandler.EventType.SHORTCUT_TRIGGERED,
125 this.generate_);
126};

129/**
130 * Generates a new PGP key using the information provided by the user.
131 * @private
132 */
133panels.GenerateKey.prototype.generate_ = function() {
134 var name = '';
135 var email = this.getElementByClass(constants.CssClass.EMAIL).value;
136 var comments = '';
137
138 // TODO(radi): Add a mechanism to allow the user to adjust this.
139 var expDate = Math.floor(new Date('9999/12/31').getTime() / 1e3);
140
141 this.callback_(this, name, email, comments, expDate);
142};

145/**
146 * Resets the key generation form.

148panels.GenerateKey.prototype.reset = function() {
149 var inputs = this.getElement().querySelectorAll('input');
150 goog.array.forEach(inputs, function(input) {
151 input.value = '';
152 });
153};

156/**
157 * Hides the signup form.
158 * @private
159 */
160panels.GenerateKey.prototype.hideSignupForm_ = function() {
161 var signupForm = goog.dom.getElement(
162 e2e.ext.constants.ElementId.GENERATE_KEY_FORM);
163 var cancelButton = goog.dom.getElementByClass(
164 e2e.ext.constants.CssClass.CANCEL, signupForm);
165 var signupPrompt = goog.dom.getElement(
166 e2e.ext.constants.ElementId.SIGNUP_PROMPT);
167 var keyringOptions = goog.dom.getElement(
168 e2e.ext.constants.ElementId.KEYRING_OPTIONS_DIV);
169
170 goog.dom.classlist.add(signupForm, e2e.ext.constants.CssClass.HIDDEN);
171 goog.dom.classlist.add(cancelButton, e2e.ext.constants.CssClass.HIDDEN);
172 goog.dom.classlist.remove(signupPrompt, e2e.ext.constants.CssClass.HIDDEN);
173 goog.dom.classlist.remove(keyringOptions, e2e.ext.constants.CssClass.HIDDEN);
174
175};
