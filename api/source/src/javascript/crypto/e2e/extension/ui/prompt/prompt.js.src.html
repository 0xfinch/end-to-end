234/**
235 * Process the retrieved content blob and display it into the prompt UI.
236 * @param {?messages.BridgeMessageRequest} contentBlob The content that the user
237 * has selected.
238 * @param {constants.Actions=} opt_action Optional. The PGP action to perform.
239 * Defaults to user-specified.
240 * @private
241 */
242ui.Prompt.prototype.processSelectedContent_ =
243 function(contentBlob, opt_action) {
244 this.clearFailure_();
245 var action = opt_action || ext.constants.Actions.USER_SPECIFIED;
246 var content = '';
247 var origin = '';
248 var recipients = [];
249 var canInject = false;
250 if (contentBlob) {
251 if (contentBlob.request) {
252 content = contentBlob.selection;
253 }
254 action = opt_action || contentBlob.action ||
255 utils.text.getPgpAction(content);
256 origin = contentBlob.origin;
257 recipients = contentBlob.recipients || [];
258 contentBlob.action = action;
259 canInject = contentBlob.canInject;
260 contentBlob.subject = contentBlob.subject || undefined;
261 }
262
263 if (!this.pgpLauncher_.hasPassphrase() &&
264 action != ext.constants.Actions.GET_PASSPHRASE) {
265 this.processSelectedContent_(
266 contentBlob, ext.constants.Actions.GET_PASSPHRASE);
267 return;
268 }
269
270 var promptPanel = null;
271 var elem = goog.dom.getElement(constants.ElementId.BODY);
272 var title = this.getElementByClass(constants.CssClass.PROMPT_TITLE);
273 title.textContent = this.getTitle_(action) || title.textContent;
274 switch (action) {
275 case constants.Actions.ENCRYPT_SIGN:
276 promptPanel = new panels.prompt.EncryptSign(
277 this.actionExecutor_,
278 /** @type {!messages.BridgeMessageRequest} */ (contentBlob || {}),
279 this.pgpLauncher_.getPreferences(),
280 goog.bind(this.displayFailure_, this));
281 break;
282 case constants.Actions.DECRYPT_VERIFY:
283 promptPanel = new panels.prompt.DecryptVerify(
284 this.actionExecutor_,
285 /** @type {!messages.BridgeMessageRequest} */ (contentBlob || {}),
286 goog.bind(this.displayFailure_, this));
287 break;
288 case constants.Actions.IMPORT_KEY:
289 promptPanel = new panels.prompt.ImportKey(
290 this.actionExecutor_,
291 /** @type {!messages.BridgeMessageRequest} */ (contentBlob || {}),
292 goog.bind(this.displayFailure_, this));
293 break;
294 case constants.Actions.GET_PASSPHRASE:
295 this.hideHeaderButtons_();
296 this.renderKeyringPassphrase_(elem, contentBlob);
297 return;
298 case constants.Actions.USER_SPECIFIED:
299 this.showMenuInline_(this.renderMenu_(elem, contentBlob));
300 return;
301 case constants.Actions.CONFIGURE_EXTENSION:
302 this.pgpLauncher_.createWindow('settings.html', this.isPopout,
303 goog.nullFunction);
304 return;
305 case constants.Actions.NO_OP:
306 this.close();
307 return;
308 }
309 this.showHeaderButtons_();
310 if (promptPanel) {
311 this.panel_ = promptPanel;
312 this.addChild(promptPanel, false);
313 promptPanel.decorate(elem);
314 title.textContent = promptPanel.getTitle();
315 } else {
316 this.panel_ = null;
317 }
318
319 this.getHandler().listen(
320 elem, goog.events.EventType.CLICK,
321 goog.bind(this.buttonClick_, this, action, origin, contentBlob));
322
323 this.getHandler().listen(
324 elem, goog.events.EventType.KEYDOWN,
325 goog.bind(this.keypressListener_, this));
326
327 if (utils.isChromeExtensionWindow() && !this.isPopout) {
328 this.getHandler().listen(
329 this.getElementByClass(constants.CssClass.POPOUT_BUTTON),
330 goog.events.EventType.CLICK,
331 goog.bind(this.handlePopout_, this)
332 );
333 }
334
335 this.renderMenu_(elem, promptPanel);
336};

373/**
374 * Opens the prompt in a new window.
375 * @param {Event} event The event that triggered the popout action.
376 * @private
377 */
378ui.Prompt.prototype.handlePopout_ = function(event) {
379 if (!utils.isChromeExtensionWindow()) {
380 throw new Error('Popouts are only supported in Chrome extension.');
381 }
382 var boundingRect = this.getElement().getBoundingClientRect();
383 var currentWidth = Math.round(boundingRect.width);
384 var currentHeight = Math.round(boundingRect.height);
385 var currentLeft = window.screenX;
386 var currentTop = window.screenY;
387 var windowBorder = 15;
388 chrome.windows.create({
389 url: 'prompt.html?popout#' + this.helperProxy_.getHelperId(),
390 focused: false,
391 type: 'popup',
392 width: currentWidth,
393 height: currentHeight + windowBorder,
394 left: currentLeft,
395 top: currentTop - windowBorder
396 }, goog.bind(function(win) {
397 setTimeout(goog.bind(function() {
398 var request = /** @type {?messages.BridgeMessageRequest} */ (
399 this.panel_.getContent());
400 var popouts = this.getPopouts_();
401 if (popouts.length > 0) {
402 goog.array.forEach(popouts, function(popout) {
403 // TODO(koto): Change to chrome.tabs.sendMessage(). Maybe.
404 popout.postMessage(request, location.origin);
405 });
406 chrome.windows.update(win.id, {
407 focused: true
408 });
409 this.close();
410 } else {
411 this.displayFailure_(new Error(
412 'Error occurred when connecting to the new window.'));
413 }
414 }, this), 500);
415 }, this));
416};

475/**
476 * Renders the main menu.
477 * @param {Element} elem The element into which the UI elements are to be
478 * rendered.
479 * @param {panels.prompt.PanelBase|messages.BridgeMessageRequest} blob The
480 * prompt UI panel that is displayed to the user or the content blob that
481 * the user has selected.
482 * @return {!goog.ui.PopupMenu} Created menu
483 * @private
484 */
485ui.Prompt.prototype.renderMenu_ = function(elem, blob) {
486 var contentBlob;
487 var displayedAction;
488 if (blob instanceof panels.prompt.PanelBase) {
489 contentBlob = blob.getContent();
490 displayedAction = contentBlob.action;
491 } else {
492 contentBlob = blob || null;
493 }
494
495 var menu = new goog.ui.PopupMenu();
496 goog.array.forEach(this.selectableActions_, function(action) {
497 var menuItem = new goog.ui.MenuItem(action.title);
498 menuItem.setValue(action.value);
499 if (displayedAction == action.value) {
500 menuItem.setEnabled(false);
501 }
502 menu.addChild(menuItem, true);
503 });
504 this.addChild(menu, false);
505 menu.render(goog.dom.getElement(constants.ElementId.BODY));
506
507 this.getHandler().listen(
508 menu,
509 goog.ui.Component.EventType.ACTION,
510 goog.partial(this.selectAction_, contentBlob));
511
512 var menuButton = this.getElementByClass(constants.CssClass.MENU_BUTTON);
513
514 menu.setToggleMode(true);
515 menu.attach(
516 menuButton,
517 goog.positioning.Corner.TOP_LEFT,
518 goog.positioning.Corner.BOTTOM_LEFT);
519
520 return menu;
521};

524/**
525 * Renders the UI elements needed for requesting the passphrase of the PGP
526 * keyring.
527 * @param {Element} elem The element into which the UI elements are to be
528 * rendered.
529 * @param {?messages.BridgeMessageRequest} contentBlob The content that the user
530 * has selected.
531 * @private
532 */
533ui.Prompt.prototype.renderKeyringPassphrase_ = function(elem, contentBlob) {
534 var dialog = new dialogs.Generic(
535 '',
536 goog.bind(function(passphrase) {
537 try {
538 // Correct passphrase entered.
539 this.pgpLauncher_.start(passphrase);
540 if (contentBlob && contentBlob.action) {
541 if (contentBlob.action == constants.Actions.GET_PASSPHRASE) {
542 contentBlob.action = undefined; // Fall into the default action.
543 }
544 this.processSelectedContent_(contentBlob, contentBlob.action);
545 } else {
546 this.close();
547 }
548 } catch (e) { // Incorrect passphrase, so ask again.
549 this.processSelectedContent_(contentBlob,
550 ext.constants.Actions.GET_PASSPHRASE);
551 }
552 goog.dispose(dialog);
553 }, this),
554 // Use a password field to ask for the passphrase.
555 dialogs.InputType.SECURE_TEXT,
556 chrome.i18n.getMessage('actionEnterPassphraseDescription'),
557 chrome.i18n.getMessage('actionEnterPassphrase'),
558 chrome.i18n.getMessage('actionCancelPgpAction'));
559
560 elem.textContent = '';
561 this.addChild(dialog, false);
562 dialog.render(elem);
563 goog.dom.classlist.remove(elem, constants.CssClass.TRANSPARENT);
564};

640/**
641 * Handles external messages to initialize the popout prompt. Will unbind itself
642 * once the message is processed.
643 * @param {!Event} event Incoming event.
644 * @private
645 */
646ui.Prompt.prototype.handleExternalMessage_ = function(event) {
647 if (!this.isPopout ||
648 (event.origin !== location.origin) ||
649 this.wasDecorated()) {
650 return;
651 }
652 var request = /**@type {?e2e.ext.messages.BridgeMessageRequest} */ (
653 event.data);
654 this.injectContent(request);
655 this.decorate(document.body);
656 window.removeEventListener('message', this.boundMessageListener_);
657};
