187/**
188 * Generates a new PGP key using the information that is provided by the user.
189 * @param {ui.panels.GenerateKey} panel The panel where the user has provided
190 * the information for the new key.
191 * @param {string} name The name to use.
192 * @param {string} email The email to use.
193 * @param {string} comments The comments to use.
194 * @param {number} expDate The expiration date to use.
195 * @private
196 */
197ui.Welcome.prototype.generateKey_ =
198 function(panel, name, email, comments, expDate) {
199 var welcomePage = this;
200 var anchorElem = this.genKeyForm_;
201 var defaults = constants.KEY_DEFAULTS;
202 utils.action.getContext(
203 function(/** !e2e.openpgp.ContextImpl */ pgpCtx) {
204 pgpCtx.isKeyRingEncrypted().addCallback(function(isEncrypted) {
205 if (isEncrypted) {
206 window.alert(chrome.i18n.getMessage('settingsKeyringLockedError'));
207 }
208
209 pgpCtx.generateKey(e2e.signer.Algorithm[defaults.keyAlgo],
210 defaults.keyLength, e2e.cipher.Algorithm[defaults.subkeyAlgo],
211 defaults.subkeyLength, name, comments, email, expDate).
212 addCallback(function(key) {
213 var dialog = new dialogs.Generic(
214 chrome.i18n.getMessage('welcomeGenKeyConfirm'),
215 this.hideKeyringSetup_,
216 dialogs.InputType.NONE);
217 this.removeChild(this.genKeyForm_, false);
218 this.addChild(dialog, false);
219 dialog.decorate(this.genKeyForm_.getElement());
220 panel.reset();
221 }, this);
222 }, this);
223 }, this.displayFailure_, this);
224
225 this.keyringMgmt_.refreshOptions();
226};

229/**
230 * Imports a keyring from a file and appends it to the current keyring.
231 * @param {!File} file The file to import.
232 * @private
233 */
234ui.Welcome.prototype.importKeyring_ = function(file) {
235 utils.readFile(file, goog.bind(function(contents) {
236 this.actionExecutor_.execute(/** @type {!messages.ApiRequest} */ ({
237 action: constants.Actions.IMPORT_KEY,
238 content: contents,
239 passphraseCallback: goog.bind(this.renderPassphraseCallback_, this)
240 }), this, goog.bind(function(res) {
241 var dialog = new dialogs.Generic(
242 chrome.i18n.getMessage('welcomeKeyImport'),
243 this.hideKeyringSetup_,
244 dialogs.InputType.NONE);
245 this.removeChild(this.keyringMgmt_, false);
246 this.addChild(dialog, false);
247 dialog.decorate(this.keyringMgmt_.getElement());
248 }, this));
249 }, this));
250};

253/**
254 * Updates the passphrase to the existing keyring.
255 * @param {string} passphrase The new passphrase to apply.
256 * @private
257 */
258ui.Welcome.prototype.updateKeyringPassphrase_ = function(passphrase) {
259 utils.action.getContext(
260 function(/** !e2e.openpgp.ContextImpl */ pgpCtx) {
261 pgpCtx.changeKeyRingPassphrase(passphrase).addCallback(function() {
262 var dialog = new dialogs.Generic(
263 chrome.i18n.getMessage('keyMgmtChangePassphraseSuccessMsg'),
264 goog.bind(function() {
265 this.removeChild(dialog, false);
266 this.keyringMgmt_ = new ui.panels.KeyringMgmtMini(
267 goog.nullFunction,
268 goog.bind(this.importKeyring_, this),
269 goog.bind(this.updateKeyringPassphrase_, this));
270 this.addChild(this.keyringMgmt_, false);
271 pgpCtx.isKeyRingEncrypted().addCallback(function(isEncrypted) {
272 this.keyringMgmt_.decorate(dialog.getElement());
273 this.keyringMgmt_.setKeyringEncrypted(isEncrypted);
274 }, this);
275 }, this),
276 dialogs.InputType.NONE);
277 this.removeChild(this.keyringMgmt_, false);
278 this.addChild(dialog, false);
279 dialog.decorate(this.keyringMgmt_.getElement());
280 }, this);
281 }, this.displayFailure_, this);
282};

285/**
286 * Renders the UI elements needed for requesting the passphrase of an individual
287 * PGP key.
288 * @param {string} uid The UID of the PGP key.
289 * @return {!e2e.async.Result<string>} A promise resolved with the user-provided
290 * passphrase.
291 * @private
292 */
293ui.Welcome.prototype.renderPassphraseCallback_ = function(uid) {
294 var result = new e2e.async.Result();
295 var popupElem = goog.dom.getElement(constants.ElementId.CALLBACK_DIALOG);
296 var dialog = new dialogs.Generic(chrome.i18n.getMessage(
297 'promptPassphraseCallbackMessage', uid),
298 function(passphrase) {
299 goog.dispose(dialog);
300 result.callback(/** @type {string} */ (passphrase));
301 },
302 // Use a password field to ask for the passphrase.
303 dialogs.InputType.SECURE_TEXT,
304 '',
305 chrome.i18n.getMessage('actionEnterPassphrase'),
306 chrome.i18n.getMessage('actionCancelPgpAction'));
307
308 this.addChild(dialog, false);
309 dialog.render(popupElem);
310 return result;
311};
