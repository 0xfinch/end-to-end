17/**
18 * @fileoverview Encrypts and signs a message using the provided recipients and
19 * signer information.
20 */

31var utils = e2e.ext.utils;

36 * Constructor for the action.

38 * @implements {e2e.ext.actions.Action.<string, string>}

40actions.EncryptSign = function() {};

43/** @inheritDoc */
44actions.EncryptSign.prototype.execute =
45 function(ctx, request, requestor, callback, errorCallback) {
46 var recipients = request.recipients || [];
47 var passphrases = request.encryptPassphrases || [];
48 var currentUser = request.currentUser || '';
49
50 if (currentUser &&
51 (recipients.length > 0 || passphrases.length > 0) &&
52 !goog.array.contains(recipients, currentUser)) {
53 // If encrypting the message, always add the sender key for him
54 // to be able to decrypt.
55 recipients.push(currentUser);
56 }
57
58 this.getEncryptKeys_(ctx, recipients, function(encryptionKeys) {
59 if (passphrases.length == 0 &&
60 encryptionKeys.length == 0 &&
61 recipients.length > 0) {
62 errorCallback(new utils.Error(
63 'No pasphrases nor keys available to encrypt.',
64 'promptNoEncryptionTarget'));
65 return;
66 }
67
68 ctx.searchPrivateKey(currentUser).addCallback(function(privKeys) {
69 var signingKey = null;
70 if (request.signMessage && privKeys && privKeys.length > 0) {
71 // Just choose one private key for now.
72 signingKey = privKeys[0];
73 }
74
75 ctx.encryptSign(
76 request.content,
77 [], // Options.
78 encryptionKeys, // Keys to encrypt to.
79 passphrases, // For symmetrically-encrypted session keys.
80 signingKey // Key to sign with.
81 ).addCallback(callback).addErrback(errorCallback);
82 }).addErrback(errorCallback);
83 }, errorCallback);
84};

87/**
88 * Parses string and looks up keys for encrypting objects in keyring.
89 * @param {!e2e.openpgp.ContextImpl} ctx A PGP context that can be used to
90 * complete the action.
91 * @param {!Array.<string>} recipients A list of recipients to get keys for.
92 * @param {!function(!Array.<!e2e.openpgp.Key>)} callback The callback where the
93 * resulting array of key objects is to be passed.
94 * @param {!function(Error)} errorCallback A callback where errors will be
95 * passed to.
96 * @private
97 */
98actions.EncryptSign.prototype.getEncryptKeys_ =
99 function(ctx, recipients, callback, errorCallback) {
100 var keys = [];
101 var userIds = goog.array.slice(recipients, 0, recipients.length);
102
103 var queryFunc = function() {
104 if (goog.array.isEmpty(userIds)) {
105 callback(keys);
106 } else {
107 var userId = goog.string.trim(userIds.pop());
108 if (userId) {
109 ctx.searchPublicKey(userId).addCallback(function(found) {
110 if (found) {
111 goog.array.extend(keys, found);
112 }
113 queryFunc();
114 }).addErrback(errorCallback);
115 }
116 }
117 };
118 queryFunc();
119};
