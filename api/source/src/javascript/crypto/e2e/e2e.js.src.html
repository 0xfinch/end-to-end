60/**
61 * Converts a non-negative number into a big-endian ByteArray.
62 * @param {!number} value The number to convert.
63 * @return {!e2e.ByteArray} The number as a big-endian ByteArray.
64 */
65e2e.numberToByteArray = function(value) {
66 if (value < 0) {
67 throw new e2e.error.InvalidArgumentsError(
68 'Cannot convert negative number to a byte array.');
69 }
70
71 var byteArray = [];
72 do {
73 byteArray.unshift(value & 0xff);
74 value >>>= 8;
75 } while (value > 0);
76 return byteArray;
77};

80/**
81 * Turns an array of big endian 32 bits numbers into a byte array.
82 * @see #byteArrayToDwordArray
83 * @param {!e2e.DwordArray} dwords DwordArray to transform to ByteArray.
84 * @return {!e2e.ByteArray} ByteArray with a length divisible by 4.
85 */
86e2e.dwordArrayToByteArray = function(dwords) {
87 var byteArray = [];
88 goog.array.forEach(dwords, function(dword) {
89 byteArray.push((dword >>> 24) & 0xff);
90 byteArray.push((dword >>> 16) & 0xff);
91 byteArray.push((dword >>> 8) & 0xff);
92 byteArray.push((dword) & 0xff);
93 });
94 return byteArray;
95};

98/**
99 * Turns an array of numbers into a big endian 32 bit numbers array.
100 * @see #dwordArrayToByteArray
101 * @param {!e2e.ByteArray} bytes ByteArray with a length divisible by 4.
102 * @return {!e2e.DwordArray} The resulting dword array.
103 */
104e2e.byteArrayToDwordArray = function(bytes) {
105 var dwordArray = [];
106 for (var i = 0; i < bytes.length; i += 4) {
107 dwordArray.push((bytes[i + 3] |
108 (bytes[i + 2] << 8) |
109 (bytes[i + 1] << 16) |
110 (bytes[i] << 24)) >>> 0);
111 }
112 return dwordArray;
113};

126/**
127 * Converts a big-endian byte array into a number. Due to memory limits of the
128 * bitwise operators and number type, the byte array should be at most 4 bytes.
129 * @param {!e2e.ByteArray} bytes A big-endian byte array.
130 * @return {number} The resulting number.
131 */
132e2e.byteArrayToNumber = function(bytes) {
133 if (bytes.length > 4) {
134 throw new e2e.error.InvalidArgumentsError(
135 'Cannot convert byte array exceeding 4 bytes to a number.');
136 }
137
138 var result = 0;
139 for (var i = 0; i < bytes.length; i++) {
140 result <<= 8;
141 result |= bytes[i];
142 }
143 return result;
144};

167/**
168 * Converts a byte array into a JS string.
169 * @param {!e2e.ByteArray|!Uint8Array} bytes The bytes to convert.
170 * @param {string=} opt_charset The charset to try (defaults to UTF-8).
171 * @return {string} The string representation of bytes.
172 */
173e2e.byteArrayToString = function(bytes, opt_charset) {
174 if (e2e.USE_TEXT_DECODER) {
175 var td = /** @type {{decode: function(Uint8Array):string}} */ (
176 new goog.global['TextDecoder'](opt_charset || 'utf-8'));
177 return td.decode(new Uint8Array(bytes));
178 } else {
179 return goog.crypt.utf8ByteArrayToString(bytes);
180 }
181};

184/**
185 * Converts a byte array into a JS string asynchronously.
186 * @param {!e2e.ByteArray} bytes The bytes to convert.
187 * @param {string=} opt_charset The charset to try (defaults to UTF-8).
188 * @return {!e2e.async.Result.<string>} The string representation of bytes.
189 */
190e2e.byteArrayToStringAsync = function(bytes, opt_charset) {
191 if (e2e.USE_TEXT_DECODER) {
192 return e2e.async.Result.toResult(
193 e2e.byteArrayToString(bytes, opt_charset));
194 } else {
195 var res = new e2e.async.Result;
196 var fr = new FileReader;
197 fr.onload = function() {
198 res.callback(fr.result);
199 };
200 fr.onerror = function(e) {
201 res.errback(new Error(String(e)));
202 };
203 fr.readAsText(
204 new Blob([new Uint8Array(bytes)]), opt_charset || 'utf-8');
205 return res;
206 }
207};

210/**
211 * Converts a string into a UTF-8 encoded byte array.
212 * Throws {@code e2e.error.InvalidArgumentsError} if the string
213 * contains unencodable codepoints (eg: surrogate characters.)
214 * @param {string} stringInput The string to convert.
215 * @return {!e2e.ByteArray} The UTF-8 byte representation of the string.
216 */
217e2e.stringToByteArray = function(stringInput) {
218 // We don't use the Closure implementation as it normalizes line ends to \n.
219 var out = [], p = 0;
220 for (var i = 0; i < stringInput.length; i++) {
221 var c = stringInput.charCodeAt(i);
222 // If present, convert surrogate pairs to Unicode code points
223 // From Section 3.7 of the Unicode Standard:
224 // If <H,L> is a surrogate pair,
225 // N = (H - 0xD800)*0x400 + (L - 0xDC00) + 0x10000
226 // See:
227 // http://unicode.org/versions/Unicode3.0.0/ch03.pdf
228 // and the String.charCodeAt() documentation
229 // http://goo.gl/d4oFPv
230 if (e2e.isHighSurrogate_(c)) {
231 i++;
232 // charCodeAt() returns NaN if i >= stringInput.length
233 var low = stringInput.charCodeAt(i);
234 if (isNaN(low) || !e2e.isLowSurrogate_(low)) {
235 throw new e2e.error.InvalidArgumentsError(
236 'Cannot encode string to utf-8.');
237 }
238 c = ((c - 0xd800) * 0x400) + (low - 0xdc00) + 0x10000;
239 }
240 else if (e2e.isLowSurrogate_(c)) {
241 throw new e2e.error.InvalidArgumentsError(
242 'Cannot encode string to utf-8.');
243 }
244
245 // Convert a code point into utf-8
246 // See:
247 // http://tools.ietf.org/html/rfc3629#section-3
248 // for the encoding procedure.
249 if (c <= 0x7f) {
250 // one byte encoding
251 // 0xxxxxxx
252 out[p++] = c;
253 } else if (c <= 0x7ff) {
254 // two byte encoding
255 // 110xxxxx 10xxxxxx
256 out[p++] = (c >> 6) | 0xc0;
257 out[p++] = (c & 0x3f) | 0x80;
258 } else if (c <= 0xffff) {
259 // three byte encoding
260 // 1110xxxx 10xxxxxx 10xxxxxx
261 out[p++] = (c >> 12) | 0xe0;
262 out[p++] = ((c >> 6) & 0x3f) | 0x80;
263 out[p++] = (c & 0x3f) | 0x80;
264 } else if (c <= 0x10ffff) {
265 // four byte encoding
266 // 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
267 out[p++] = ((c >> 18) & 0x7) | 0xf0;
268 out[p++] = ((c >> 12) & 0x3f) | 0x80;
269 out[p++] = ((c >> 6) & 0x3f) | 0x80;
270 out[p++] = (c & 0x3f) | 0x80;
271 } else {
272 throw new e2e.error.InvalidArgumentsError(
273 'Cannot encode character codes > 0x10ffff');
274 }
275 }
276 return out;
277};

300/**
301 * Returns whether the number is a valid 'byte' (0-255 no decimals).
302 * @param {number} b The number to test.
303 * @return {boolean} If the send number is a byte.
304 */
305e2e.isByte = function(b) {
306 return (typeof b == 'number' &&
307 b >= 0 &&
308 b <= 255 &&
309 b - Math.floor(b) == 0);

313/**
314 * Verifies a given ByteArray is indeed made of bytes.
315 * @param {!e2e.ByteArray} bytes The bytearray to test.
316 * @return {boolean} If the array if a byteArray.
317 */
318e2e.isByteArray = function(bytes) {
319 var yes = 1;
320 for (var i = 0; i < bytes.length; i++) {
321 yes &= e2e.isByte(bytes[i]) | 0;
322 }
323 return yes == 1;
324};

327/**
328 * Does near constant time ByteArray comparison.
329 * @param {!e2e.ByteArray} ba1 The first bytearray to check.
330 * @param {!e2e.ByteArray} ba2 The second bytearray to check.
331 * @return {boolean} If the array are equal.
332 */
333e2e.compareByteArray = function(ba1, ba2) {
334 if (ba1.length !== ba2.length) {
335 return false;
336 }
337 if (!e2e.isByteArray(ba1) || !e2e.isByteArray(ba2)) {
338 return false;
339 }
340 var yes = 1;
341 for (var i = 0; i < ba1.length; i++) {
342 yes &= !(ba1[i] ^ ba2[i]) | 0;
343 }
344 return yes == 1;
345};

365/**
366 * Increments (++ByteArray) an unsigned big endian in a ByteArray.
367 * @template T
368 * @param {T} n The number to increment.
369 * @return {T} The incremented array.
370 */
371e2e.incrementByteArray = function(n) {
372 /**
373 * Inner function to allow static type checking.
374 * @param {(!e2e.ByteArray|!Uint8Array)} n The number to increment.
375 * @return {(!e2e.ByteArray|!Uint8Array)} The incremented array.
376 */
377 var fn = function(n) {
378 var carry = 1; // initial increment
379 for (var i = n.length - 1; i >= 0; --i) {
380 n[i] += carry;
381 carry = (n[i] & 0x100) >>> 8;
382 n[i] &= 0xff;
383 }
384 return n;
385 };
386 return fn(n);
387};

391/**
392 * A ImmutableArray is a collection that just offers indexed access to its
393 * elements, and a limited set of (static) operations that do not
394 * mutate the collection. It can also store an optional "state" object
395 * that may be retrieved later on. The reference to this state cannot
396 * be modified through public methods.
397 *
398 * @param {!e2e.ImmutableArray<T>|!Array<T>|!goog.array.ArrayLike} elements
399 * @param {S=} opt_state opaque data that may be retrieved via the
400 * getState() method.
401 * @template T,S
402 * @constructor
403 */
404e2e.ImmutableArray = function(elements, opt_state) {
405
406 /**
407 * An underlying array which holds elements in our collection.
408 * @type {!Array<T>}
409 * @private
410 */
411 this.elements_ = (elements instanceof e2e.ImmutableArray) ?
412 goog.array.toArray(elements.elements_) :
413 goog.array.toArray(elements);
414
415 /**
416 * An optional opaque state object that can be stored along
417 * with this array.
418 * @type {S|undefined}
419 * @private
420 */
421 this.state_ = opt_state;
422};

453/**
454 * Returns a new ImmutableArray after appending an element to
455 * a provided ImmutableArray. If the source array is null or
456 * undefined, it will return an array containing just the
457 * element. Note that the returned ImmutableArray will have
458 * its state set to undefined.
459 *
460 * @param {e2e.ImmutableArray<T>} arr the source array.
461 * @param {!T} element the element to append to the collection.
462 * @return {!e2e.ImmutableArray<T>} the new ImmutableArray
463 * @template T
464 */
465e2e.ImmutableArray.pushCopy = function(arr, element) {
466 if (goog.isDefAndNotNull(arr)) {
467 return new e2e.ImmutableArray(
468 goog.array.concat(arr.elements_, element));
469 } else {
470 return new e2e.ImmutableArray([element]);
471 }

475/**
476 * Create a new ImmutableArray after removing/adding some elements from a
477 * provided ImmutableArray. Note that the state in the returned ImmutableArray
478 * is undefined.
479 * @param {!e2e.ImmutableArray<T>} arr the source array.
480 * @param {number} index The index at which to start changing the array.
481 * @param {number} howMany How many elements to remove (0 means no removal.)
482 * @param {...T} var_args Optional, additional elements to insert into the
483 * array.
484 * @return {!e2e.ImmutableArray<T>} the result ImmutableArray.
485 * @template T
486 */
487e2e.ImmutableArray.spliceCopy = function(arr, index, howMany, var_args) {
488 // swap in a clone before passing it to the destructive splice operator.
489 var elements = goog.array.clone(arr.elements_);
490 arguments[0] = elements;
491 goog.array.splice.apply(null, arguments);
492 return new e2e.ImmutableArray(elements);
493};

496/**
497 * Retuns a new ImmutableArray that is the concatenation
498 * all its arguments. Note that the state in the returned ImmutableArray is
499 * undefined.

513/**
514 * Calls a function for each element in the provided ImmutableArray. Skips
515 * holes in the array.
516 *
517 * @param {e2e.ImmutableArray<T>} sarray ImmutableArray over which to iterate.
518 * @param {function(this: S, T, number): ?} f The function to call for every
519 * element. This function takes 2 arguments (the element and the index).
520 * The return value is ignored.
521 * @param {S=} opt_obj The object to be used as the value of 'this' within f.
522 * @template T,S
523 */
524e2e.ImmutableArray.forEach = function(sarray, f, opt_obj) {
525 // We don't use goog.array.forEach directly, as it potentially
526 // offers access to the underlying array.
527 var arr = sarray.elements_;
528 var len = arr.length;
529 for (var i = 0; i < len; i++) {
530 if (i in arr) {
531 f.call(opt_obj, arr[i], i);
532 }
533 }
