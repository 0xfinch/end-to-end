34/**
35 * Diffie-Hellman generator.
36 * @constructor
37 * @param {!e2e.ByteArray|!Uint8Array} p The modulus.
38 * @param {!e2e.ByteArray} g The base.
39 */
40e2e.cipher.DiffieHellman = function(p, g) {
41 this.p_ = new e2e.BigNumModulus(p);
42 this.g_ = g;
43 this.x_ = this.generateExponent_();
44};

47/**
48 * Caches the result of g^x for future use.
49 * @type {!e2e.ByteArray} g^x.

51e2e.cipher.DiffieHellman.prototype.cachedResult_;

54/**
55 * Computes g^x mod p.
56 * @param {!e2e.ByteArray=} opt_g The base.
57 * @return {!e2e.ByteArray} g^x mod p.
58 */
59e2e.cipher.DiffieHellman.prototype.generate = function(opt_g) {
60 if (opt_g) {
61 goog.asserts.assert(this.isValidBase(opt_g));
62 return this.p_.pow(opt_g, this.x_);
63 }
64 if (!this.cachedResult_) {
65 goog.asserts.assert(this.isValidBase(this.g_));
66 this.cachedResult_ = this.p_.pow(this.g_, this.x_);
67 }
68 return this.cachedResult_;
69};

72/**
73 * Computes a safe exponent for Diffie-Hellman.
74 * @private
75 * @return {!e2e.ByteArray} The generated exponent.
76 */
77e2e.cipher.DiffieHellman.prototype.generateExponent_ = function() {
78 do {
79 var res = e2e.random.getRandomBytes(Math.ceil(this.p_.getBitLength() / 8));
80 var check = this.p_.pow(this.g_, res);
81 } while (e2e.compareByteArray(check, [1]));
82 return res;
83};

86/**
87 * Checks if a base is valid.
88 * @param {!e2e.ByteArray|!Uint8Array} base The base to check.
89 * @return {boolean} Whether the base is valid.
90 */
91e2e.cipher.DiffieHellman.prototype.isValidBase = function(base) {
92 var n = new e2e.BigNum(base);
93 var two = new e2e.BigNum([2]);
94 return n.compare(two) >= 0 && n.compare(this.p_.subtract(two)) <= 0;
95};
