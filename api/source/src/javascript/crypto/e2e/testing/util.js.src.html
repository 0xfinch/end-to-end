35/**
36 * Runs performance tests.
37 *
38 * The argument to this method should be an array of benchmark tests. Each
39 * element of the array is of the form {benchmark: <function>, label:name}
40 *
41 * @param {!Array.<{benchmark:!function():*, label:string}>} benchmarks
42 * An array consisting of things we want to time, and the name to show
43 * for it in the performance table.
44 * @param {number=} opt_numSamples Number of times to run the test functions.
45 * Defaults to 10.
46 * @param {number=} opt_timeoutInterval Number of milliseconds after which the
47 * test is to be aborted. Defaults to 5000.
48 * @return {!goog.Promise.<!Object.<!string,!Object>>} Performance stats
49 * promise. Promise will be resolved when last test completes.
50 */
51e2e.testing.Util.runPerfTests = function(benchmarks, opt_numSamples,
52 opt_timeoutInterval) {
53 var timer = new goog.testing.PerformanceTimer(opt_numSamples,
54 opt_timeoutInterval);
55 var doc;
56 try {
57 doc = goog.dom.getDocument();
58 } catch (e) {}
59 var allResults = {};
60 var deferredResults = [];
61 var displayResultsFun;
62 if (doc) { // We run in a browser.
63 displayResultsFun = e2e.testing.Util.createPerformanceTable(timer);
64 }
65
66 var processTaskResults = function(label, taskResults) {
67 allResults[label] = taskResults;
68 if (goog.isFunction(displayResultsFun)) {
69 displayResultsFun(label, taskResults);
70 }
71 };
72
73 goog.array.forEach(benchmarks, function(benchmark) {
74 var task = new goog.testing.PerformanceTimer.Task(benchmark.benchmark);
75 if (benchmark.async) {
76 var deferredResult = timer.runAsyncTask(task);
77 deferredResult.then(goog.bind(processTaskResults, this,
78 benchmark.label));
79 deferredResults.unshift(deferredResult);
80 } else {
81 processTaskResults(benchmark.label, timer.runTask(task));
82 }
83 });
84
85 return new goog.Promise.all(deferredResults).then(function() {
86 return allResults;
87 });
88};

91/**
92 * Creates performance table in the DOM to display the results in.
93 * @param {goog.testing.PerformanceTimer} timer Performance timer to use.
94 * @return {?function(!string,!Object)} callback function filling the results
95 * in the created table
96 */
97e2e.testing.Util.createPerformanceTable = function(timer) {
98 var body = goog.dom.getDocument().body;
99 var perfTable = goog.dom.createElement('div');
100 goog.dom.appendChild(body, perfTable);
101 var table = new goog.testing.PerformanceTable(perfTable, timer);
102 return function(label, taskResults) {
103 table.recordResults(taskResults, label);
104 };
105};

108/**
109 * Adds a benchmark function to performance test array.
110 * Benchmark function will be executed repeatedly and the execution time will be
111 * measured. Return values and errors thrown from the function are ignored.
112 * @param {!Array.<!{benchmark:!function():?goog.async.Deferred, label:string,
113 * async: boolean}>} benchmarks Array to add the benchmark to.
114 * @param {!function():?goog.async.Deferred} fun Benchmark test to add. The
115 * function execution time will be measured. Iff opt_async is true,
116 * the function should return a deferred and execute deferred callback()
117 * when the test is complete.
118 * @param {string} label Benchmark test label
119 * @param {boolean=} opt_async Iff true, the test is asynchronous.
120 */
121e2e.testing.Util.addBenchmark = function(benchmarks, fun, label, opt_async) {
122 benchmarks.push({
123 benchmark: fun,
124 label: label,
125 async: Boolean(opt_async)
126 });
127};

130/**
131 * Checks if the test URL has been called with runPerf parameter. To be used in
132 * performance test HTML files in shouldRunTests function.
133 * @return {boolean} True if runPerf is present in the URL, false otherwise.
134 */
135e2e.testing.Util.hasRunPerfParameter = function() {
136 if (goog.dom.getDocument().location.href.indexOf('runPerf') !== -1) {
137 return true;
138 }
139 console.log('Add runPerf parameter to the URL to run this performance test.');
140 return false;
141};
