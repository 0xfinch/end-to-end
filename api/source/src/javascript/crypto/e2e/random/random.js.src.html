79/**
80 * @param {number} size The number of bytes to generate.
81 * @return {?e2e.ByteArray} A list of size random bytes.
82 * @private
83 */
84e2e.random.tryGetWebCryptoRandomInternal_ = function(size) {
85 var cryptoObject = e2e.random.getWebCryptoObject_();
86 if (cryptoObject) {
87 var array = new Uint8Array(size);
88 cryptoObject['getRandomValues'](array);
89 return goog.array.clone(array);
90 }
91 return null;
92};

95/**
96 * The security of this random number is based on HMAC-SHA256 being a
97 * pseudorandom function. As such, it should be impossible to guess from
98 * the output of it the message that was hashed.
99 * @return {!e2e.ByteArray} RNG_INTERNAL_BYTES_ of random bytes.
100 * @private
101 */
102e2e.random.getRandomBytesInternal_ = function() {
103 var webCryptoRandom = e2e.random.tryGetWebCryptoRandomInternal_(
104 e2e.random.RNG_INTERNAL_BYTES_);
105 if (webCryptoRandom) {
106 return webCryptoRandom;
107 }
108 if (!e2e.random.initialized_) {
109 if (e2e.random.seed_.length < e2e.random.RNG_INTERNAL_BYTES_) {
110 throw new Error('Seed is too small.');
111 }
112 e2e.random.state_ = e2e.random.seed_.slice(
113 0, e2e.random.RNG_INTERNAL_BYTES_);
114 // Destroy the seed.
115 for (var i = 0; i < e2e.random.seed_.length; ++i) {
116 e2e.random.seed_[i] = 0;
117 }
118 e2e.random.hasher_ = new goog.crypt.Sha256();
119 e2e.random.initialized_ = true;
120 }
121 var hmacer = new goog.crypt.Hmac(e2e.random.hasher_, e2e.random.state_);
122
123 // The new state is HMAC-SHA256(key=state, 0).
124 e2e.random.state_ = hmacer.getHmac([0]).slice(
125 0, e2e.random.RNG_INTERNAL_BYTES_);
126 // The result is HMAC-SHA256(key=state, 1).
127 var res = hmacer.getHmac([1]);
128 if (goog.isNull(res)) {
129 throw new Error('HMAC returned a null result.');
130 }
131 return /** @type {!e2e.ByteArray} */ (res);
132};

135/**
136 * Get WebCrypto object, if it's available
137 * @return {?Object}
138 * @private

140e2e.random.getWebCryptoObject_ = function() {
141 if (e2e.random.USE_WEB_CRYPTO &&
142 (typeof goog.global['crypto'] == 'object' ||
143 typeof goog.global['msCrypto'] == 'object')) {
144 return goog.global['crypto'] || goog.global['msCrypto'];
145 }
146 return null;
147};

150/**
151 * @param {number} size The number of bytes to generate.
152 * @param {Array.<number>=} opt_blacklist A list of bytes to avoid generating.
153 * @return {!e2e.ByteArray} A list of size random bytes.
154 */
155e2e.random.getRandomBytes = function(size, opt_blacklist) {
156 var random = [];
157 var blacklist = opt_blacklist || [];
158 if (blacklist.length > 0) {
159 while (random.length < size) {
160 goog.array.extend(random, goog.array.filter(
161 e2e.random.getRandomBytesInternal_(), function(elem) {
162 return blacklist.indexOf(elem) < 0;
163 }));
164 }
165 } else { // optimize
166 var webCryptoRandom = e2e.random.tryGetWebCryptoRandomInternal_(size);
167 if (webCryptoRandom) {
168 return webCryptoRandom;
169 }
170 while (random.length < size) {
171 goog.array.extend(random, e2e.random.getRandomBytesInternal_());
172 }
173 }
174 return random.slice(0, size);
175};

178/**
179 * The size of the seed is the size of the biggest data sent in one call.
180 * @param {!Array.<number>} bytes The random bytes to add to the pool.
181 */
182e2e.random.seedRandomBytes = function(bytes) {
183 e2e.random.seed_.splice(0, bytes.length);
184 goog.array.extend(e2e.random.seed_, bytes);
185};

188/**
189 * Adds the specified numbers of bytes into the seed from the WebCrypto random.
190 * @param {number} size The number of bytes to add to the entropy pool.
191 */
192e2e.random.seedRandomBytesWebCrypto = function(size) {
193 var array = new Uint8Array(size);
194 var cryptoObject = goog.global['crypto'] || goog.global['msCrypto'];
195 cryptoObject['getRandomValues'](array);
196 e2e.random.seedRandomBytes(goog.array.clone(array));
197};
