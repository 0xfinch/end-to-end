163/**
164 * Contains a list of all registered implementations for each algorithm.
165 * @type {!Object.<e2e.cipher.Algorithm,
166 * function(new:e2e.Algorithm, !e2e.cipher.Algorithm,
167 * e2e.cipher.key.Key=)>}
168 * @private
169 */
170e2e.cipher.factory.ciphers_ = {};

173/**
174 * Registers a class for a specific algorithm.
175 * @param {function(new:e2e.Algorithm, !e2e.cipher.Algorithm,
176 * e2e.cipher.key.Key=)} cipher The implementation.
177 * @param {e2e.cipher.Algorithm=} opt_algorithm The name to register
178 * the cipher to if different from the cipher's prototype algorithm.
179 */
180e2e.cipher.factory.add = function(cipher, opt_algorithm) {
181 var algorithm = opt_algorithm || cipher.prototype.algorithm;
182 var current = e2e.cipher.factory.ciphers_[algorithm];
183 if (!goog.isDef(current)) {
184 e2e.cipher.factory.ciphers_[algorithm] = cipher;
185 }
186};

199/**
200 * Returns an instance of the required cipher, or null if not available.
201 * @param {!e2e.cipher.Algorithm} algorithm The algorithm to retrieve.
202 * @param {e2e.cipher.key.Key=} opt_keyData The key to use (public or
203 * private).
204 * @return {?e2e.cipher.Cipher} The cipher instance requested.
205 */
206e2e.cipher.factory.get = function(algorithm, opt_keyData) {
207 if (e2e.cipher.factory.ciphers_.hasOwnProperty(algorithm)) {
208 var constructor = e2e.cipher.factory.ciphers_[algorithm];
209 return /** @type {e2e.cipher.Cipher}*/ (
210 new constructor(algorithm, opt_keyData));
211 } else {
212 return null;
213 }
214};

226/**
227 * Returns an instance of the required cipher, or throws if not available.
228 * @param {!e2e.cipher.Algorithm} algorithm The algorithm to retrieve.
229 * @param {e2e.cipher.key.Key=} opt_keyData The key to use (public or
230 * private).
231 * @return {!e2e.cipher.Cipher} The cipher instance requested.
232 */
233e2e.cipher.factory.require = function(algorithm, opt_keyData) {
234 var ret = e2e.cipher.factory.get(algorithm, opt_keyData);
235 if (goog.isNull(ret)) {
236 throw new e2e.cipher.Error('Required algorithm not available: ' +
237 algorithm);
238 }
239 return ret;
240};
