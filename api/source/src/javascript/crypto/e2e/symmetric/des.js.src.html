93/**
94 * Basic implementation of DES.
95 * @param {e2e.cipher.Algorithm} algorithm The algorithm being
96 * implemented.
97 * @param {e2e.cipher.key.Key=} opt_keyObj The key to use.
98 * @implements {e2e.cipher.SymmetricCipher}
99 * @extends {e2e.AlgorithmImpl}
100 * @constructor
101 */
102e2e.cipher.Des = function(algorithm, opt_keyObj) {
103 /**
104 * 16 subkeys, each 56 bits, but stored as 64-bit values.
105 * @type {Array.<goog.math.Long>}
106 * @private
107 */
108 this.subkeys_ = [];
109 goog.base(this, algorithm, opt_keyObj);

115e2e.cipher.Des.prototype.blockSize = 8; // 64 bits.

118/** @inheritDoc */
119e2e.cipher.Des.prototype.setKey = function(keyObj) {
120 goog.base(this, 'setKey', keyObj, keyObj.key.length);
121 this.keyExpansion_();

125/**
126 * Generates subkeys from this.key, stores in this.subkeys_.
127 * @private
128 */
129e2e.cipher.Des.prototype.keyExpansion_ = function() {
130 var permutedKey = this.permuteBlock_(this.key.key,
131 e2e.cipher.Des.permutedChoice1);
132 var leftKeys = this.keyRotate_(permutedKey.shiftRightUnsigned(28).toInt());
133 var rightKeys = this.keyRotate_(permutedKey.toInt() & 0x0fffffff);
134 for (var i = 0; i < 16; i++) {
135 var blockInt = goog.math.Long.fromInt(leftKeys[i]).shiftLeft(28).or(
136 goog.math.Long.fromInt(rightKeys[i]));
137 var block = e2e.longToByteArray(blockInt);
138 block.shift(); // remove first byte, so it's 56bits instead of 64bits
139 this.subkeys_[i] = this.permuteBlock_(block,
140 e2e.cipher.Des.permutedChoice2);
141 }
142};

145/**
146 * @param {number} key 32-bit value.
147 * @return {!e2e.ByteArray} 16 keys, each 28 bits.
148 * @private
149 */
150e2e.cipher.Des.prototype.keyRotate_ = function(key) {
151 var keys = [];
152 var previous = key;
153 for (var i = 0; i < 16; i++) {
154 previous = keys[i] = (
155 ((previous << e2e.cipher.Des.leftShifts[i]) & 0x0fffffff) |
156 ((previous << 4) >>> (32 - e2e.cipher.Des.leftShifts[i])));
157 }
158 return keys;
159};

162/**
163 * @param {!e2e.ByteArray} key
164 * @param {!e2e.ByteArray} permutedChoice
165 * @return {!goog.math.Long}
166 * @private
167 */
168e2e.cipher.Des.prototype.permuteBlock_ = function(key, permutedChoice) {
169 var result = goog.math.Long.fromNumber(0);
170 for (var i = 0; i < permutedChoice.length; i++) {
171 var keyByte = key[((permutedChoice[i] - 1) >>> 3)];
172 var bit = ((keyByte << ((permutedChoice[i] - 1) & 0x7)) & 0xff) >>> 7;
173 var bitLong = goog.math.Long.fromNumber(bit);
174 result = result.or(bitLong.shiftLeft(((permutedChoice.length - 1) - i)));
175 }
176 return result;
177};

192/**
193 * Implements encryption and decryption.
194 * @param {!e2e.ByteArray} data The data to encrypt.
195 * @param {boolean} encrypt If true, does encryption, otherwise decrypttion.
196 * @return {!e2e.async.Result} The result of encryption.
197 * @private
198 */
199e2e.cipher.Des.prototype.crypt_ = function(data, encrypt) {
200 var block = this.permuteBlock_(data,
201 e2e.cipher.Des.initialPermutation);
202 var left = block.getHighBits();
203 var right = block.getLowBitsUnsigned();
204 for (var i = 0; i < 16; i++) {
205 var previousRight = right;
206 if (encrypt) {
207 right = left ^ this.feistel_(right, this.subkeys_[i]);
208 } else { // decrypt
209 right = left ^ this.feistel_(right, this.subkeys_[15 - i]);
210 }
211 left = previousRight;
212 }
213 var combined = e2e.longToByteArray(
214 goog.math.Long.fromBits(left, right)); // Left and right are swapped.
215 var result = this.permuteBlock_(combined,
216 e2e.cipher.Des.initialPermutationInverse);
217 return e2e.async.Result.toResult(e2e.longToByteArray(result));
218};

221/**
222 * @param {number} right 32-bit value to scramble.
223 * @param {goog.math.Long} key 64-bit key to use.
224 * @return {number} 32-bit value.
225 * @private
226 */
227e2e.cipher.Des.prototype.feistel_ = function(right, key) {
228 var rightExpanded = this.permuteBlock_(
229 e2e.dwordArrayToByteArray([right]),
230 e2e.cipher.Des.eBitSelection);
231 var xorResult = key.xor(rightExpanded);
232 var sBoxResult = 0;
233 for (var i = 0; i < 8; i++) {
234 var rowCol = (xorResult.shiftLeft(16 + (6 * i))
235 .shiftRightUnsigned(58).toInt()) & 0xff;
236 var row = (rowCol & 0x1) | ((rowCol & 0x20) >>> 4);
237 var col = ((rowCol << 3) & 0xff) >>> 4;
238 sBoxResult |= (e2e.cipher.Des.sBoxes[i][row][col] <<
239 (4 * (7 - i))) & 0xffffffff;
240 }
241 return this.permuteBlock_(e2e.dwordArrayToByteArray([sBoxResult]),
242 e2e.cipher.Des.permutationFunction).getLowBitsUnsigned();
243};
