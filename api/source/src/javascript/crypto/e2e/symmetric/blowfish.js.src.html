38/**
39 * Basic implementation of Blowfish, 16 rounds.
40 * OpenPGP specifies a 128 bit key, but this implementation supports arbitrary
41 * key lengths.
42 * @see http://www.schneier.com/blowfish.html
43 * @param {e2e.cipher.Algorithm} algorithm The algorithm being
44 * implemented.
45 * @param {e2e.cipher.key.Key=} opt_keyObj The key to use.
46 * @implements {e2e.cipher.SymmetricCipher}
47 * @extends {e2e.AlgorithmImpl}
48 * @constructor
49 */
50e2e.cipher.Blowfish = function(algorithm, opt_keyObj) {
51 /**
52 * @type {!e2e.DwordArray}
53 * @private
54 */
55 this.p_ = []; // 18 uint32 values.
56 this.keySize = 16;
57 goog.base(this, algorithm, opt_keyObj);
58};
59goog.inherits(e2e.cipher.Blowfish, e2e.AlgorithmImpl);

73/**
74 * Key expansion using P.
75 * @private
76 */
77e2e.cipher.Blowfish.prototype.keyExpansion_ = function() {
78 this.s0_ = goog.array.clone(e2e.cipher.Blowfish.S0);
79 this.s1_ = goog.array.clone(e2e.cipher.Blowfish.S1);
80 this.s2_ = goog.array.clone(e2e.cipher.Blowfish.S2);
81 this.s3_ = goog.array.clone(e2e.cipher.Blowfish.S3);
82 // No need to copy P to this.p_ because it's initialized below.
83
84 var keyIndex = 0; // this.key is byte array, not uint32 array.
85 for (var pIndex = 0; pIndex < 18; pIndex++) {
86 // d = next 4 bytes of key as a uint32.
87 var d = this.key.key[keyIndex] << 24;
88 keyIndex = (keyIndex + 1) % this.keySize;
89 d |= this.key.key[keyIndex] << 16;
90 keyIndex = (keyIndex + 1) % this.keySize;
91 d |= this.key.key[keyIndex] << 8;
92 keyIndex = (keyIndex + 1) % this.keySize;
93 d |= this.key.key[keyIndex];
94 keyIndex = (keyIndex + 1) % this.keySize;
95 this.p_[pIndex] = d ^ e2e.cipher.Blowfish.P[pIndex];
96 }
97
98 var xl = 0;
99 var xr = 0;
100 var result; // used for tuple-like unpacking.
101
102 // Encrypt all-zero string and assign to p1 and p2.
103 for (var pIndex = 0; pIndex < 18; pIndex += 2) {
104 result = this.encryptBlock_(xl, xr);
105 this.p_[pIndex] = xl = result[0];
106 this.p_[pIndex + 1] = xr = result[1];
107 }
108
109 for (var i = 0; i < 256; i += 2) {
110 result = this.encryptBlock_(xl, xr);
111 this.s0_[i] = xl = result[0];
112 this.s0_[i + 1] = xr = result[1];
113 }
114
115 for (var i = 0; i < 256; i += 2) {
116 result = this.encryptBlock_(xl, xr);
117 this.s1_[i] = xl = result[0];
118 this.s1_[i + 1] = xr = result[1];
119 }
120
121 for (var i = 0; i < 256; i += 2) {
122 result = this.encryptBlock_(xl, xr);
123 this.s2_[i] = xl = result[0];
124 this.s2_[i + 1] = xr = result[1];
125 }
126
127 for (var i = 0; i < 256; i += 2) {
128 result = this.encryptBlock_(xl, xr);
129 this.s3_[i] = xl = result[0];
130 this.s3_[i + 1] = xr = result[1];
131 }
132};

135/** @inheritDoc */
136e2e.cipher.Blowfish.prototype.encrypt = function(data) {
137 var dword = e2e.byteArrayToDwordArray(data);
138 var result = this.encryptBlock_(dword[0], dword[1]); // l, r
139 dword = e2e.dwordArrayToByteArray(result);
140 return e2e.async.Result.toResult(dword);
141};

148 dword = e2e.dwordArrayToByteArray(result);
149 return e2e.async.Result.toResult(dword);
150};

153/**
154 * Encrypt block.
155 * @param {number} xl 32bit number for left half.
156 * @param {number} xr 32bit number for right half.
157 * @return {!e2e.DwordArray} Computed values of [xl, xr].
158 * @private
159 */
160e2e.cipher.Blowfish.prototype.encryptBlock_ = function(xl, xr) {
161 xl ^= this.p_[0];
162 xr ^= ((this.s0_[(xl >>> 24) & 0xff] + this.s1_[(xl >>> 16) & 0xff]) ^
163 this.s2_[(xl >>> 8) & 0xff]) + this.s3_[(xl) & 0xff] ^ this.p_[1];
164 xl ^= ((this.s0_[(xr >>> 24) & 0xff] + this.s1_[(xr >>> 16) & 0xff]) ^
165 this.s2_[(xr >>> 8) & 0xff]) + this.s3_[(xr) & 0xff] ^ this.p_[2];
166 xr ^= ((this.s0_[(xl >>> 24) & 0xff] + this.s1_[(xl >>> 16) & 0xff]) ^
167 this.s2_[(xl >>> 8) & 0xff]) + this.s3_[(xl) & 0xff] ^ this.p_[3];
168 xl ^= ((this.s0_[(xr >>> 24) & 0xff] + this.s1_[(xr >>> 16) & 0xff]) ^
169 this.s2_[(xr >>> 8) & 0xff]) + this.s3_[(xr) & 0xff] ^ this.p_[4];
170 xr ^= ((this.s0_[(xl >>> 24) & 0xff] + this.s1_[(xl >>> 16) & 0xff]) ^
171 this.s2_[(xl >>> 8) & 0xff]) + this.s3_[(xl) & 0xff] ^ this.p_[5];
172 xl ^= ((this.s0_[(xr >>> 24) & 0xff] + this.s1_[(xr >>> 16) & 0xff]) ^
173 this.s2_[(xr >>> 8) & 0xff]) + this.s3_[(xr) & 0xff] ^ this.p_[6];
174 xr ^= ((this.s0_[(xl >>> 24) & 0xff] + this.s1_[(xl >>> 16) & 0xff]) ^
175 this.s2_[(xl >>> 8) & 0xff]) + this.s3_[(xl) & 0xff] ^ this.p_[7];
176 xl ^= ((this.s0_[(xr >>> 24) & 0xff] + this.s1_[(xr >>> 16) & 0xff]) ^
177 this.s2_[(xr >>> 8) & 0xff]) + this.s3_[(xr) & 0xff] ^ this.p_[8];
178 xr ^= ((this.s0_[(xl >>> 24) & 0xff] + this.s1_[(xl >>> 16) & 0xff]) ^
179 this.s2_[(xl >>> 8) & 0xff]) + this.s3_[(xl) & 0xff] ^ this.p_[9];
180 xl ^= ((this.s0_[(xr >>> 24) & 0xff] + this.s1_[(xr >>> 16) & 0xff]) ^
181 this.s2_[(xr >>> 8) & 0xff]) + this.s3_[(xr) & 0xff] ^ this.p_[10];
182 xr ^= ((this.s0_[(xl >>> 24) & 0xff] + this.s1_[(xl >>> 16) & 0xff]) ^
183 this.s2_[(xl >>> 8) & 0xff]) + this.s3_[(xl) & 0xff] ^ this.p_[11];
184 xl ^= ((this.s0_[(xr >>> 24) & 0xff] + this.s1_[(xr >>> 16) & 0xff]) ^
185 this.s2_[(xr >>> 8) & 0xff]) + this.s3_[(xr) & 0xff] ^ this.p_[12];
186 xr ^= ((this.s0_[(xl >>> 24) & 0xff] + this.s1_[(xl >>> 16) & 0xff]) ^
187 this.s2_[(xl >>> 8) & 0xff]) + this.s3_[(xl) & 0xff] ^ this.p_[13];
188 xl ^= ((this.s0_[(xr >>> 24) & 0xff] + this.s1_[(xr >>> 16) & 0xff]) ^
189 this.s2_[(xr >>> 8) & 0xff]) + this.s3_[(xr) & 0xff] ^ this.p_[14];
190 xr ^= ((this.s0_[(xl >>> 24) & 0xff] + this.s1_[(xl >>> 16) & 0xff]) ^
191 this.s2_[(xl >>> 8) & 0xff]) + this.s3_[(xl) & 0xff] ^ this.p_[15];
192 xl ^= ((this.s0_[(xr >>> 24) & 0xff] + this.s1_[(xr >>> 16) & 0xff]) ^
193 this.s2_[(xr >>> 8) & 0xff]) + this.s3_[(xr) & 0xff] ^ this.p_[16];
194 xr ^= this.p_[17];
195 return [xr, xl]; // Note swapped order of xr and xl.
196};
