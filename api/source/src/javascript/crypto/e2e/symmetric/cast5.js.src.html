37/**

39 * @param {!e2e.cipher.Algorithm} algorithm The algorithm being
40 * implemented.
41 * @param {e2e.cipher.key.Key=} opt_keyObj The key to use.
42 * @implements {e2e.cipher.SymmetricCipher}
43 * @extends {e2e.AlgorithmImpl}
44 * @constructor

46e2e.cipher.Cast5 = function(algorithm, opt_keyObj) {
47 this.keySize = 16;
48 goog.base(this, algorithm, opt_keyObj);

53/** @inheritDoc */
54e2e.cipher.Cast5.prototype.setKey = function(keyObj) {
55 // TODO(adhintz) Implement other key sizes if needed.
56 if (keyObj.key.length != 16) {
57 throw new e2e.cipher.Error('Wrong key size for Cipher.');
58 }
59
60 if (!e2e.isByteArray(keyObj.key)) {
61 throw new e2e.cipher.Error('Invalid values for key.');
62 }
63
64 var schedule = this.keySchedule_(keyObj.key);
65 keyObj.masking = schedule.masking;
66 keyObj.rotate = schedule.rotate;
67
68 // Superclass setKey sets the key to this.key_ and keySize.
69 goog.base(this, 'setKey', keyObj, keyObj.key.length);
70};

328/**
329 * Computes 16 pairs of subkeys {Kmi, Kri} from the original key (K).
330 * @param {!e2e.ByteArray} inputKey Original key.
331 * @return {!Object.<!e2e.ByteArray>} Rotate and masking subkeys.
332 * @private
333 */
334e2e.cipher.Cast5.prototype.keySchedule_ = function(inputKey) {
335 var t = new Array(8);
336 var outputKey = new Array(32);
337
338 for (var i = 0; i < 4; i++) {
339 var j = i * 4;
340 t[i] = inputKey[j] << 24 | inputKey[j + 1] << 16 | inputKey[j + 2] << 8 |
341 inputKey[j + 3];
342 }
343
344 var x = [6, 7, 4, 5];
345 var ki = 0;
346
347 for (var half = 0; half < 2; half++) {
348 for (var round = 0; round < 4; round++) {
349 for (var j = 0; j < 4; j++) {
350 var a = e2e.cipher.Cast5.keyScheduleA[round][j];
351 var w = t[a[1]];
352 w ^= e2e.cipher.Cast5.SBOX[4][
353 (t[a[2] >>> 2] >>> (24 - 8 * (a[2] & 3))) & 0xff];
354 w ^= e2e.cipher.Cast5.SBOX[5][
355 (t[a[3] >>> 2] >>> (24 - 8 * (a[3] & 3))) & 0xff];
356 w ^= e2e.cipher.Cast5.SBOX[6][
357 (t[a[4] >>> 2] >>> (24 - 8 * (a[4] & 3))) & 0xff];
358 w ^= e2e.cipher.Cast5.SBOX[7][
359 (t[a[5] >>> 2] >>> (24 - 8 * (a[5] & 3))) & 0xff];
360 w ^= e2e.cipher.Cast5.SBOX[x[j]][
361 (t[a[6] >>> 2] >>> (24 - 8 * (a[6] & 3))) & 0xff];
362 t[a[0]] = w;
363 }
364 for (var j = 0; j < 4; j++) {
365 var b = e2e.cipher.Cast5.keyScheduleB[round][j];
366 var w = e2e.cipher.Cast5.SBOX[4][
367 (t[b[0] >>> 2] >>> (24 - 8 * (b[0] & 3))) & 0xff];
368 w ^= e2e.cipher.Cast5.SBOX[5][
369 (t[b[1] >>> 2] >>> (24 - 8 * (b[1] & 3))) & 0xff];
370 w ^= e2e.cipher.Cast5.SBOX[6][
371 (t[b[2] >>> 2] >>> (24 - 8 * (b[2] & 3))) & 0xff];
372 w ^= e2e.cipher.Cast5.SBOX[7][
373 (t[b[3] >>> 2] >>> (24 - 8 * (b[3] & 3))) & 0xff];
374 w ^= e2e.cipher.Cast5.SBOX[4 + j][
375 (t[b[4] >>> 2] >>> (24 - 8 * (b[4] & 3))) & 0xff];
376 outputKey[ki] = w;
377 ki++;
378 }
379 }
380 }
381
382 var c = {};
383 c.masking = new Array(16);
384 c.rotate = new Array(16);
385 for (var i = 0; i < 16; i++) {
386 c.masking[i] = outputKey[i];
387 c.rotate[i] = outputKey[16 + i] & 0x1f;
388 }
389 return c;
390};

393/**
394 * f1 function
395 * @param {number} d
396 * @param {number} m
397 * @param {number} r
398 * @return {number}
399 * @private
400 */
401e2e.cipher.Cast5.prototype.f1_ = function(d, m, r) {
402 var t = m + d;
403 var I = (t << r) | (t >>> (32 - r));
404 return (
405 (e2e.cipher.Cast5.SBOX[0][I >>> 24] ^
406 e2e.cipher.Cast5.SBOX[1][(I >>> 16) & 0xff]) -
407 e2e.cipher.Cast5.SBOX[2][(I >>> 8) & 0xff]) +
408 e2e.cipher.Cast5.SBOX[3][I & 0xff];
409};

412/**
413 * f2 function
414 * @param {number} d
415 * @param {number} m
416 * @param {number} r
417 * @return {number}
418 * @private
419 */
420e2e.cipher.Cast5.prototype.f2_ = function(d, m, r) {
421 var t = m ^ d;
422 var I = (t << r) | (t >>> (32 - r));
423 return ((e2e.cipher.Cast5.SBOX[0][I >>> 24] -
424 e2e.cipher.Cast5.SBOX[1][(I >>> 16) & 0xff]) +
425 e2e.cipher.Cast5.SBOX[2][(I >>> 8) & 0xff]) ^
426 e2e.cipher.Cast5.SBOX[3][I & 0xff];
427};
