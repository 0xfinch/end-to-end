35/**
36 * Basic implementation of AES in pure JavaScript as specified in FIPS 197.
37 * The key size has to be 16, 24 or 32 bytes, or it will throw an exception.
38 * @see http://csrc.nist.gov/publications/fips/fips197/fips-197.pdf
39 * @param {e2e.cipher.Algorithm} algorithm The algorithm being
40 * implemented.
41 * @param {e2e.cipher.key.Key=} opt_keyObj The key to use.
42 * @implements {e2e.cipher.SymmetricCipher}
43 * @extends {e2e.AlgorithmImpl}
44 * @constructor
45 */
46e2e.cipher.Aes = function(algorithm, opt_keyObj) {
47 switch (algorithm) {
48 case e2e.cipher.Algorithm.AES128:
49 this.keySize = 16;
50 break;
51 case e2e.cipher.Algorithm.AES192:
52 this.keySize = 24;
53 break;
54 case e2e.cipher.Algorithm.AES256:
55 this.keySize = 32;
56 break;
57 default:
58 throw new e2e.cipher.Error('Invalid algorithm.');
59 }
60 goog.base(this, algorithm, opt_keyObj);
61};
62goog.inherits(e2e.cipher.Aes, e2e.AlgorithmImpl);
63
64
65/** @inheritDoc */
66e2e.cipher.Aes.prototype.setKey = function(keyObj) {
67 if (!keyObj.key || !e2e.isByteArray(keyObj.key)) {
68 throw new e2e.cipher.Error('Invalid values for key.');
69 }
70
71 if (keyObj.key.length != this.keySize) {
72 throw new e2e.cipher.Error(
73 'Wrong key size for Cipher: ' + keyObj.key.length);
74 }
75
76 // Superclass setKey sets the key to this.key and keySize.
77 goog.base(this, 'setKey', keyObj, keyObj.key.length);
78
79 /**
80 * Size of the key (in 4 byte words).
81 * @type {number}
82 * @private
83 */
84 this.Nk_ = this.key.key.length / 4;
85
86 /**
87 * Cycles of repetition.
88 * @type {number}
89 * @private
90 */
91 this.Nr_ = this.Nk_ + 6;
92
93 /**
94 * Words per block.
95 * TODO(evn): This probably should be a constant.
96 * @type {number}
97 * @private
98 */
99 this.Nb_ = 4;
100
101 /**
102 * The internal state.
103 * @type {Array.<number>}
104 * @private
105 */
106 this.state_ = [[], [], [], []];
107
108 /**
109 * Used to swap state_ when required.
110 * @type {Array.<number>}
111 * @private
112 */
113 this.temp_ = [[], [], [], []];
114
115 this.keyExpansion_();
116};

123/**
124 * Performs the encrypt operation synchronously.
125 * @param {!e2e.ByteArray} data The data to encrypt.
126 * @return {!e2e.cipher.ciphertext.Symmetric} The result of encryption.
127 */
128e2e.cipher.Aes.prototype.encryptSync = function(data) {
129 var output = [];
130
131 this.copyInput_(data, 0);
132 this.addRoundKey_(0);
133
134 for (var round = 1; round < this.Nr_; ++round) {
135 this.subBytes_(e2e.cipher.Aes.SBOX);
136 this.shiftRows_();
137 this.mixColumns_();
138 this.addRoundKey_(round);
139 }
140
141 this.subBytes_(e2e.cipher.Aes.SBOX);
142 this.shiftRows_();
143 this.addRoundKey_(this.Nr_);
144 this.copyOutput_(output, 0);
145
146 return output;
147};
148
149
150/** @inheritDoc */
151e2e.cipher.Aes.prototype.encrypt = function(data) {
152 return e2e.async.Result.toResult(this.encryptSync(data));
153};

156/**
157 * Performs the decrypt operation synchronously.
158 * @param {!e2e.cipher.ciphertext.Symmetric} data The encrypted data.
159 * @return {!e2e.ByteArray} The result of decryption.
160 */
161e2e.cipher.Aes.prototype.decryptSync = function(data) {
162 var output = [];
163
164 this.copyInput_(data, 0);
165 this.addRoundKey_(this.Nr_);
166
167 for (var round = 1; round < this.Nr_; ++round) {
168 this.invShiftRows_();
169 this.subBytes_(e2e.cipher.Aes.INV_SBOX);
170 this.addRoundKey_(this.Nr_ - round);
171 this.invMixColumns_();
172 }
173
174 this.invShiftRows_();
175 this.subBytes_(e2e.cipher.Aes.INV_SBOX);
176 this.addRoundKey_(0);
177 this.copyOutput_(output, 0);
178 return output;
179};

182/** @inheritDoc */
183e2e.cipher.Aes.prototype.decrypt = function(data) {
184 return e2e.async.Result.toResult(this.decryptSync(data));
185};

188/**
189 * Helper to copy input into the AES state matrix.
190 * @param {Array.<number>} input The input to the internal state.
191 * @param {number} startAt Offset relative to the input.
192 * @private
193 */
194e2e.cipher.Aes.prototype.copyInput_ = function(input, startAt) {
195 var v, p;
196
197 for (var r = 0; r < this.Nb_; r++) {
198 for (var c = 0; c < 4; c++) {
199 p = c * 4 + r + startAt;
200 v = input[p];
201
202 this.state_[r][c] = v;
203 }
204 }
205};
206
207
208/**
209 * Helper to copy the state matrix into the output.
210 * @param {Array.<number>} output Array to copy the internal state to.
211 * @param {number} startAt The offset where to start writing to output.
212 * @private
213 */
214e2e.cipher.Aes.prototype.copyOutput_ = function(output, startAt) {
215 for (var r = 0; r < this.Nb_; r++) {
216 for (var c = 0; c < 4; c++) {
217 output[c * 4 + r + startAt] = this.state_[r][c];
218 }
219 }
220};

223/**
224 * AES's AddRoundKey procedure. Add the current round key to the state.
225 * As specified in FIPS 197 Section 5.1.2.
226 * @param {number} round The round number.
227 * @private
228 */
229e2e.cipher.Aes.prototype.addRoundKey_ = function(round) {
230 for (var r = 0; r < 4; r++) {
231 for (var c = 0; c < 4; c++) {
232 this.state_[r][c] ^= this.w_[round * 4 + c][r];
233 }
234 }
235};

238/**
239 * AES's SubBytes procedure. Substitute bytes from the precomputed SBox lookup
240 * into the state. As specified in FIPS 197 Section 5.1.1.
241 * @param {Array.<number>} box The SBOX to use for the substitution.
242 * @private
243 */
244e2e.cipher.Aes.prototype.subBytes_ = function(box) {
245 for (var r = 0; r < 4; r++) {
246 for (var c = 0; c < 4; c++) {
247 this.state_[r][c] = box[this.state_[r][c]];
248 }
249 }
250};

252
253/**
254 * AES's ShiftRows procedure. Shift the values in each row to the right. Each
255 * row is shifted one more slot than the one above it.
256 * As specified in FIPS 197 Section 5.1.2.
257 * @private
258 */
259e2e.cipher.Aes.prototype.shiftRows_ = function() {
260 for (var r = 1; r < 4; r++) {
261 for (var c = 0; c < 4; c++) {
262 this.temp_[r][c] = this.state_[r][c];
263 }
264 }
265
266 for (var r = 1; r < 4; r++) {
267 for (var c = 0; c < 4; c++) {
268 this.state_[r][c] = this.temp_[r][(c + r) % this.Nb_];
269 }
270 }
271};
272
273
274/**
275 * AES's InvShiftRows procedure. Shift the values in each row to the right.
276 * As specified in FIPS 197 Section 5.3.1.
277 * @private
278 */
279e2e.cipher.Aes.prototype.invShiftRows_ = function() {
280 for (var r = 1; r < 4; r++) {
281 for (var c = 0; c < 4; c++) {
282 this.temp_[r][(c + r) % this.Nb_] = this.state_[r][c];
283 }
284 }
285
286 for (var r = 1; r < 4; r++) {
287 for (var c = 0; c < 4; c++) {
288 this.state_[r][c] = this.temp_[r][c];
289 }
290 }
291};

294/**
295 * AES's Mix Columns operation equivalent to (over GD 2^8):
296 * s0 = 2*t[0] + 3*t[1] + 1*t[2] + 1*t[3]
297 * s1 = 1*t[0] + 2*t[1] + 3*t[2] + 1*t[3]
298 * s2 = 1*t[0] + 1*t[1] + 2*t[2] + 3*t[3]
299 * s3 = 3*t[0] + 1*t[1] + 1*t[2] + 2*t[3]
300 * As defined in FIPS 197 Section 5.1.3.
301 * @private
302 */
303e2e.cipher.Aes.prototype.mixColumns_ = function() {
304 var s = this.state_;
305 var t = this.temp_[0];
306
307 for (var c = 0; c < 4; c++) {
308 t[0] = s[0][c];
309 t[1] = s[1][c];
310 t[2] = s[2][c];
311 t[3] = s[3][c];
312
313 s[0][c] = (
314 e2e.cipher.Aes.MULT_2[t[0]] ^
315 e2e.cipher.Aes.MULT_3[t[1]] ^
316 t[2] ^
317 t[3]);
318 s[1][c] = (t[0] ^
319 e2e.cipher.Aes.MULT_2[t[1]] ^
320 e2e.cipher.Aes.MULT_3[t[2]] ^
321 t[3]);
322 s[2][c] = (t[0] ^
323 t[1] ^
324 e2e.cipher.Aes.MULT_2[t[2]] ^
325 e2e.cipher.Aes.MULT_3[t[3]]);
326 s[3][c] = (e2e.cipher.Aes.MULT_3[t[0]] ^
327 t[1] ^
328 t[2] ^
329 e2e.cipher.Aes.MULT_2[t[3]]);
330 }
331};

334/**
335 * Inverse Mix Columns operation equivalent to (over GD 2^8):
336 * s0 = 0xE*t[0] + 0x9*t[3] + 0xD*t[2] + 0xB*t[1]
337 * s1 = 0xE*t[1] + 0x9*t[0] + 0xD*t[3] + 0xB*t[2]
338 * s2 = 0xE*t[2] + 0x9*t[1] + 0xD*t[0] + 0xB*t[3]
339 * s3 = 0xE*t[3] + 0x9*t[2] + 0xD*t[1] + 0xB*t[0]
340 * As defined in FIPS 197 Section 5.3.3.
341 * @private
342 */
343e2e.cipher.Aes.prototype.invMixColumns_ = function() {
344 var s = this.state_;
345 var t = this.temp_[0];
346
347 for (var c = 0; c < 4; c++) {
348 t[0] = s[0][c];
349 t[1] = s[1][c];
350 t[2] = s[2][c];
351 t[3] = s[3][c];
352
353 s[0][c] = (e2e.cipher.Aes.MULT_E[t[0]] ^
354 e2e.cipher.Aes.MULT_B[t[1]] ^
355 e2e.cipher.Aes.MULT_D[t[2]] ^
356 e2e.cipher.Aes.MULT_9[t[3]]);
357
358 s[1][c] = (e2e.cipher.Aes.MULT_9[t[0]] ^
359 e2e.cipher.Aes.MULT_E[t[1]] ^
360 e2e.cipher.Aes.MULT_B[t[2]] ^
361 e2e.cipher.Aes.MULT_D[t[3]]);
362
363 s[2][c] = (e2e.cipher.Aes.MULT_D[t[0]] ^
364 e2e.cipher.Aes.MULT_9[t[1]] ^
365 e2e.cipher.Aes.MULT_E[t[2]] ^
366 e2e.cipher.Aes.MULT_B[t[3]]);
367
368 s[3][c] = (e2e.cipher.Aes.MULT_B[t[0]] ^
369 e2e.cipher.Aes.MULT_D[t[1]] ^
370 e2e.cipher.Aes.MULT_9[t[2]] ^
371 e2e.cipher.Aes.MULT_E[t[3]]);
372 }
373};

376/**
377 * AES's KeyExpansion procedure. Create the key schedule (this.w_) from the
378 * initial key as specified in FIPS 197 Section 5.2.
379 * @private
380 */
381e2e.cipher.Aes.prototype.keyExpansion_ = function() {
382 this.w_ = new Array(this.Nb_ * (this.Nr_ + 1));
383
384 for (var rowNum = 0; rowNum < this.Nk_; rowNum++) {
385 this.w_[rowNum] = [
386 this.key.key[4 * rowNum],
387 this.key.key[4 * rowNum + 1],
388 this.key.key[4 * rowNum + 2],
389 this.key.key[4 * rowNum + 3]
390 ];
391 }
392
393 var temp = new Array(4);
394
395 for (var rowNum = this.Nk_; rowNum < (this.Nb_ * (this.Nr_ + 1)); rowNum++) {
396 temp[0] = this.w_[rowNum - 1][0];
397 temp[1] = this.w_[rowNum - 1][1];
398 temp[2] = this.w_[rowNum - 1][2];
399 temp[3] = this.w_[rowNum - 1][3];
400
401 if (rowNum % this.Nk_ == 0) {
402 this.rotWord_(temp);
403 this.subWord_(temp);
404
405 temp[0] ^= e2e.cipher.Aes.RCON[rowNum / this.Nk_][0];
406 temp[1] ^= e2e.cipher.Aes.RCON[rowNum / this.Nk_][1];
407 temp[2] ^= e2e.cipher.Aes.RCON[rowNum / this.Nk_][2];
408 temp[3] ^= e2e.cipher.Aes.RCON[rowNum / this.Nk_][3];
409 } else if (this.Nk_ > 6 && rowNum % this.Nk_ == 4) {
410 this.subWord_(temp);
411 }
412
413 this.w_[rowNum] = new Array(4);
414 this.w_[rowNum][0] = this.w_[rowNum - this.Nk_][0] ^ temp[0];
415 this.w_[rowNum][1] = this.w_[rowNum - this.Nk_][1] ^ temp[1];
416 this.w_[rowNum][2] = this.w_[rowNum - this.Nk_][2] ^ temp[2];
417 this.w_[rowNum][3] = this.w_[rowNum - this.Nk_][3] ^ temp[3];
418 }
419};

422/**
423 * AES's SubWord procedure that takes a four-byte input word and applies
424 * an S-box to each of the four bytes to produce an output word.
425 * As specified by FIPS 197 Section 5.2. and Section 5.1.1.
426 * @param {Array.<number>} w An array to substitute with the SBOX.
427 * @return {Array.<number>} The substituted word.
428 * @private
429 */
430e2e.cipher.Aes.prototype.subWord_ = function(w) {
431 w[0] = e2e.cipher.Aes.SBOX[w[0]];
432 w[1] = e2e.cipher.Aes.SBOX[w[1]];
433 w[2] = e2e.cipher.Aes.SBOX[w[2]];
434 w[3] = e2e.cipher.Aes.SBOX[w[3]];
435
436 return w;
437};

440/**
441 * AES's RotWord procedure that takes a four-byte word and performs a
442 * cyclic permutation. As specified in FIPS 197 Section 5.2.
443 * @param {Array.<number>} w The word to be rotated.
444 * @return {Array.<number>} The rotated word.
445 * @private
446 */
447e2e.cipher.Aes.prototype.rotWord_ = function(w) {
448 var temp = w[0];
449
450 w[0] = w[1];
451 w[1] = w[2];
452 w[2] = w[3];
453 w[3] = temp;
454
455 return w;
456};

541/**
542 * Precomputed Rcon (Round Constant) lookup. Contains the values given by:
543 * Rcon[i] = [x^(i-1),{00},{00},{00}] in the field GF(2^8).
