67/**
68 * This class is the common superclass of FastModulus.FFFFFF and
69 * FastModulus.Ox1000000, which have almost identical residue() methods.
70 *
71 * @constructor
72 * @implements {e2e.FastModulus}
73 * @param {!e2e.BigNum} modulus The modulus.
74 * @param {number} shift The amount the modulus and dividend need to be shifted
75 * to be in the correct special form.
76 */
77e2e.FastModulus.SpecialForm = function(modulus, shift) {
78 /** @override */
79 this.useForMultiplication = false;
80
81 /**
82 * The modulus.
83 * @private {!e2e.BigNum}
84 */
85 this.modulus_ = modulus;
86
87 /**
88 * The modulus shifted so that its high word has a special form
89 * @private {!e2e.BigNum}
90 */
91 this.shiftedModulus_ = modulus.shiftLeft(shift);
92
93 /**
94 * Amount by which modulus_ is shifted left to get shiftedModulus_ .
95 * @private {number}
96 */
97 this.shift_ = shift;
98};

101/** @override */
102e2e.FastModulus.SpecialForm.prototype.residue = function(value) {
103 var divisor = this.shiftedModulus_;
104 var dividend = this.shift_ == 0 ?
105 value.clone() : value.shiftLeft(this.shift_);
106 var length = divisor.n.length;
107 dividend.n.push(0);
108 for (var delta = dividend.n.length - length - 1; delta >= 0; --delta) {
109 // LOOP INVARIANT: (where base = 2^24)
110 // dividend / (divisor * base^delta) < base.
111 // Get the tentative quotient digit. The actual quotient must either be
112 // qhat or qhat - 1.
113 var qhat = this.calculateQuotient(dividend, length + delta);
114 // Subtract qhat * divisor * b^delta
115 var accu = 0;
116 for (var i = 0; i <= length; i++) {
117 accu += dividend.n[i + delta] - (divisor.n[i] | 0) * qhat;
118 dividend.n[i + delta] = accu & e2e.BigNum.BASE_MASK;
119 accu = Math.floor(accu / e2e.BigNum.BASE);
120 }
121 goog.asserts.assert(accu == 0 || accu == -1);
122 // Add divisor * b^delta back if the result went negative
123 var mask = accu;
124 accu = 0;
125 for (var i = 0; i <= length; i++) {
126 accu += dividend.n[i + delta] + (divisor.n[i] & mask);
127 dividend.n[i + delta] = accu & e2e.BigNum.BASE_MASK;
128 // accu is sum of two 24-bit values, so >>= is okay
129 accu >>= e2e.BigNum.BASE_LEN;
130 }
131 }
132 return dividend.shiftRight(this.shift_).cloneWithSize(
133 this.modulus_.getSize());
134};

137/**
138 * Returns a tentative quotient digit. The correct quotient digit is either the
139 * value returned by this method, or one less than the value returned.
140 *
141 * When this method is called, the loop invariant guarantees that
142 * dividend / (divisor * base ^ delta) < base [base = 2^24]
143 * and the correct quotient digit we seek is
144 * q = floor(dividend / (divisor * base ^ delta))
145 * This method returns either q or q + 1, and always returns a value less
146 * than base.
147 *
148 * @param {!e2e.BigNum} dividend The current dividend
149 * @param {number} deltaPlusLength delta + length, where delta is the amount by
150 * which we are currently shifting the divisor, and length is the length, in
151 * base 2^24 words, of the divisor.
152 */
153e2e.FastModulus.SpecialForm.prototype.calculateQuotient =
154 goog.abstractMethod;

158/**
159 * An implementation of FastModulus that handles the case in which the high
160 * 24 bits of the modulus are all 1s.
161 *
162 * This is useful for the modulus and size of the NIST elliptic curves.
163 *
164 * We find the residue by repeatedly subtracting multiples of the divisor from
165 * the dividend. Since the first digit of the divisor is base - 1, the first
166 * digit of the quotient must either be the first digit of the dividend, or one
167 * greater. We use one greater than the first digit as a tentative multiplier
168 * and subtract the multiplier times the divisor from the dividend. If the
169 * result is negative, we add the divisor back in.
170 *
171 * Here is a base 100 example of 57998823 % 9963
172 * 57 99 88 23
173 * -57 78 54 subtract divisor * 5800
174 * ===========
175 * 21 34 23
176 * -21 91 86 subtract divisor * 22
177 * 99 63 result went negative, add divisor back in
178 * ============
179 * 42 00
180 *
181 * @constructor
182 * @extends {e2e.FastModulus.SpecialForm}
183 * @param {!e2e.BigNum} modulus The value of the BigNum for which we are
184 * building a fast modulus function. Its 24 high order bits must be 1.
185 */
186e2e.FastModulus.FFFFFF = function(modulus) {
187 var n = modulus.getBitLength();
188 for (var i = n - e2e.BigNum.BASE_LEN; i < n; i++) {
189 if (!modulus.isBitSet(i)) {
190 throw new e2e.openpgp.error.InvalidArgumentsError(
191 'Bignum must start with 0xFFFFFF.');
192 }
193 }
194 var shift = (((n + e2e.BigNum.BASE_LEN - 1) /
195 e2e.BigNum.BASE_LEN) | 0) * e2e.BigNum.BASE_LEN - n;
196 e2e.FastModulus.FFFFFF.base(this, 'constructor', modulus, shift);
197};
198goog.inherits(e2e.FastModulus.FFFFFF,
199 e2e.FastModulus.SpecialForm);

202/** @override */
203e2e.FastModulus.FFFFFF.prototype.calculateQuotient =
204 function(dividend, deltaPlusLength) {
205 // Nomenclature: b = 2^24, Δ = delta, u = dividend, v = divisor, L = length
206 // We are trying to find the next quotient digit q < b such that
207 // q = floor(u / (v b^Δ))
208 // Since the high digit of the v is b-1,
209 // (1 - 1/b) b^L ≤ v < b^L.
210 // Let t be the digit in the b^(L + Δ) place of u, so that
211 // t b^(L + Δ) ≤ u ≤ (t + 1) b^(L + Δ)
212 // We know that t is a plausible quotient digit, because
213 // t v b^Δ < t b^L b^Δ = t^(L + Δ) ≤ u
214 // We that that t + 2 is not a plausible quotient digit
215 // because either t ≥ b - 2 (from the loop invariant q < b) or
216 // (t + 2) v b^Δ ≥ (t + 2)(1 - 1/b) b^(L + Δ)
217 // = (t + 1 + (b - t - 2)/b) b^(L + Δ)
218 // ≥ (t + 1) b^(L + Δ) > u
219 return Math.min(
220 dividend.n[deltaPlusLength] + 1, e2e.BigNum.BASE_MASK);
221};

225/**
226 * An implementation of FastModulus that handles the case in which the high
227 * bits of the modulus are a 1 followed by 24 zeros.
228 *
229 * This is useful for the order of the base point of Curve25519 and Ed25519.
230 *
231 * We find the residue by repeatedly subtracting multiples of the divisor from
232 * the dividend. Since the first digit of the divisor is 1, the first digit
233 * of the quotient must either be the first digit of the dividend or one less
234 * than that. We use the the dividend digit as the multiplier and subtract
235 * the multiplier times the divisor from the dividend. If the result is
236 * negative, we add the divisor back in. [In rare cases, we end up with a
237 * uncleared "1" from the previous digit, always followed by a 0. The divisor
238 * is base - 1 when that happens. See example below.]
239 *
240 * Here is a base 100 example of 7647508816 % 10063
241 * 76 47 50 88 16
242 * -76 47 88 subtract 760000 * 10063
243 * 1 00 63 result went negatve, add back 10000 * 10063
244 * ===============
245 * 1 00 25 88 16
246 * -99 62 37 hanging 100; subtract 99.00 * 1.00.63
247 * ===============
248 * 63 51 16
249 * -63 39 69 subtract 63 * 1.00.63
250 * ===============
251 * 11 47
252 *
253 * @constructor
254 * @extends {e2e.FastModulus.SpecialForm}
255 * @param {!e2e.BigNum} modulus The value of the BigNum for which we are
256 * building a fast modulus function. Its 24 high order bits must be 1.
257 */
258e2e.FastModulus.Ox1000000 = function(modulus) {
259 var n = modulus.getBitLength();
260 for (var i = n - e2e.BigNum.BASE_LEN - 1; i < n - 1; i++) {
261 if (modulus.isBitSet(i)) {
262 throw new e2e.openpgp.error.InvalidArgumentsError(
263 'Bignum must start with e2e.BigNum.BASE .');
264 }
265 }
266 // expectedBitLength is the smallest value >= bitLength such that
267 // expectedBitLength % e2e.BigNum.BASE_LEN == 1
268 var expectedBitLength = (((n + e2e.BigNum.BASE_LEN - 2) /
269 e2e.BigNum.BASE_LEN) | 0) * e2e.BigNum.BASE_LEN + 1;
270 var shift = expectedBitLength - n;
271 e2e.FastModulus.Ox1000000.base(this, 'constructor', modulus, shift);
272};
273goog.inherits(e2e.FastModulus.Ox1000000,
274 e2e.FastModulus.SpecialForm);

277/** @override */
278e2e.FastModulus.Ox1000000.prototype.calculateQuotient =
279 function(dividend, deltaPlusLength) {
280 // Nomenclature: b = 2^24, Δ = delta, u = dividend, v = divisor, L = length
281 // We are trying to find the next quotient digit q < b such that
282 // q = floor(u / (v b^Δ))
283 // Since the high digit of v is 1, and the next digit is 0:
284 // b^(L - 1) ≤ v < (1 + 1/b) b^(L - 1)
285 // If u ≥ b^(L + Δ), the quotient digit q must be b - 1, since q < b and
286 // (b - 1) v b^Δ < (b - 1)(1 + 1/b) b^(L - 1) b^Δ
287 // = (b - 1 + (b - 1)/b) b^(L - 1) b^Δ
288 // < b b^(L + Δ - 1) = b^(L + Δ) ≤ v
289 // Otherwise, let t be the digit in the b^(length + Δ - 1) place of u, i.e.
290 // t b^(L + Δ - 1) ≤ u < (t + 1) b^(L + Δ - 1)
291 // We know that t + 1 is too large to be the quotient digit, because
292 // (t + 1) v b^Δ ≥ (t + 1) b^(L - 1) b^Δ > u
293 // We also know that t - 1 is a possible quotient digit, because
294 // (t - 1) v b^Δ < (t - 1)(1 + 1/b) b^(L + Δ - 1)
295 // = (t - 1 + (t - 1)/b) b^(L + Δ - 1)
296 // < t b^(L + Δ - 1) ≤ u
297 // Hence, the quotient must be t or t - 1.
298 goog.asserts.assert(dividend.n[deltaPlusLength] == 0 ||
299 dividend.n[deltaPlusLength - 1] == 0,
300 'Dividend too large for invariant');
301 var qhat = dividend.n[deltaPlusLength] * e2e.BigNum.BASE_MASK +
302 dividend.n[deltaPlusLength - 1];
303 return qhat;
304};
