29/**
30 * BigPrimeNums are odd prime BigNumModuluses that could be use as the modulus
31 * in modular arithmetic operations in crypto schemes such as ECDSA or ECDH.
32 * @param {!e2e.ByteArray} modulus The modulus to use.
33 * @constructor
34 * @extends {e2e.BigNumModulus}
35 */
36e2e.BigPrimeNum = function(modulus) {
37 e2e.BigPrimeNum.base(this, 'constructor', modulus);
38};
39goog.inherits(e2e.BigPrimeNum, e2e.BigNumModulus);
40
41
42/**
43 * Computes this - 1 - a, thus Vx,
44 * this.mul(this.pow(x, a), this.pow(x, this.negateExponent(a))) == 1.
45 * @param {!e2e.ByteArray} input Number to negate.
46 * @return {!e2e.ByteArray} Negated number.
47 */
48e2e.BigNum.prototype.negateExponent = function(input) {
49 return this.negateExponent_(input);
50};
51
52
53/**
54 * Compute this - 1 - a, with 0 <= a < this.
55 * @param {!e2e.ByteArray} a number to negate.
56 * @return {!e2e.ByteArray} The negation of the input.
57 * @private
58 */
59e2e.BigPrimeNum.prototype.negateExponent_ = function(a) {
60 return this.subtract(e2e.BigNum.ONE).subtract(
61 new e2e.BigNum(a)).toByteArray();
62};

65/**
66 * Calculates x^{-1} mod this using Fermat's Little Theorem. Only works if
67 * this is a prime and x is smaller than q.
68 * @param {!e2e.BigNum} x The number to be inversed.
69 * @return {!e2e.BigNum}
70 */
71e2e.BigPrimeNum.prototype.modInverse = function(x) {
72 if (!x.isBetween(e2e.BigNum.ZERO, this)) {
73 throw new e2e.openpgp.error.InvalidArgumentsError(
74 'The number to be inversed should be in [1, this-1].');
75 }
76 // x^-1 = x^{q-2} (mod q).
77 return this.modPower(x, this.subtract(e2e.BigNum.TWO));
78};
