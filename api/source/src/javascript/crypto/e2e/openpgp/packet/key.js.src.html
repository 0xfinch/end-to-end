57/**
58 * A Key Packet that is the parent of SecretKey and PublicKey.
59 * @param {number} version The version of the key. Should be 0x04.
60 * @param {number} timestamp The creation time of the key.
61 * @param {!e2e.cipher.Cipher|!e2e.signer.Signer} cipher An
62 * instance of the cipher used.
63 * @param {!e2e.openpgp.KeyFingerprint=} opt_fingerprint The fingerprint of the
64 * key.
65 * @param {!e2e.openpgp.KeyId=} opt_keyId The key ID of the key. Should be
66 * passed in for v3 keys, but not for v4 keys.
67 * @extends {e2e.openpgp.packet.Packet}
68 * @constructor
69 */
70e2e.openpgp.packet.Key = function(
71 version, timestamp, cipher, opt_fingerprint, opt_keyId) {
72 goog.base(this);
73 /**
74 * The version of the key.
75 * @type {number}
76 */
77 this.version = version;
78
79 var keyId;
80 if (version == 3 || version == 2) {
81 keyId = opt_keyId;
82 }
83
84 if (goog.isDefAndNotNull(opt_fingerprint)) {
85 /**
86 * The fingerprint of the key.
87 * @type {!e2e.openpgp.KeyFingerprint}
88 */
89 this.fingerprint = opt_fingerprint;
90
91 if (version == 4) { // V4 fingerprint specified at 12.2 in RFC.
92 keyId = this.fingerprint.slice(-8);
93 }
94 }
95 /**
96 * If available, the key ID of the key.
97 * @type {!e2e.openpgp.KeyId|undefined}
98 */
99 this.keyId = keyId;
100 /**
101 * The timestamp of the creation of this public key.
102 * @type {number}
103 */
104 this.timestamp = timestamp;
105 /**
106 * The cipher of this public key (RSA/DSA/etc..).
107 * @type {!e2e.cipher.Cipher|!e2e.signer.Signer}
108 */
109 this.cipher = cipher;
110 /**
111 * @type {!e2e.ImmutableArray.<!e2e.openpgp.packet.Signature,
112 * e2e.openpgp.packet.KeyCertificationState_>}
113 * @private
114 */
115 this.bindingSignatures_ = new e2e.ImmutableArray([]);
116 /**
117 * @type {!e2e.ImmutableArray.<!e2e.openpgp.packet.Signature,
118 * e2e.openpgp.packet.KeyCertificationState_>}
119 * @private
120 */
121 this.revocations_ = new e2e.ImmutableArray([]);
122};

150/**
151 * Adds the key binding signature. Only possible for subkeys.
152 * Caution! Signature is not verified, use verifySignatures() function to verify
153 * the signature.
154 * @param {!e2e.openpgp.packet.Signature} signature
155 */
156e2e.openpgp.packet.Key.prototype.addBindingSignature = function(signature) {
157 if (!this.isSubkey ||
158 (signature.signatureType !==
159 e2e.openpgp.packet.Signature.SignatureType.SUBKEY)) {
160 throw new e2e.openpgp.error.ParseError(
161 'Invalid binding signature type.');
162 }
163 if (Boolean(signature.attributes && signature.attributes.KEY_FLAG_SIGN)) {
164 // RFC 4880 5.2.1.
165 // A signature that binds a signing subkey MUST have
166 // an Embedded Signature subpacket in this binding signature that
167 // contains a 0x19 signature made by the signing subkey on the
168 // primary key and subkey.
169 if (!signature.embeddedSignature) {
170 throw new e2e.openpgp.error.ParseError(
171 'Missing required key cross-signature.');
172 }
173 var crossSignature = signature.embeddedSignature;
174 if (!this.keyId ||
175 !e2e.compareByteArray(crossSignature.getSignerKeyId(), this.keyId) ||
176 crossSignature.signatureType !==
177 e2e.openpgp.packet.Signature.SignatureType.PRIMARY_KEY) {
178 throw new e2e.openpgp.error.ParseError('Invalid key cross-signature.');
179 }
180 }
181 this.bindingSignatures_ = e2e.ImmutableArray.pushCopy(
182 this.bindingSignatures_, signature);
183};

186/**
187 * Checks if a key has a valid (unrevoked) binding signature to a given
188 * key packet. The provided key must be the top-level key if this is a
189 * subkey, otherwise it must be the key itself. This function will
190 * throw SignatureError if any signature verification fails.
191 * @param {!e2e.openpgp.packet.Key} verifyingKey key packet that should
192 * verify the signatures
193 * @return {boolean} True if key has valid binding.
194 */
195e2e.openpgp.packet.Key.prototype.verifySignatures = function(verifyingKey) {
196 // Always process signatures to throw errors on any signature tampering.
197
198 var verifyingKeyId = verifyingKey.keyId;
199 if (!verifyingKeyId || !this.keyId) {
200 // Reject check on keys with no keyids.
201 return false;
202 }
203
204 // Ensure we're passed a toplevel key.
205 if (verifyingKey.isSubkey) {
206 throw new e2e.openpgp.error.SignatureError(
207 'Cannot verify key signatures with a subkey.');
208 }
209
210 // If we're a primary key, we can only be verified by ourselves.
211 if (!this.isSubkey &&
212 !e2e.compareByteArray(this.keyId, verifyingKeyId)) {
213 throw new e2e.openpgp.error.SignatureError(
214 'Cannot verify primary key with a different key.');
215 }
216
217 // There should be no valid revocation signatures.
218 var isRevoked = false;
219 e2e.ImmutableArray.forEach(this.revocations_, function(signature) {
220 if (this.verifyRevocation_(signature, verifyingKey)) {
221 isRevoked = true;
222 }
223 }, this);
224
225 if (isRevoked) {
226 return false;
227 }
228
229 // There are no valid revocation signatures, remove everything from
230 // our revocation array, and mark it as verified.
231 this.revocations_ = new e2e.ImmutableArray(
232 [], e2e.openpgp.packet.KeyCertificationState_.VERIFIED);
233
234 // Checks for a primary key are now complete.
235 if (!this.isSubkey) {
236 // We never add binding signatures to primary keys, so assert it.
237 e2e.assert(this.bindingSignatures_.size() == 0);
238 // Also mark binding array as verified.
239 this.bindingSignatures_ = new e2e.ImmutableArray(
240 [], e2e.openpgp.packet.KeyCertificationState_.VERIFIED);
241 return true;
242 }
243
244 // Subkeys must have a binding signature. See RFC 4880 11.1.
245 // If we have multiple valid binding signatures, use the most recent
246 // one.
247 var newestSignature = this.findNewestBindingSignature_(verifyingKey);
248 if (goog.isNull(newestSignature)) {
249 return false;
250 }
251 // Mark our binding array as being verified, and remove all elements
252 // from it except for the newest signature.
253 this.bindingSignatures_ = new e2e.ImmutableArray(
254 [newestSignature], e2e.openpgp.packet.KeyCertificationState_.VERIFIED);
255 return true;
256};

259/**
260 * Finds the newest valid binding signature on this key made by the
261 * verifyingKey, or null if none exists.
262 * @param {!e2e.openpgp.packet.Key} verifyingKey
263 * @return {e2e.openpgp.packet.Signature}
264 * @private
265 */
266e2e.openpgp.packet.Key.prototype.findNewestBindingSignature_ = function(
267 verifyingKey) {
268 var latestSignatureTime = -1;
269 var latestSignature = null;
270 e2e.ImmutableArray.forEach(this.bindingSignatures_, function(signature) {
271 if (this.verifyBindingSignature_(signature, verifyingKey)) {
272 // Check if we need to update the most recent signature.
273 if (signature.creationTime >= latestSignatureTime) {
274 latestSignature = signature;
275 latestSignatureTime = signature.creationTime;
276 }
277 }
278 }, this);
279 return latestSignature;
280};

344/**
345 * Verifies the signature, throwing error when verification fails or if
346 * signature is not of expected type.
347 * @param {!e2e.openpgp.packet.Signature} signature Revocation signature
348 * @param {!e2e.openpgp.packet.Key} verifyingKey key packet that should
349 * verify the signature
350 * @param {!e2e.ByteArray} signedData data that was signed.
351 * @param {string} verificationErrorMsg error message when signature did not
352 * verify.
353 * @return {boolean} True iff signature verified correctly.
354 * @private
355 */
356e2e.openpgp.packet.Key.prototype.verifySignatureInternal_ = function(signature,
357 verifyingKey, signedData, verificationErrorMsg) {
358 if (!verifyingKey.keyId || !e2e.compareByteArray(signature.getSignerKeyId(),
359 verifyingKey.keyId)) {
360 // Key mismatch, ignore signature.
361 return false;
362 }
363 var signer = /** @type {!e2e.signer.Signer} */ (verifyingKey.cipher);
364 try {
365 var signatureVerified = signature.verify(signedData,
366 goog.asserts.assertObject(signer));
367 } catch (e) {
368 // Ignore signatures that throw unsupported errors (e.g. weak hash
369 // algorithms) or expired signatures.
370 if (e instanceof e2e.openpgp.error.UnsupportedError) {
371 return false;
372 } else if (e instanceof e2e.openpgp.error.SignatureExpiredError) {
373 return false;
374 }
375 throw e;
376 }
377 if (!signatureVerified) {
378 throw new e2e.openpgp.error.SignatureError(verificationErrorMsg);
379 }
380 // Process embedded signature
381 if (this.isSubkey &&
382 signature.attributes &&
383 signature.attributes.KEY_FLAG_SIGN &&
384 signature.signatureType ===
385 e2e.openpgp.packet.Signature.SignatureType.SUBKEY) {
386 if (!signature.embeddedSignature) {
387 throw new e2e.openpgp.error.SignatureError(
388 'Missing cross-signature for a signing subkey.');
389 }
390 if (signature.embeddedSignature.signatureType !==
391 e2e.openpgp.packet.Signature.SignatureType.PRIMARY_KEY) {
392 throw new e2e.openpgp.error.SignatureError(
393 'Invalid cross-signature type.');
394 }
395 return this.verifySignatureInternal_(
396 signature.embeddedSignature,
397 this.getPublicKeyPacket(),
398 signedData,
399 'Cross-signature verification failed.');
400 }
401 return true;
402};

420/**
421 * Creates and attaches a SubKey Binding Signature, issued by the specified key
422 * packet.
423 * @param {!e2e.openpgp.packet.SecretKey} bindingKey
424 * @param {!e2e.openpgp.packet.Signature.SignatureType} type
425 * @param {number=} opt_keyFlags key usage flags to embed in the signature
426 * @return {!e2e.async.Result.<undefined>}
427 */
428e2e.openpgp.packet.Key.prototype.bindTo = function(bindingKey, type,
429 opt_keyFlags) {
430 // Check that we're a subkey, and that the provided key is a
431 // top-level key.
432 if (!this.isSubkey || bindingKey.isSubkey) {
433 throw new e2e.openpgp.error.SignatureError(
434 'Subkeys must be bound by top-level keys.');
435 }
436 var data = this.getKeyBindingSignatureData_(bindingKey);
437 var attributes = {
438 'SIGNATURE_CREATION_TIME': e2e.dwordArrayToByteArray(
439 [Math.floor(new Date().getTime() / 1e3)]),
440 'ISSUER': bindingKey.keyId
441 };
442 if (goog.isDef(opt_keyFlags)) {
443 attributes.KEY_FLAGS = [opt_keyFlags];
444 }
445 var sigRes = e2e.openpgp.packet.Signature.construct(
446 bindingKey,
447 data,
448 type,
449 attributes);
450
451 var bindingPublicKey = bindingKey.getPublicKeyPacket();
452 return sigRes.addCallback(function(sig) {
453 // This newly minted signature should become the latest bound
454 // certification on this key. We check this by explicitly
455 // verifying the newly created signature.
456 this.addBindingSignature(sig);
457 if (!this.verifySignatures(bindingPublicKey)) {
458 throw new e2e.openpgp.error.SignatureError(
459 'Unexpected - newly bound signature could not be verified.');
460 }
461 if (sig != this.getVerifiedCertification_(bindingPublicKey)) {
462 throw new e2e.openpgp.error.SignatureError(
463 'Unexpected - newly bound signature was not the latest.');
464 }
465 }, this);
466};

481/**
482 * Specifies whether the key packet can be used for a specific use.
483 * It only takes the cipher capabilities into consideration. In order to check
484 * whether the key owner wants the key to be used in a certain way, call
485 * isCertifiedTo().
486 * @param {e2e.openpgp.packet.Key.Usage} use Either 'sign' or 'encrypt'.
487 * @return {boolean}
488 */
489e2e.openpgp.packet.Key.prototype.can = function(use) {
490 if (use == e2e.openpgp.packet.Key.Usage.ENCRYPT) {
491 return e2e.cipher.factory.has(
492 /** @type {e2e.cipher.Algorithm} */ (this.cipher.algorithm));
493 } else if (use == e2e.openpgp.packet.Key.Usage.SIGN) {
494 return e2e.signer.factory.has(
495 /** @type {e2e.signer.Algorithm} */ (this.cipher.algorithm));
496 } else {
497 return false;
498 }

502/**
503 * Specifies whether the key packet has been certified for a specific use.
504 * Can only ask this question of a subkey.
505 * This information is obtained from the KEY_FLAGS property of
506 * its certification signature. If the signature contains no KEY_FLAGS
507 * at all, the subkey is considered to be certified for all uses.
508 * @param {!e2e.openpgp.packet.Key} verifyingKey the main key that originally
509 * certified this key.
510 * @param {!e2e.openpgp.packet.Key.Usage} use Either 'sign' or 'encrypt'.
511 * @return {boolean}
512 */
513e2e.openpgp.packet.Key.prototype.isCertifiedTo = function(verifyingKey, use) {
514 var sig = this.getVerifiedCertification_(verifyingKey);
515
516 // If the signature has no KEY_FLAGS property at all, the key
517 // is considered to be certified for all uses.
518 if (!sig.attributes.hasOwnProperty('KEY_FLAGS')) {
519 return true;
520 }
521 // Otherwise, look for a suitable attribute on the signature.
522 if (use === e2e.openpgp.packet.Key.Usage.SIGN) {
523 return sig.attributes.KEY_FLAG_SIGN !== 0;
524 } else if (use === e2e.openpgp.packet.Key.Usage.ENCRYPT) {
525 return sig.attributes.KEY_FLAG_ENCRYPT_COMMUNICATION ||
526 sig.attributes.KEY_FLAG_ENCRYPT_STORAGE !== 0;
527 } else {
528 return false;
529 }

564/**
565 * Return a previously verified certification, or raise an exception
566 * if none were found. This method only makes sense for a subkey.
567 * @param {!e2e.openpgp.packet.Key} verifyingKey the main key that
568 * certified this key.
569 * @return {!e2e.openpgp.packet.Signature}
570 * @private
571 */
572e2e.openpgp.packet.Key.prototype.getVerifiedCertification_ = function(
573 verifyingKey) {
574 if (!this.isSubkey) {
575 throw new e2e.openpgp.error.SignatureError(
576 'Cannot directly certify a primary key.');
577 }
578
579 // 1. Check that both the revocation and binding arrays are marked
580 // as verified.
581 if ((this.revocations_.getState() !==
582 e2e.openpgp.packet.KeyCertificationState_.VERIFIED) ||
583 (this.bindingSignatures_.getState() !==
584 e2e.openpgp.packet.KeyCertificationState_.VERIFIED)) {
585 throw new e2e.openpgp.error.SignatureError('This key is not certified.');
586 }
587
588 // 2. Sanity check - revocations array must be empty, and there should
589 // be exactly one element in bindingSignatures.
590 e2e.assert(this.revocations_.size() === 0);
591 e2e.assert(this.bindingSignatures_.size() === 1);
592
593 // signature should never be undefined, and this check also convinces
594 // closure that signature is defined.
595 var signature = goog.asserts.assertObject(this.bindingSignatures_.get(0));
596
597 // 3. Binding signature must be made by the verifyingKey
598 if (!verifyingKey.keyId ||
599 !e2e.compareByteArray(verifyingKey.keyId, signature.getSignerKeyId())) {
600 throw new e2e.openpgp.error.SignatureError(
601 'This key was certified by a different issuer.');
602 }
603 return signature;
604};

607/**
608 * Returns human-readable key fingerprint formatted as a hexadecimal string,
609 * with spaces separating each 4 hex digits, and 5 digit blocks.
