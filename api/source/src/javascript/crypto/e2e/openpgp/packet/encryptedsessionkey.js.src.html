29/**
30 * Represents encrypted session key packets (Tag 1 and 3) as specified in
31 * RFC 4880 Section 5.1 and Section 5.3.
32 * @param {number} version The version of the key packet.
33 * @param {e2e.cipher.Algorithm} algorithm The public key algorithm.
34 * @param {e2e.cipher.ciphertext.CipherText} encryptedKey Encrypted
35 * session key.
36 * @extends {e2e.openpgp.packet.Packet}
37 * @constructor
38 */
39e2e.openpgp.packet.EncryptedSessionKey = function(
40 version, algorithm, encryptedKey) {
41 this.version = version;
42 this.algorithm = algorithm;
43 this.encryptedKey = encryptedKey;
44 /**
45 * keyObj containing the symmetric session key.
46 * @type {e2e.cipher.key.SymmetricKey|undefined}
47 */
48 this.sessionKey;
49 /**
50 * Symmetric encryption algorithm used to encrypt the following
51 * Symmetrically Encrypted Data Packet. Used with this.sessionKey.
52 * @type {!e2e.cipher.Algorithm|undefined}
53 */
54 this.symmetricAlgorithm;
55};
56goog.inherits(e2e.openpgp.packet.EncryptedSessionKey,
57 e2e.openpgp.packet.Packet);

60/**
61 * Creates a cipher decrypting the session key encrpted in this packet.
62 * @param {!e2e.cipher.key.Key} key The key to decrypt the session key.
63 * @return {!e2e.cipher.Cipher} Cipher
64 */
65e2e.openpgp.packet.EncryptedSessionKey.prototype.createCipher =
66 goog.abstractMethod;

69/**
70 * Decrypts the encrypted session key.
71 * @param {!e2e.cipher.key.Key} key The key to decrypt the session key.
72 * @return {!e2e.async.Result.<boolean>} True if decryption succeeded.
73 */
74e2e.openpgp.packet.EncryptedSessionKey.prototype.decryptSessionKey =
75 function(key) {
76 return e2e.async.Result.toResult(key)
77 .addCallback(this.createCipher, this)
78 .addCallback(this.decryptSessionKeyWithCipher, this);
79};

83 * Decrypts the encrypted session key.

85 * @return {!e2e.async.Result.<boolean>} True if decryption succeeded.

87e2e.openpgp.packet.EncryptedSessionKey.prototype.decryptSessionKeyWithCipher =

91/**
92 * Obtains the decrypted session keyObj if available.
93 * Throws {@code Error} if the session key does not exist. This happens if
94 * decryptSessionKey() has not yet been called.
95 * @return {e2e.cipher.key.Key} The decrypted session key as a keyObj.
96 */
97e2e.openpgp.packet.EncryptedSessionKey.prototype.getSessionKey =
98 function() {
99 if (this.sessionKey) {
100 return this.sessionKey;
101 }
102 throw new Error('Invalid session key.');
103};

106/**
107 * Ensures that the cipher can be applied to the symmetric key decryption for
108 * this packet.
109 * @param {!e2e.cipher.Cipher} cipher The cipher.
110 * @return {!e2e.cipher.Cipher} The cipher.
111 * @protected
112 */
113e2e.openpgp.packet.EncryptedSessionKey.prototype.validateCipher = function(
114 cipher) {
115 // Algoritm mismatch.
116 if (cipher.algorithm !== this.algorithm) {
117 throw new e2e.openpgp.error.InvalidArgumentsError(
118 'Invalid cipher during SKESK packet decryption.');
119 }
120 return cipher;
