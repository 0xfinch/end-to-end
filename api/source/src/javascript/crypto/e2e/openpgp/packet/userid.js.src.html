60/**
61 * User ID Packet (Tag 13) defined in RFC 4880 section 5.11.
62 * @param {string} userId The user ID.
63 * @constructor
64 * @extends {e2e.openpgp.packet.Packet}
65 */
66e2e.openpgp.packet.UserId = function(userId) {
67 goog.base(this);
68 /**
69 * UTF-8 text representing the name and email address of the key holder.
70 * @type {string}
71 */
72 this.userId = userId;
73 /**
74 * Map from Key ID to an array of certifications or revocations from
75 * that Key ID. Each array also contains a state indicating whether
76 * the certifications in it have been verified.
77 * @type {!Object.<string,
78 * !e2e.ImmutableArray<!e2e.openpgp.packet.Signature,
79 * e2e.openpgp.packet.UserIdCertificationState_>>}

83};

107/**
108 * Query if the provided key has certified the User ID for a specific
109 * use. This information is obtained from the KEY_FLAGS property of
110 * its certification signature. If the signature contains no KEY_FLAGS
111 * at all, the userid is considered to be certified for all uses.
112 * If the key was never certified (through verifySignatures())
113 * this method will throw an exception.
114 * @param {!e2e.openpgp.packet.Key} key
115 * @param {!e2e.openpgp.packet.Key.Usage} use
116 * @return {boolean}
117 */
118e2e.openpgp.packet.UserId.prototype.isCertifiedTo = function(key, use) {
119 var sig = this.getVerifiedCertification_(key);
120 if (goog.isNull(sig)) {
121 throw new e2e.openpgp.error.SignatureError(
122 'User ID is not certified by the provided key');
123 }
124
125 // If the signature has no KEY_FLAGS property at all, the User ID
126 // is considered to be certified for all uses.
127 if (!sig.attributes.hasOwnProperty('KEY_FLAGS')) {
128 return true;
129 }
130 // Otherwise, look for a suitable attribute on the signature.
131 if (use === e2e.openpgp.packet.Key.Usage.SIGN) {
132 return sig.attributes.KEY_FLAG_SIGN !== 0;
133 } else if (use === e2e.openpgp.packet.Key.Usage.ENCRYPT) {
134 return sig.attributes.KEY_FLAG_ENCRYPT_COMMUNICATION ||
135 sig.attributes.KEY_FLAG_ENCRYPT_STORAGE !== 0;
136 } else {
137 return false;
138 }
139};

248/**
249 * @param {e2e.openpgp.packet.Signature} signature
250 * @param {!e2e.openpgp.packet.Key} verifyingKey key packet that should
251 * verify the certification signature.
252 * @param {!e2e.ByteArray} signedData data that was signed.
253 * @param {string} verificationErrorMsg error message when signature did not
254 * verify.
255 * @return {boolean} True iff signature verified correctly.
256 * @private
257 */
258e2e.openpgp.packet.UserId.prototype.verifySignatureInternal_ = function(
259 signature, verifyingKey, signedData, verificationErrorMsg) {
260 if (!verifyingKey.keyId || !e2e.compareByteArray(signature.getSignerKeyId(),
261 verifyingKey.keyId)) {
262 // Different key, ignore signature.
263 return false;
264 }
265 var signer = /** @type {!e2e.signer.Signer} */ (verifyingKey.cipher);
266 try {
267 var signatureVerified = signature.verify(signedData,
268 goog.asserts.assertObject(signer));
269 } catch (e) {
270 e2e.openpgp.packet.UserId.console_.warn(
271 'Unable to verify signature for ', this.userId, e);
272 // Ignore signatures that throw unsupported errors (e.g. weak hash
273 // algorithms) and expired signatures.
274 if (e instanceof e2e.openpgp.error.UnsupportedError) {
275 return false;
276 } else if (e instanceof e2e.openpgp.error.SignatureExpiredError) {
277 return false;
278 }
279 throw e;
280 }
281 if (!signatureVerified) {
282 throw new e2e.openpgp.error.SignatureError(verificationErrorMsg);
283 }
284 return true;
285};

288/**
289 * Checks if a packet has valid certification signature by a given
290 * key packet. This function will throw error if signature verification
291 * fails.
292 * @param {!e2e.openpgp.packet.Key} verifyingKey key packet that should
293 * verify the signatures
294 * @return {boolean} True if key has a valid certification.
295 */
296e2e.openpgp.packet.UserId.prototype.verifySignatures = function(verifyingKey) {
297 // 0. Only permit verification by keys with a Key ID.
298 if (!goog.isDef(verifyingKey.keyId)) {
299 e2e.openpgp.packet.UserId.console_.warn('No Key ID in verifying key');
300 return false;
301 }
302
303 // 1. Check if we have any signatures by the verifying Key ID.
304 var kid = e2e.openpgp.packet.UserId.keyIdToString_(verifyingKey.keyId);
305 var signatures = this.keyIdSignatureMap_[kid];
306 if (!goog.isDefAndNotNull(signatures)) {
307 e2e.openpgp.packet.UserId.console_.warn('No signatures made by ', kid);
308 return false;
309 }
310
311 // We first find the newest valid signature made by the verifying key.
312 //
313 // The User ID is considered certified iff such a valid signature
314 // exists, and is a certification signature.
315 //
316 // The rationale is that revocations on uid certifications are only
317 // applicable for certifications with timestamps older than them.
318 // See https://tools.ietf.org/html/rfc4880#section-5.2.1 and
319 // documentation on 0x30 signatures.
320 //
321 // Further, only the most recent certification is to be considered
322 // on User IDs. See
323 // https://tools.ietf.org/html/rfc4880#section-5.2.3.3
324 // This permits sequences of UID signatures like
325 // certify -- t1
326 // revoke -- t2
327 // certify -- t3
328 // where t3 > t2 > t1
329 // In this case, the User ID is considered certified, and any
330 // metadata is to be used from the t3 certification. (In practice as
331 // well, this is the model used by gpg.)
332 // However, if t2 > t3 and t1, then the User ID is considered
333 // revoked.
334 //
335 // If certified, all other signatures made by this key are removed,
336 // and only the latest (valid) signature is stored in the
337 // ImmutableArray.
338
339 var newestSignature = this.findNewestValidSignature_(
340 signatures, verifyingKey);
341
342 if (goog.isNull(newestSignature)) {
343 e2e.openpgp.packet.UserId.console_.warn('No usable signatures from ', kid);
344 return false;
345 }
346
347 // Reject unless the most recent signature is a certification signature.
348 if (!newestSignature.isCertificationSignature()) {
349 return false;
350 }
351
352 // 3. Now we clear the ImmutableArray of everything except
353 // the one valid signature, and mark it as verified.
354 this.keyIdSignatureMap_[kid] = new e2e.ImmutableArray(
355 [newestSignature],
356 e2e.openpgp.packet.UserIdCertificationState_.VERIFIED);
357 return true;
358};

361/**
362 * Returns the newest valid signature within the provided array made
363 * by the verifyingKey, or null if none exists.
364 * @param {!e2e.ImmutableArray<
365 * !e2e.openpgp.packet.Signature,
366 * e2e.openpgp.packet.UserIdCertificationState_>} signatures
367 * @param {!e2e.openpgp.packet.Key} verifyingKey
368 * @return {e2e.openpgp.packet.Signature}
369 * @private
370 */
371e2e.openpgp.packet.UserId.prototype.findNewestValidSignature_ = function(
372 signatures, verifyingKey) {
373 var latestSignature = null;
374 var latestTimestamp = -1;
375 var signatureData = this.getCertificationSignatureData_(verifyingKey);
376 e2e.ImmutableArray.forEach(signatures, function(signature) {
377 if (this.verifySignatureInternal_(
378 signature, verifyingKey, signatureData,
379 'User ID signature verification failed.')) {
380 // See if this is newer than our current newest signature.
381 if (signature.creationTime >= latestTimestamp) {
382 latestSignature = signature;
383 latestTimestamp = signature.creationTime;
384 }
385 }
386 }, this);
387 return latestSignature;
388};

391/**
392 * Return a verified certification made by the provided key, or null if
393 * no verified signatures were found.
394 * @param {!e2e.openpgp.packet.Key} key
395 * @return {e2e.openpgp.packet.Signature}
396 * @private
397 */
398e2e.openpgp.packet.UserId.prototype.getVerifiedCertification_ = function(key) {
399 // 0. Only allow keys with Key IDs.
400 if (!goog.isDef(key.keyId)) {
401 return null;
402 }
403
404 // 1. Get the ImmutableArray containing signatures made by
405 // this Key ID.
406 var kid = e2e.openpgp.packet.UserId.keyIdToString_(key.keyId);
407 var array = this.keyIdSignatureMap_[kid];
408 if (!goog.isDef(array)) {
409 return null;
410 }
411
412 // 2. Check whether we've assigned a certification state for this
413 // array.
414 if (!goog.isDef(array.getState())) {
415 return null;
416 }
417
418 // 3. Sanity checks. We only ever set a VERIFIED state. Furthermore,
419 // there must be exactly one certification signature.
420 e2e.assert((array.getState() ===
421 e2e.openpgp.packet.UserIdCertificationState_.VERIFIED) &&
422 (array.size() === 1));
423 var ret = array.get(0);
424 e2e.assert(ret.isCertificationSignature());
425 return ret;
426};

444/**
445 * @param {e2e.openpgp.packet.SecretKey} key
446 */
447e2e.openpgp.packet.UserId.prototype.certifyBy = function(key) {
448 var data = goog.array.flatten(
449 key.getPublicKeyPacket().getBytesToSign(),
450 this.getBytesToSign()
451 );
452 var sigResult = e2e.openpgp.packet.Signature.construct(
453 key,
454 data,
455 e2e.openpgp.packet.Signature.SignatureType.GENERIC_USER_ID,
456 this.getSignatureAttributes_(key));
457
458 var pubkey = key.getPublicKeyPacket();
459 sigResult.addCallback(function(sig) {
460 // This newly minted signature should become the latest, valid
461 // certification on this User ID. We check this by explicitly
462 // verifying the newly created signature.
463 this.addCertification(sig);
464 if (!this.verifySignatures(pubkey)) {
465 throw new e2e.openpgp.error.SignatureError(
466 'Unexpected - newly certified signature could not be verified.');
467 }
468 if (sig != this.getVerifiedCertification_(pubkey)) {
469 throw new e2e.openpgp.error.SignatureError(
470 'Unexpected - new certification was not the latest.');
471 }
472 }, this);
473};

476/**
477 * Returns key certification signature attributes, including End-to-End
478 * algorithm preferences.
479 * @param {e2e.openpgp.packet.SecretKey} key
480 * @return {!Object.<string, number|!e2e.ByteArray>} Attributes
481 * @private
482 */
483e2e.openpgp.packet.UserId.prototype.getSignatureAttributes_ = function(key) {
484 // Prefer only SHA-2 family.
485 var hashAlgos = [
486 e2e.hash.Algorithm.SHA256,
487 e2e.hash.Algorithm.SHA384,
488 e2e.hash.Algorithm.SHA512,
489 e2e.hash.Algorithm.SHA224
490 ];
491 var hashIds = goog.array.map(hashAlgos, e2e.openpgp.constants.getId);
492 // Prefer all available compression mechanisms.
493 var compressionAlgos = e2e.compression.factory.getAvailable();
494 var compressionIds = goog.array.map(compressionAlgos,
495 e2e.openpgp.constants.getId);
496 // Prefer only the default symmetric algorithm (AES-256).
497 var symAlgos = [
498 e2e.openpgp.constants.DEFAULT_SYMMETRIC_CIPHER
499 ];
500 var symIds = goog.array.map(symAlgos, e2e.openpgp.constants.getId);
501
502 return {
503 'SIGNATURE_CREATION_TIME': e2e.dwordArrayToByteArray(
504 [Math.floor(new Date().getTime() / 1e3)]),
505 'ISSUER': key.keyId,
506 'PREFERRED_SYMMETRIC_ALGORITHMS': symIds,
507 'PREFERRED_HASH_ALGORITHMS': hashIds,
508 'PREFERRED_COMPRESSION_ALGORITHMS': compressionIds,
509 'FEATURES': [0x01], // Modification detection. See RFC 4880 5.2.3.24.
510 'KEY_FLAGS': [
511 e2e.openpgp.packet.SignatureSub.KeyFlags.CERTIFY |
512 e2e.openpgp.packet.SignatureSub.KeyFlags.SIGN]
513 };
514};

517/**
518 * Gets a byte array representing the User ID data to create the signature over.
519 * This is intended for signatures of type 0x10 through 0x13.
520 * See RFC 4880 5.2.4 for details.
521 * @return {!e2e.ByteArray} The bytes to sign.
522 * @protected
523 */
524e2e.openpgp.packet.UserId.prototype.getBytesToSign = function() {
525 var data = e2e.stringToByteArray(this.userId);
526 return goog.array.flatten(
527 0xB4, // v4 User ID certifications must hash this constant first
528 e2e.dwordArrayToByteArray([data.length]),
529 data
530 );
531};
