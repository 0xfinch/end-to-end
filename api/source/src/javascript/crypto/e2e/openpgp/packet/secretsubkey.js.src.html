31/**
32 * A Secret Subkey Packet (Tag 7) RFC 4880 Section 5.5.1.4.
33 * @param {number} version The version of the key.
34 * @param {number} timestamp The creation time of the key.
35 * @param {!e2e.openpgp.EncryptedCipher} cipher The encrypted cipher.
36 * @param {!e2e.openpgp.KeyFingerprint=} opt_fingerprint The fingerprint of the
37 * key.
38 * @extends {e2e.openpgp.packet.SecretKey}
39 * @constructor
40 */
41e2e.openpgp.packet.SecretSubkey = function(
42 version, timestamp, cipher, opt_fingerprint) {
43 goog.base(this, version, timestamp, cipher, opt_fingerprint);
44};
45goog.inherits(e2e.openpgp.packet.SecretSubkey,
46 e2e.openpgp.packet.SecretKey);

49/** @inheritDoc */
50e2e.openpgp.packet.SecretSubkey.prototype.tag = 7;

53/**
54 * @override

56e2e.openpgp.packet.SecretSubkey.prototype.isSubkey = true;

59/** @override */
60e2e.openpgp.packet.SecretSubkey.prototype.getPublicKeyPacket = function() {
61 return new e2e.openpgp.packet.PublicSubkey(
62 this.version, this.timestamp, this.cipher, this.fingerprint);
63};

66/**
67 * Extracts a Secret Subkey Packet from the body, and returns a
68 * SecretSubkey.
69 * @param {!e2e.ByteArray} body The body from where to extract the data.
70 * @return {!e2e.openpgp.packet.SecretSubkey} The generated subkey.
71 */
72e2e.openpgp.packet.SecretSubkey.parse = function(body) {
73 var seckey = e2e.openpgp.packet.SecretKey.parse(body);
74 var subkey = new e2e.openpgp.packet.SecretSubkey(seckey.version,
75 seckey.timestamp,
76 seckey.cipher,
77 seckey.fingerprint);
78 return subkey;
79};
