55/**
56 * A Signature Packet (Tag 2) RFC 4880 Section 5.2.
57 * @param {number} version The version of the signature packet.
58 * @param {!e2e.openpgp.packet.Signature.SignatureType} signatureType
59 * The signature type.
60 * @param {!e2e.signer.Algorithm} pubKeyAlgorithm The public key
61 * algorithm.
62 * @param {!e2e.hash.Algorithm} hashAlgorithm The hash algorithm.
63 * @param {!e2e.signer.signature.Signature} signature The signature.
64 * @param {!e2e.ByteArray} leftTwoBytes The left two bytes of the hash.
65 * @param {Array.<e2e.openpgp.packet.SignatureSub>=} opt_hashedSubpackets
66 * The hashed subpackets of the signature.
67 * @param {Array.<e2e.openpgp.packet.SignatureSub>=} opt_unhashedSubpackets
68 * The non hashed subpackets of the signature.
69 * @param {!e2e.openpgp.KeyId=} opt_signerKeyId The key id of the signer.
70 * @param {number=} opt_creationTime The time the signature was created.
71 * @constructor
72 * @extends {e2e.openpgp.packet.Packet}
73 */
74e2e.openpgp.packet.Signature = function(
75 version, signatureType,
76 pubKeyAlgorithm, hashAlgorithm,
77 signature, leftTwoBytes,
78 opt_hashedSubpackets, opt_unhashedSubpackets,
79 opt_signerKeyId, opt_creationTime) {
80 goog.base(this);
81
82 var creationTime;
83 if (version == 0x04) {
84 /**
85 * Hashed signature subpackets.
86 * @type {!Array.<!e2e.openpgp.packet.SignatureSub>}
87 */
88 this.hashedSubpackets = opt_hashedSubpackets || [];
89 /**
90 * @type {!Object.<string, number|!e2e.ByteArray>}
91 */
92 this.attributes = {};
93 goog.array.forEach(this.hashedSubpackets, function(subpacket) {
94 e2e.openpgp.packet.SignatureSub.populateAttribute(
95 this.attributes, subpacket, false);
96 }, this);
97
98 // rfc 4880: 5.2.3.4 - signature creation time MUST be present
99 // within the hashed area.
100 if (!this.attributes.hasOwnProperty('SIGNATURE_CREATION_TIME')) {
101 throw new e2e.openpgp.error.InvalidArgumentsError(
102 'Missing signature timestamp.');
103 }
104 creationTime = this.attributes.SIGNATURE_CREATION_TIME;
105
106 /**
107 * Non hashed signature subpackets.
108 * @type {!Array.<!e2e.openpgp.packet.SignatureSub>}
109 */
110 this.unhashedSubpackets = opt_unhashedSubpackets || [];
111 /**
112 * @type {!Object.<string, number|!e2e.ByteArray>}
113 */
114 this.untrustedAttributes = {};
115 goog.array.forEach(this.unhashedSubpackets, function(subpacket) {
116 e2e.openpgp.packet.SignatureSub.populateAttribute(
117 this.untrustedAttributes, subpacket, false);
118 }, this);
119 /**
120 * Embedded signature.
121 * @type {e2e.openpgp.packet.Signature}
122 */
123 this.embeddedSignature = null;
124 var sigBytes = this.attributes.EMBEDDED_SIGNATURE ||
125 this.untrustedAttributes.EMBEDDED_SIGNATURE;
126 if (sigBytes) {
127 this.embeddedSignature = e2e.openpgp.packet.Signature.parse(
128 goog.array.clone(sigBytes));
129 }
130 } else if (version == 0x03 || version == 0x02) {
131 if (!goog.isDef(opt_signerKeyId) ||
132 !goog.isDef(opt_creationTime)) {
133 throw new e2e.openpgp.error.InvalidArgumentsError(
134 'Missing key data.');
135 }
136 creationTime = opt_creationTime;
137 /**
138 * ID of the key that generated this signature..
139 * @type {!e2e.openpgp.KeyId}
140 */
141 this.signerKeyId = opt_signerKeyId;
142 } else {
143 throw new e2e.openpgp.error.InvalidArgumentsError(
144 'Invalid Signature Packet version.');
145 }
146 /**
147 * Version of this signature.
148 * @type {number}
149 */
150 this.version = version;
151 /**
152 * Creation time of the signature.
153 * @type {number}
154 */
155 this.creationTime = creationTime;
156 /**
157 * Type of the signature.
158 * @type {!e2e.openpgp.packet.Signature.SignatureType}
159 */
160 this.signatureType = signatureType;
161 /**
162 * Public key algorithm used in this signature.
163 * @type {!e2e.signer.Algorithm}
164 */
165 this.pubKeyAlgorithm = pubKeyAlgorithm;
166 /**
167 * Hash algorithm used in this signature.
168 * @type {!e2e.hash.Algorithm}
169 */
170 this.hashAlgorithm = hashAlgorithm;
171 /**
172 * Signature data.
173 * @type {!e2e.signer.signature.Signature}
174 */
175 this.signature = signature;
176 /**
177 * Left Two Bytes of hash.
178 * @type {!e2e.ByteArray}
179 */
180 this.leftTwoBytes = leftTwoBytes;
181};
182goog.inherits(e2e.openpgp.packet.Signature,
183 e2e.openpgp.packet.Packet);

190/** @inheritDoc */
191e2e.openpgp.packet.Signature.prototype.serializePacketBody = function() {
192 var serialized = [];
193 serialized.push(this.version);
194 if (this.version == 0x03 || this.version == 0x02) {
195 goog.array.extend(
196 serialized,
197 0x05, // Length of metadata.
198 this.signatureType,
199 e2e.dwordArrayToByteArray([this.creationTime]),
200 this.signerKeyId,
201 e2e.openpgp.constants.getId(this.pubKeyAlgorithm),
202 e2e.openpgp.constants.getId(this.hashAlgorithm),
203 this.leftTwoBytes);
204 } else if (this.version == 0x04) {
205 goog.array.extend(
206 serialized,
207 this.signatureType,
208 e2e.openpgp.constants.getId(this.pubKeyAlgorithm),
209 e2e.openpgp.constants.getId(this.hashAlgorithm),
210 this.serializeHashedSubpackets(),
211 this.serializeUnhashedSubpackets(),
212 this.leftTwoBytes);
213 } else {
214 throw new e2e.openpgp.error.SerializationError('Invalid version.');
215 }
216 var sig = this.signature;
217 switch (this.pubKeyAlgorithm) {
218 case e2e.cipher.Algorithm.RSA:
219 case e2e.signer.Algorithm.RSA_SIGN:
220 goog.array.extend(serialized,
221 e2e.openpgp.Mpi.serialize(sig['s']));
222 break;
223 case e2e.signer.Algorithm.ECDSA:
224 case e2e.signer.Algorithm.DSA:
225 goog.array.extend(serialized,
226 e2e.openpgp.Mpi.serialize(sig['r']),
227 e2e.openpgp.Mpi.serialize(sig['s']));
228 break;
229 default:
230 throw new e2e.openpgp.error.UnsupportedError(
231 'Unsupported algorithm for signature verification.');
232 }
233 return serialized;
234};

265/** @inheritDoc */
266e2e.openpgp.packet.Signature.parse = function(data) {
267 var version = data.shift();
268 e2e.openpgp.packet.Signature.console_.info(
269 'Signature packet Ver ', version);
270 if (version == 0x03 || version == 0x02) {
271 var hashedMaterialLength = data.shift();
272 if (hashedMaterialLength != 0x05) {
273 throw new e2e.openpgp.error.ParseError('Invalid material length.');
274 }
275 var signatureType =
276 /** @type {e2e.openpgp.packet.Signature.SignatureType} */(
277 data.shift());
278 var creationTime = e2e.byteArrayToDwordArray(data.splice(0, 4))[0];
279 var signerKeyId = data.splice(0, 8);
280 var pubKeyAlgorithm = /** @type {e2e.cipher.Algorithm} */ (
281 e2e.openpgp.constants.getAlgorithm(
282 e2e.openpgp.constants.Type.PUBLIC_KEY, data.shift()));
283 var hashAlgorithm = /** @type {e2e.hash.Algorithm} */ (
284 e2e.openpgp.constants.getAlgorithm(
285 e2e.openpgp.constants.Type.HASH, data.shift()));
286 e2e.openpgp.packet.Signature.console_.info(
287 ' Sig type ', signatureType);
288 e2e.openpgp.packet.Signature.console_.info(
289 ' Pub alg ', pubKeyAlgorithm);
290 e2e.openpgp.packet.Signature.console_.info(
291 ' Hash alg ', hashAlgorithm);
292 e2e.openpgp.packet.Signature.console_.info(
293 ' Issuer ID ', signerKeyId);
294 } else if (version == 0x04) {
295 var signatureType =
296 /** @type {e2e.openpgp.packet.Signature.SignatureType} */(
297 data.shift());
298 var pubKeyAlgorithm = /** @type {e2e.signer.Algorithm} */ (
299 e2e.openpgp.constants.getAlgorithm(
300 e2e.openpgp.constants.Type.PUBLIC_KEY, data.shift()));
301 var hashAlgorithm = /** @type {e2e.hash.Algorithm} */ (
302 e2e.openpgp.constants.getAlgorithm(
303 e2e.openpgp.constants.Type.HASH, data.shift()));
304 e2e.openpgp.packet.Signature.console_.info(
305 ' Sig type ', signatureType);
306 e2e.openpgp.packet.Signature.console_.info(
307 ' Pub alg ', pubKeyAlgorithm);
308 e2e.openpgp.packet.Signature.console_.info(
309 ' Hash alg ', hashAlgorithm);
310 var hashedSubpacketLength = e2e.byteArrayToWord(
311 data.splice(0, 2));
312 e2e.openpgp.packet.Signature.console_.info(
313 ' Hashed subpackets');
314 var hashedSubpackets = e2e.openpgp.packet.SignatureSub.parse(
315 data.splice(0, hashedSubpacketLength));
316 e2e.openpgp.packet.Signature.console_.info(
317 ' Unhashed subpackets');
318 var unhashedSubpacketLength = e2e.byteArrayToWord(
319 data.splice(0, 2));
320 var unhashedSubpackets = e2e.openpgp.packet.SignatureSub.parse(
321 data.splice(0, unhashedSubpacketLength));
322 } else {
323 throw new e2e.openpgp.error.UnsupportedError(
324 'Unsupported signature packet version:' + version);
325 }
326
327 var leftTwoBytes = data.splice(0, 2);
328 var signature = {
329 's': []
330 };
331 switch (pubKeyAlgorithm) {
332 case e2e.signer.Algorithm.RSA:
333 case e2e.signer.Algorithm.RSA_SIGN:
334 signature['s'] = e2e.openpgp.Mpi.parse(data);
335 break;
336 case e2e.signer.Algorithm.DSA:
337 case e2e.signer.Algorithm.ECDSA:
338 signature['r'] = e2e.openpgp.Mpi.parse(data);
339 signature['s'] = e2e.openpgp.Mpi.parse(data);
340 break;
341 default: // Unsupported signature algorithm.
342 e2e.openpgp.packet.Signature.console_.warn(
343 'Unsupported Signature Algorithm', pubKeyAlgorithm);
344 return null;
345 }
346 return new e2e.openpgp.packet.Signature(
347 version, signatureType,
348 pubKeyAlgorithm, hashAlgorithm,
349 signature, leftTwoBytes,
350 hashedSubpackets, unhashedSubpackets,
351 signerKeyId, creationTime);
352};

355/**
356 * Extracts key ID used to place the signature directly from the packet
357 * (for version 3 packets) or from subpackets (for version 4 packets).
358 * @return {!e2e.openpgp.KeyId} signer key ID
359 */
360e2e.openpgp.packet.Signature.prototype.getSignerKeyId = function() {
361 if (this.version == 0x03 || this.version == 0x02) {
362 return this.signerKeyId;
363 }
364 if (this.version == 0x04) {
365 if (this.attributes.hasOwnProperty('ISSUER')) {
366 return /** @type {!e2e.openpgp.KeyId} */ (this.attributes['ISSUER']);
367 }
368 if (this.untrustedAttributes.hasOwnProperty('ISSUER')) {
369 // GnuPG puts Key ID in unhashed subpacket.
370 return /** @type {!e2e.openpgp.KeyId} */ (
371 this.untrustedAttributes['ISSUER']);
372 }
373 }
374 return e2e.openpgp.constants.EMPTY_KEY_ID;
375};

396/**
397 * Verifies that a key pair actually signed the specified data.
398 * @param {!e2e.ByteArray|!Uint8Array} data The signed data.
399 * @param {!e2e.signer.Signer} signer Signer with the public key that
400 * signed the data.
401 * @param {string=} opt_hashAlgo message digest algorithm declared in the
402 * message.
403 * @return {boolean} True if the signature correctly verifies.
404 */
405e2e.openpgp.packet.Signature.prototype.verify = function(data, signer,
406 opt_hashAlgo) {
407 if (this.pubKeyAlgorithm != signer.algorithm) {
408 return false;
409 }
410
411 // Hash algorithm declared in signature may differ from the one used
412 // when instantiating the signer. Use the one declared in signature
413 // (if it's allowed).
414 var allowedAlgo = e2e.openpgp.SignatureDigestAlgorithm[this.hashAlgorithm];
415 if (!allowedAlgo) {
416 throw new e2e.openpgp.error.UnsupportedError(
417 'Specified hash algorithm is not allowed for signatures.');
418 }
419 if (allowedAlgo !== signer.getHashAlgorithm()) {
420 signer.setHash(e2e.hash.factory.require(allowedAlgo));
421 }
422
423 if (goog.isDef(opt_hashAlgo) && opt_hashAlgo !== signer.getHashAlgorithm()) {
424 // Hash algorithm mismatch.
425 return false;
426 }
427 if (this.version != 0x04) {
428 throw new e2e.openpgp.error.UnsupportedError(
429 'Verification of old signature packets is not implemented.');
430 }
431 var signatureVerified = e2e.async.Result.getValue(
432 e2e.openpgp.packet.Signature.getSignatureScheme_(signer).verify(
433 e2e.openpgp.packet.Signature.getDataToHash(
434 data,
435 this.signatureType, this.pubKeyAlgorithm,
436 this.hashAlgorithm, this.hashedSubpackets),
437 this.signature));
438 if (signatureVerified &&
439 this.attributes.SIGNATURE_EXPIRATION_TIME &&
440 this.attributes.SIGNATURE_EXPIRATION_TIME <
441 Math.floor(new Date().getTime() / 1e3)) {
442 throw new e2e.openpgp.error.SignatureExpiredError(
443 'Signature expired.');
444 }
445 return signatureVerified;
446};

451 * public key packet certification.
452 * @return {boolean} True if the it's a certification signature, false otherwise

454e2e.openpgp.packet.Signature.prototype.isCertificationSignature = function() {

465/**
466 * Signs the data and creates a signature packet.
467 * @param {!e2e.openpgp.packet.SecretKeyInterface} key Key to sign with.
468 * @param {!e2e.ByteArray} data Data to sign.
469 * @param {!e2e.openpgp.packet.Signature.SignatureType} signatureType
470 * @param {!Object.<string, number|!e2e.ByteArray>} attributes
471 * The signature attributes. The SIGNATURE_CREATION_TIME attribute must
472 * always be present within it.
473 * @param {Object.<string, number|!e2e.ByteArray>=}
474 * opt_untrustedAttributes The signature untrusted attributes.
475 * @return {!e2e.async.Result.<!e2e.openpgp.packet.Signature>} Signature packet.
476 */
477e2e.openpgp.packet.Signature.construct = function(
478 key, data, signatureType, attributes, opt_untrustedAttributes) {
479 // Described in RFC4880 section 5.2.4.
480 // 5.2.3.4 - SIGNATURE_CREATION_TIME must be present as a hashed attribute.
481 if (!attributes.hasOwnProperty('SIGNATURE_CREATION_TIME')) {
482 throw new e2e.openpgp.error.InvalidArgumentsError(
483 'Missing required SIGNATURE_CREATION_TIME attribute.');
484 }
485
486 var hashedSubpackets = e2e.openpgp.packet.SignatureSub.construct(attributes);
487
488 var unhashedSubpackets = opt_untrustedAttributes ?
489 e2e.openpgp.packet.SignatureSub.construct(opt_untrustedAttributes) :
490 [];
491 var signer = /** @type {!e2e.signer.Signer} */ (key.cipher);
492 var algorithm = /** @type {!e2e.signer.Algorithm} */ (signer.algorithm);
493 goog.asserts.assert(algorithm in e2e.signer.Algorithm);
494
495 var plaintext = e2e.openpgp.packet.Signature.getDataToHash(
496 data,
497 signatureType,
498 algorithm,
499 signer.getHashAlgorithm(),
500 hashedSubpackets);
501 var resultSig;
502 var scheme = e2e.openpgp.packet.Signature.getSignatureScheme_(signer);
503 return scheme.sign(plaintext).addCallback(function(signature) {
504 return new e2e.openpgp.packet.Signature(
505 4, // version
506 signatureType,
507 algorithm,
508 signer.getHashAlgorithm(),
509 signature,
510 signature['hashValue'].slice(0, 2),
511 hashedSubpackets,
512 unhashedSubpackets);
513 });
514};

517/**
518 * Returns the data that has to be hashed for the signature algorithm.
519 * @param {!e2e.ByteArray|!Uint8Array} data The data that was signed.
520 * @param {!e2e.openpgp.packet.Signature.SignatureType} signatureType
521 * The signature type.
522 * @param {!e2e.signer.Algorithm} pubKeyAlgorithm The public key
523 * algorithm.
524 * @param {!e2e.hash.Algorithm} hashAlgorithm The hash algorithm.
525 * @param {!Array.<!e2e.openpgp.packet.SignatureSub>} subpackets
526 * The hashed subpackets of the signature.
527 * @return {!e2e.ByteArray} The data to hash.
528 */
529e2e.openpgp.packet.Signature.getDataToHash = function(
530 data, signatureType, pubKeyAlgorithm, hashAlgorithm, subpackets) {
531 var serializedHashedSubpackets =
532 e2e.openpgp.packet.Signature.serializeSubpackets(subpackets);
533 return goog.array.flatten(
534 data,
535 4, // version
536 signatureType,
537 e2e.openpgp.constants.getId(pubKeyAlgorithm),
538 e2e.openpgp.constants.getId(hashAlgorithm),
539 serializedHashedSubpackets,
540 4, // version
541 0xFF,
542 // Length of this hashTrailer (not including 6 final bytes):
543 e2e.dwordArrayToByteArray(
544 [serializedHashedSubpackets.length + 4])
545 );
546};

549/**
550 * Serializes a list of subpackets.
551 * @param {!Array.<!e2e.openpgp.packet.SignatureSub>} subpackets
552 * @return {!e2e.ByteArray} The serialized subpackets.
553 */
554e2e.openpgp.packet.Signature.serializeSubpackets =
555 function(subpackets) {
556 var serialized = goog.array.flatten(
557 goog.array.map(subpackets,
558 function(packet) {
559 return packet.serialize();
560 }));
561 if (serialized.length > 0xFFFF) {
562 throw new e2e.openpgp.error.SerializationError(
563 'Subpacket length is too long.');
564 }
565 var length = e2e.dwordArrayToByteArray(
566 [serialized.length]).slice(2);
567 return length.concat(serialized);
568};

610/**
611 * Returns a signature scheme for a given Signer.
612 * @param {e2e.signer.Signer} signer
613 * @return {!e2e.scheme.SignatureScheme|!e2e.signer.Signer}
614 * @private
615 */
616e2e.openpgp.packet.Signature.getSignatureScheme_ = function(signer) {
617 // KeyProviderCipher implements a scheme remotely, return it directly.
618 if (signer instanceof e2e.openpgp.KeyProviderCipher) {
619 return signer;
620 }
621 switch (signer.algorithm) {
622 case e2e.cipher.Algorithm.RSA:
623 case e2e.signer.Algorithm.RSA_SIGN:
624 return new e2e.scheme.Rsassa(signer);
625 break;
626 case e2e.signer.Algorithm.ECDSA:
627 return new e2e.scheme.Ecdsa(signer);
628 break;
629 case e2e.signer.Algorithm.DSA:
630 return signer;
631 break;
632 }
633 throw new e2e.openpgp.error.InvalidArgumentsError('Unsupported signer.');
634};
