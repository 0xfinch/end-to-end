33/**
34 * Signature Subpackets as defined in RFC 4880, section 5.2.3.1.
35 * @param {!e2e.openpgp.packet.SignatureSub.Type} type Subpacket type.
36 * @param {boolean} critical True if interpretation of this type is critical.
37 * @param {!e2e.ByteArray} body Unparsed data for this subpacket.
38 * @param {!e2e.ByteArray=} opt_packetLengthBytes The encoded length of
39 * the packet length.
40 * @constructor
41 */
42e2e.openpgp.packet.SignatureSub = function(
43 type, critical, body, opt_packetLengthBytes) {
44 this.type = type;
45 this.critical = critical;
46 this.body = body;
47 this.packetLengthBytes = opt_packetLengthBytes || this.getLength_();
48};

51/**
52 * Calculates the encoded length of the signature subpacket.
53 * @return {!e2e.ByteArray}
54 * @private
55 */
56e2e.openpgp.packet.SignatureSub.prototype.getLength_ = function() {
57 // Always use the 4 byte length.
58 return goog.array.flatten(
59 e2e.openpgp.parse.FIVE_BYTE_LENGTH_VAL,
60 e2e.dwordArrayToByteArray([this.body.length + 1]));
61};

66 * an exception otherwise.
67 * @param {!e2e.ByteArray} data The provided array.

69 * @private

72 if (data.length !== size) {
73 throw new e2e.openpgp.error.ParseError(
74 'Invalid array length, expected ' + size + ' but got ' + data.length);

79/**
80 * Shifts the first byte from the provided array but throws an
81 * exception if the array is empty. The array is also modified
82 * in-place.
83 * @param {!e2e.ByteArray} data The provided array.
84 * @return {number} The first element from the array.
85 * @private
86 */
87e2e.openpgp.packet.SignatureSub.checkedShift_ = function(data) {
88 if (data.length === 0) {
89 throw new e2e.openpgp.error.ParseError('Invalid array length.');
90 }
91 return data.shift();
92};

95/**
96 * Splices the provided array but throws an exception if the limits do
97 * not fall within the length of the array. The array is also modified
98 * in-place.
99 * @param {!e2e.ByteArray} data The provided array.
100 * @param {number} start The start index for the selection.
101 * @param {number} count The number of elements to splice.
102 * @return {!e2e.ByteArray} The spliced portion of the array
103 * @private
104 */
105e2e.openpgp.packet.SignatureSub.checkedSplice_ = function(data, start, count) {
106 var actual;
107 if (start < 0) {
108 actual = data.length + start;
109 } else {
110 actual = start;
111 }
112
113 if ((count < 0) || ((actual + count) > data.length)) {
114 throw new e2e.openpgp.error.ParseError(
115 'Invalid array splice request (' +
116 data.length + ', ' + start + ', ' + count + ')');
117 }
118 return data.splice(start, count);
119};

125e2e.openpgp.packet.SignatureSub.prototype.serialize = function() {
126 var ptag = (Number(this.critical) << 7) | this.type;

128 this.packetLengthBytes,

134/**
135 * @param {!Object.<string, !e2e.ByteArray>} attributes
136 * @return {!Array.<!e2e.openpgp.packet.SignatureSub>}
137 */
138e2e.openpgp.packet.SignatureSub.construct = function(attributes) {
139 var packets = [];
140 var map = new goog.structs.Map(e2e.openpgp.packet.SignatureSub.Type);
141 goog.array.forEach(
142 map.getKeys(),
143 function(typeName) {
144 if (attributes.hasOwnProperty(typeName)) {
145 var type = map.get(typeName);
146 packets.push(
147 new e2e.openpgp.packet.SignatureSub(
148 type, true, attributes[typeName]));
149 }
150 });
151 return packets;
152};

155/**
156 * Parses data and returns all of the parsed subpackets.
157 * @param {!e2e.ByteArray} data Data for subpackets.
158 * @return {!Array.<!e2e.openpgp.packet.SignatureSub>} Parsed subpackets.
159 */
160e2e.openpgp.packet.SignatureSub.parse = function(data) {
161 var packets = [];
162 var attributes = {};
163 while (data.length > 0) {
164 // This format is similar to e2e.openpgp.parse.getBodyNewFormatPacket_
165 // with the exception of partial body lengths.
166 var packetLengthBytes, packetLength;
167 var firstByte = e2e.openpgp.packet.SignatureSub.checkedShift_(data);
168 if (firstByte < e2e.openpgp.parse.TWO_BYTE_LENGTH_MIN) {
169 // One byte length packet size.
170 packetLength = firstByte;
171 packetLengthBytes = [firstByte];
172 } else if (firstByte < e2e.openpgp.parse.FIVE_BYTE_LENGTH_VAL) {
173 // Two bytes length packet size.
174 var secondByte = e2e.openpgp.packet.SignatureSub.checkedShift_(data);
175 packetLength = (
176 (firstByte - e2e.openpgp.parse.TWO_BYTE_LENGTH_MIN) << 8) +
177 (secondByte + e2e.openpgp.parse.TWO_BYTE_LENGTH_MIN);
178 packetLengthBytes = [firstByte, secondByte];
179 } else if (firstByte == e2e.openpgp.parse.FIVE_BYTE_LENGTH_VAL) {
180 // Five bytes length packet size.
181 packetLengthBytes =
182 e2e.openpgp.packet.SignatureSub.checkedSplice_(data, 0, 4);
183 packetLength = e2e.byteArrayToDwordArray(
184 packetLengthBytes)[0];
185 packetLengthBytes.unshift(
186 e2e.openpgp.parse.FIVE_BYTE_LENGTH_VAL);
187 }
188
189 var type = e2e.openpgp.packet.SignatureSub.checkedShift_(data);
190 var critical = Boolean(type & 0x80);
191 type = /** @type {e2e.openpgp.packet.SignatureSub.Type} */ (
192 type & 0x7F);
193 var body = e2e.openpgp.packet.SignatureSub.checkedSplice_(
194 data, 0, packetLength - 1);
195 e2e.openpgp.packet.SignatureSub.console_.info(
196 ' Sub', type, ' (' + body.length + ') bytes', body);
197
198 packets.push(new e2e.openpgp.packet.SignatureSub(type, critical, body,
199 packetLengthBytes));
200 }
201
202 return packets;
203};

206/**
207 * Parses the subpacket and populates the attributes object with the data.
208 * @param {!Object.<string, number|!e2e.ByteArray>} attributes Attributes
209 * record object to populate.
210 * @param {!e2e.openpgp.packet.SignatureSub} subpacket Subpacket to parse.
211 * @param {boolean} hashed True If subpacket is in hashed section of signature.
212 */
213e2e.openpgp.packet.SignatureSub.populateAttribute = function(
214 attributes, subpacket, hashed) {
215 // TODO(adhintz) Checks for which subpackets are fine for unhashed and
216 // different signature types.
217 switch (subpacket.type) {
218 case e2e.openpgp.packet.SignatureSub.Type.SIGNATURE_CREATION_TIME:
219 e2e.openpgp.packet.SignatureSub.assertSize_(subpacket.body, 4);
220 attributes.SIGNATURE_CREATION_TIME =
221 e2e.byteArrayToDwordArray(subpacket.body)[0];
222 break;
223 case e2e.openpgp.packet.SignatureSub.Type.SIGNATURE_EXPIRATION_TIME:
224 e2e.openpgp.packet.SignatureSub.assertSize_(subpacket.body, 4);
225 attributes.SIGNATURE_EXPIRATION_TIME =
226 e2e.byteArrayToDwordArray(subpacket.body)[0];
227 break;
228 case e2e.openpgp.packet.SignatureSub.Type.REGULAR_EXPRESSION:
229 attributes.REGULAR_EXPRESSION = e2e.byteArrayToString(subpacket.body);
230 break;
231 case e2e.openpgp.packet.SignatureSub.Type.KEY_EXPIRATION_TIME:
232 e2e.openpgp.packet.SignatureSub.assertSize_(subpacket.body, 4);
233 attributes.KEY_EXPIRATION_TIME =
234 e2e.byteArrayToDwordArray(subpacket.body)[0];
235 break;
236 case e2e.openpgp.packet.SignatureSub.Type.
237 PREFERRED_SYMMETRIC_ALGORITHMS:
238 attributes.PREFERRED_SYMMETRIC_ALGORITHMS = subpacket.body;
239 break;
240 case e2e.openpgp.packet.SignatureSub.Type.ISSUER:
241 e2e.openpgp.packet.SignatureSub.assertSize_(subpacket.body, 8);
242 attributes.ISSUER = subpacket.body;
243 break;
244 case e2e.openpgp.packet.SignatureSub.Type.PREFERRED_HASH_ALGORITHMS:
245 attributes.PREFERRED_HASH_ALGORITHMS = subpacket.body;
246 break;
247 case e2e.openpgp.packet.SignatureSub.Type.
248 PREFERRED_COMPRESSION_ALGORITHMS:
249 attributes.PREFERRED_COMPRESSION_ALGORITHMS = subpacket.body;
250 break;
251 case e2e.openpgp.packet.SignatureSub.Type.KEY_SERVER_PREFERENCES:
252 attributes.KEY_SERVER_PREFERENCES = subpacket.body[0];
253 break;
254 case e2e.openpgp.packet.SignatureSub.Type.PRIMARY_USER_ID:
255 e2e.openpgp.packet.SignatureSub.assertSize_(subpacket.body, 1);
256 attributes.PRIMARY_USER_ID = subpacket.body[0];
257 break;
258 case e2e.openpgp.packet.SignatureSub.Type.KEY_FLAGS:
259 if (subpacket.body.length == 0) {
260 attributes.KEY_FLAGS = 0;
261 } else {
262 // RFC 4880 section 5.2.3.21.
263 attributes.KEY_FLAGS = subpacket.body[0];
264 attributes.KEY_FLAG_CERTIFY = attributes.KEY_FLAGS &
265 e2e.openpgp.packet.SignatureSub.KeyFlags.CERTIFY;
266 attributes.KEY_FLAG_SIGN = attributes.KEY_FLAGS &
267 e2e.openpgp.packet.SignatureSub.KeyFlags.SIGN;
268 attributes.KEY_FLAG_ENCRYPT_COMMUNICATION = attributes.KEY_FLAGS &
269 e2e.openpgp.packet.SignatureSub.KeyFlags.ENCRYPT_COMMUNICATION;
270 attributes.KEY_FLAG_ENCRYPT_STORAGE = attributes.KEY_FLAGS &
271 e2e.openpgp.packet.SignatureSub.KeyFlags.ENCRYPT_STORAGE;
272 attributes.KEY_FLAG_SPLIT = attributes.KEY_FLAGS &
273 e2e.openpgp.packet.SignatureSub.KeyFlags.SPLIT;
274 attributes.KEY_FLAG_AUTHENTICATION = attributes.KEY_FLAGS &
275 e2e.openpgp.packet.SignatureSub.KeyFlags.AUTHENTICATION;
276 attributes.KEY_FLAG_SHARED = attributes.KEY_FLAGS &
277 e2e.openpgp.packet.SignatureSub.KeyFlags.SHARED;
278 }
279 break;
280 case e2e.openpgp.packet.SignatureSub.Type.REVOCATION_REASON:
281 attributes.REVOCATION_REASON = subpacket.body[0];
282 attributes.REVOCATION_REASON_TEXT = e2e.byteArrayToString(
283 goog.array.slice(subpacket.body, 1));
284 break;
285 case e2e.openpgp.packet.SignatureSub.Type.FEATURES:
286 if (subpacket.body.length == 0) {
287 attributes.FEATURES = 0;
288 } else {
289 attributes.FEATURES = subpacket.body[0]; // MDC is 0x01
290 }
291 break;
292 case e2e.openpgp.packet.SignatureSub.Type.EMBEDDED_SIGNATURE:
293 attributes.EMBEDDED_SIGNATURE = subpacket.body;
294 break;
295 default:
296 if (subpacket.critical) {
297 // TODO(adhintz): Treat the signature as invalid instead of throwing.
298 throw new e2e.openpgp.error.ParseError(
299 'Critical signature subpacket not recognized: ' + subpacket.type);
300 }
301 }
302};
