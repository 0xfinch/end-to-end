41/**
42 * A Secret Key Packet (Tag 5) RFC 4880 Section 5.5.1.3.
43 * @param {number} version The version of the key.
44 * @param {number} timestamp The creation time of the key.
45 * @param {!e2e.cipher.Cipher} cipher The cipher with the keys.
46 * @param {!e2e.openpgp.KeyFingerprint=} opt_fingerprint The fingerprint of the
47 * key.
48 * @param {!e2e.openpgp.KeyId=} opt_keyId The key ID of the key. Should be
49 * passed in for v3 keys, but not for v4 keys.
50 * @extends {e2e.openpgp.packet.Key}
51 * @constructor
52 */
53e2e.openpgp.packet.SecretKey = function(
54 version, timestamp, cipher, opt_fingerprint, opt_keyId) {
55 goog.asserts.assert(cipher instanceof e2e.openpgp.EncryptedCipher,
56 'The cipher for a secret key packet should be encrypted.');
57 goog.base(this, version, timestamp, cipher, opt_fingerprint, opt_keyId);
58};
59goog.inherits(e2e.openpgp.packet.SecretKey,
60 e2e.openpgp.packet.Key);

75/** @inheritDoc */
76e2e.openpgp.packet.SecretKey.prototype.serializePacketBody = function() {
77 var pubKey = new e2e.openpgp.packet.PublicKey(
78 this.version, this.timestamp, this.cipher, this.fingerprint, this.keyId);
79 var serializedPubKey = pubKey.serializePacketBody();
80 var kd = this.cipher.getKeyDerivationType();
81 if (kd == e2e.openpgp.EncryptedCipher.KeyDerivationType.PLAINTEXT) {
82 return goog.array.flatten(
83 serializedPubKey,
84 kd,
85 this.cipher.encryptedKeyData);
86 } else if (
87 kd == e2e.openpgp.EncryptedCipher.KeyDerivationType.S2K_CHECKSUM ||
88 kd == e2e.openpgp.EncryptedCipher.KeyDerivationType.S2K_SHA1) {
89 var s2k = this.cipher.getKeyDerivationS2k(),
90 iv = this.cipher.getKeyDerivationIv(),
91 kda = this.cipher.getKeyDerivationAlgorithm();
92 if (!goog.isDef(s2k) || !goog.isDef(iv) || !goog.isDef(kda)) {
93 throw new e2e.openpgp.error.SerializationError(
94 'Missing key metadata.');
95 }
96 return goog.array.flatten(
97 serializedPubKey, kd, e2e.openpgp.constants.getId(kda),
98 s2k.serialize(), iv, this.cipher.encryptedKeyData);
99 }
100 throw new e2e.openpgp.error.UnsupportedError(
101 'Key derivation type not supported.');
102};

112/**
113 * Extracts a Secret Key Packet from the body, and returns a SecretKey.
114 * @param {!e2e.ByteArray} body The body from where to extract the data.
115 * @return {!e2e.openpgp.packet.SecretKey} The generated packet.
116 */
117e2e.openpgp.packet.SecretKey.parse = function(body) {
118 var pubkey = e2e.openpgp.packet.PublicKey.parse(body);
119 var pubCipher = /** @type {e2e.cipher.AsymmetricCipher} */ (
120 pubkey.cipher);
121 var kd = body.shift();
122 var symAlgo, algId = -1, s2k, iv, encrypted = true;
123 switch (kd) {
124 case e2e.openpgp.EncryptedCipher.KeyDerivationType.S2K_CHECKSUM:
125 case e2e.openpgp.EncryptedCipher.KeyDerivationType.S2K_SHA1:
126 kd = /** @type {e2e.openpgp.EncryptedCipher.KeyDerivationType} */ (
127 kd);
128 algId = body.shift();
129 s2k = e2e.openpgp.S2k.parse(body);
130 e2e.openpgp.packet.SecretKey.console_.info(
131 ' key-derivation', kd);
132 e2e.openpgp.packet.SecretKey.console_.info(
133 ' Sym alg', algId);
134 e2e.openpgp.packet.SecretKey.console_.info(
135 ' S2K-type', s2k.type);
136 break;
137 case e2e.openpgp.EncryptedCipher.KeyDerivationType.PLAINTEXT:
138 kd = /** @type {e2e.openpgp.EncryptedCipher.KeyDerivationType} */ (
139 kd);
140 e2e.openpgp.packet.SecretKey.console_.info(
141 ' key-derivation', kd);
142 encrypted = false;
143 break;
144 default:
145 algId = kd;
146 kd = e2e.openpgp.EncryptedCipher.KeyDerivationType.MD5;
147 e2e.openpgp.packet.SecretKey.console_.info(
148 ' key-derivation', kd);
149 e2e.openpgp.packet.SecretKey.console_.info(
150 ' Sym alg', algId);
151 break;
152 }
153 if (encrypted) {
154 symAlgo = e2e.openpgp.constants.getAlgorithm(
155 e2e.openpgp.constants.Type.SYMMETRIC_KEY, algId);
156 symAlgo = /** @type {e2e.cipher.Algorithm} */ (symAlgo);
157 var symCipher = /** @type {e2e.cipher.SymmetricCipher} */ (
158 e2e.cipher.factory.require(symAlgo));
159 iv = body.splice(0, symCipher.blockSize);
160 }
161 var encryptedKeyData = body.splice(0, body.length);
162 var encCipher = new e2e.openpgp.EncryptedCipher(
163 encryptedKeyData, kd, pubCipher, symAlgo, iv, s2k);
164 return new e2e.openpgp.packet.SecretKey(pubkey.version,
165 pubkey.timestamp,
166 encCipher,
167 pubkey.fingerprint,
168 pubkey.keyId);
169};
