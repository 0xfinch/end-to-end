35/**
36 * Representation of a Compressed Data packet.
37 * @param {e2e.compression.Algorithm} algorithm The compression
38 * algorithm.
39 * @param {!e2e.ByteArray} compressedData The compressed data.
40 * @extends {e2e.openpgp.packet.Data}
41 * @constructor
42 */
43e2e.openpgp.packet.Compressed = function(algorithm, compressedData) {
44 goog.base(this);
45 /**
46 * The data compression algorithm.
47 * @type {e2e.compression.Algorithm}
48 * @protected
49 */
50 this.algorithm = algorithm;
51 /**
52 * The compressed data.
53 * @type {!e2e.ByteArray}
54 * @protected
55 */
56 this.compressedData = compressedData;
57 /** @inheritDoc */
58 this.data = [];
59};
60goog.inherits(e2e.openpgp.packet.Compressed,
61 e2e.openpgp.packet.Data);

68/**
69 * Decompresses the compressedData and populates this.data.
70 */
71e2e.openpgp.packet.Compressed.prototype.decompress = function() {
72 var compress = e2e.compression.factory.require(this.algorithm);
73 this.data = /** @type {!e2e.ByteArray} */ (
74 e2e.async.Result.getValue(compress.decompress(this.compressedData)));
75};

78/**
79 * Makes a compressed packet starting with uncompressed data, typically a
80 * serialized literal data packet.
81 * @param {!e2e.ByteArray} data The data to compress.
82 * @return {e2e.openpgp.packet.Compressed} packet.
83 */
84e2e.openpgp.packet.Compressed.construct = function(data) {
85 // TODO(adhintz) Add optional parameter specifying the compression algorithm.
86 var algorithm = e2e.compression.factory.require(
87 e2e.compression.Algorithm.ZLIB);
88 var compressedData = /** @type {!e2e.ByteArray} */ (
89 e2e.async.Result.getValue(algorithm.compress(data)));
90 return new e2e.openpgp.packet.Compressed(algorithm.algorithm,
91 compressedData);
92};

98 return goog.array.concat(id, this.compressedData);

102/**
103 * Parses and extracts the data from the body.
104 * Throws a {@code e2e.openpgp.error.ParseError} if malformed.
105 * @param {!e2e.ByteArray} body The data to parse.
106 * @return {e2e.openpgp.packet.Compressed} packet.
107 */
108e2e.openpgp.packet.Compressed.parse = function(body) {
109 var algorithm = /** @type {e2e.compression.Algorithm} */ (
110 e2e.openpgp.constants.getAlgorithm(
111 e2e.openpgp.constants.Type.COMPRESSION,
112 body.shift()));
113 return new e2e.openpgp.packet.Compressed(algorithm, body);
114};
