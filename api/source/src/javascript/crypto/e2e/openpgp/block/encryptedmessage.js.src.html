47/**
48 * Representation of an Encrypted Message block. According to RFC 4880
49 * Section 11.3, this block is represented as:
50 * - Optional repeated Encrypted Session Key (ESK) packets -- either public
51 * or symmetric.
52 * - One Encrypted Data block. Can be either a Symmetrically Encrypted Data
53 * Packet or a Symmetrically Encrypted Integrity Protected Data Packet.
54 * @param {Array.<e2e.openpgp.packet.EncryptedSessionKey>=} opt_eskPackets
55 * List of Encrypted Session Key (ESK) packets in this block.
56 * @param {e2e.openpgp.packet.EncryptedData=} opt_encryptedData
57 * Packet for the encrypted data block.
58 * @param {Array.<!e2e.openpgp.packet.Signature>=} opt_signatures
59 * @extends {e2e.openpgp.block.Message}
60 * @constructor
61 */
62e2e.openpgp.block.EncryptedMessage = function(opt_eskPackets,
63 opt_encryptedData, opt_signatures) {
64 /**
65 * List of Encrypted Session Key (ESK) packets in this block.
66 * @type {Array.<!e2e.openpgp.packet.EncryptedSessionKey>}
67 */
68 this.eskPackets = opt_eskPackets || null;
69 /**
70 * Packet for the encrypted data block.
71 * NOTE: The contents for this packet are another OpenPGP message.
72 * @type {e2e.openpgp.packet.EncryptedData}
73 */
74 this.encryptedData = opt_encryptedData || null;
75 goog.base(this, opt_signatures);
76};
77goog.inherits(e2e.openpgp.block.EncryptedMessage,
78 e2e.openpgp.block.Message);

81/**
82 * Decrypts the encrypted message and returns the containing decrypted block.
83 * @param {function(!e2e.openpgp.KeyId, !e2e.cipher.Algorithm):
84 * goog.async.Deferred<e2e.cipher.Cipher>} getCipherForSessionKeyCallback
85 * A callback to get a cipher for a given Key ID and algorithm.
86 * @param {function(string):!e2e.async.Result<string>} passphraseCallback A
87 * callback to get a passphrase for a given hint.
88 * @return {!e2e.async.Result.<!e2e.openpgp.block.Message>}
89 */
90e2e.openpgp.block.EncryptedMessage.prototype.decrypt = function(
91 getCipherForSessionKeyCallback, passphraseCallback) {
92 // Search for a secret key that can decrypt the session key. foundSecretKeys
93 // will contain an entry for every eskPacket.
94 // Some entries might be null - those represent symmetrically encrypted
95 // packets or packets for which a matching key has not been found.
96 var secretKeysResult = goog.array.map(this.eskPackets, function(eskPacket) {
97 if (eskPacket instanceof e2e.openpgp.packet.PKEncryptedSessionKey) {
98 return getCipherForSessionKeyCallback(eskPacket.keyId,
99 eskPacket.algorithm);
100 } else {
101 return e2e.async.Result.toResult(null);
102 }
103 }, this);
104
105 var result = goog.async.DeferredList.gatherResults(secretKeysResult)
106 .addCallback(function(foundSecretKeys) {
107 // Try to decrypt with all matched secret keys first.
108 var decryptResults =
109 /** @type {!Array.<
110 !goog.async.Deferred.<!e2e.openpgp.block.Message>>} */ (
111 goog.array.filter(
112 goog.array.map(foundSecretKeys, this.decryptWithCipher_, this),
113 goog.isDefAndNotNull));
114
115 var res;
116 if (decryptResults.length == 0) {
117 // No private keys could decrypt, try to find passphrases.
118 return this.decryptWithPassphrase_(passphraseCallback);
119 } else {
120 return new goog.async.DeferredList(decryptResults);
121 }
122 }, this)
123 .addCallback(this.decryptCallback_, this);
124 return /** @type {!e2e.async.Result.<!e2e.openpgp.block.Message>} */ (result);
125};

128/**
129 * Returns the first valid decryption result, or throws an exception if none.
130 * @param {!Array} list The list of results.
131 * @return {!e2e.openpgp.block.Message}
132 * @private
133 */
134e2e.openpgp.block.EncryptedMessage.prototype.decryptCallback_ = function(
135 list) {
136 var valid = goog.array.find(list, function(r) {
137 return r[0];
138 });
139 if (valid) {
140 return valid[1];
141 }
142 throw new e2e.openpgp.error.DecryptError(
143 'Encrypted message decryption failed.');
144};

147/**
148 * Tries to decrypt the session key (as specified by index) with a given cipher.
149 * @param {e2e.cipher.Cipher} cipher The cipher to try with.
150 * @param {number} index The index (on eskPackets) to try to decrypt.
151 * @return {e2e.async.Result.<!e2e.openpgp.block.Message>} Deferred decryption
152 * result or null, if no cipher was passed.
153 * @private
154 */
155e2e.openpgp.block.EncryptedMessage.prototype.decryptWithCipher_ =
156 function(cipher, index) {
157 if (!cipher) {
158 return null;
159 }
160 return this.decryptKeyAndMessage_(/** @type {!e2e.cipher.Cipher} */ (cipher),
161 this.eskPackets[index]);
162};

165/**
166 * Tries to decrypt the session key and then the message.
167 * @param {!e2e.cipher.Cipher} cipher The cipher to decrypt the session key.
168 * @param {!e2e.openpgp.packet.EncryptedSessionKey} eskPacket The encrypted
169 * session key packet to decrypt with the key object.
170 * @return {!e2e.async.Result.<!e2e.openpgp.block.Message>}
171 * @private
172 */
173e2e.openpgp.block.EncryptedMessage.prototype.decryptKeyAndMessage_ = function(
174 cipher, eskPacket) {
175 var decryptSuccess = eskPacket.decryptSessionKeyWithCipher(cipher);
176 return decryptSuccess.addCallback(function(success) {
177 if (!success) {
178 throw new e2e.openpgp.error.DecryptError(
179 'Session key decryption failed.');
180 }
181 return this.decryptMessage_(eskPacket);
182 }, this);
183};

186/**
187 * Attempts to decrypt the block with a passphrase. Will return an exception
188 * in the errback if it fails.
189 * @param {function(string):!e2e.async.Result<string>} passphraseCallback The
190 * callback for the passphrase.
191 * @return {!e2e.async.Result.<!e2e.openpgp.block.Message>}
192 * @private
193 */
194e2e.openpgp.block.EncryptedMessage.prototype.decryptWithPassphrase_ = function(
195 passphraseCallback) {
196 var result = new e2e.async.Result();
197 var symEskPackets = goog.array.filter(
198 this.eskPackets, function(esk) {
199 return esk instanceof e2e.openpgp.packet.SymmetricKey;
200 });
201 if (symEskPackets.length == 0) {
202 throw new e2e.openpgp.error.DecryptError('No keys found for message.');
203 }
204 // try to find the correct passphrase
205 this.silencePassphraseCallback_ = false;
206 this.repeatPassphraseCallback_ = false;
207 var decryptCallback = goog.bind(function(passphrase) {
208 try {
209 var res = this.testPassphrase_(
210 passphraseCallback, symEskPackets, passphrase);
211 res.addCallback(function(list) {
212 var failed = goog.array.every(list, function(f) {
213 return !f[0];
214 });
215 if (!failed) {
216 result.callback(list);
217 } else {
218 if (this.silencePassphraseCallback_ ||
219 !this.repeatPassphraseCallback_) {
220 throw new e2e.openpgp.error.DecryptError(
221 'Passphrase decryption failed');
222 } else {
223 passphraseCallback('').addCallback(decryptCallback);
224 }
225 }
226 }, this);
227 } catch (e) {
228 result.errback(e);
229 }
230 }, this);
231 passphraseCallback('').addCallback(decryptCallback);
232 return result;
233};

236/**
237 * Tries to decrypt the ESK packets with a given passphrase.
238 * @param {function(string):!e2e.async.Result<string>} passphraseCallback A
239 * callback to get a passphrase for a given hint.
240 * @param {!Array.<!e2e.openpgp.packet.EncryptedSessionKey>} symEskPackets The
241 * list of symmetrically encrypted session key packets.
242 * @param {string} passphraseString The passphrase to try to use to decrypt the
243 * ESK packet.
244 * @return {!goog.async.DeferredList} The deferred list of message blocks.
245 * @private
246 */
247e2e.openpgp.block.EncryptedMessage.prototype.testPassphrase_ = function(
248 passphraseCallback, symEskPackets, passphraseString) {
249 var passphraseKey = {'passphrase': e2e.stringToByteArray(passphraseString)};
250 var results = goog.array.map(
251 symEskPackets, goog.bind(this.testPassphraseKey_, this, passphraseKey));
252 return new goog.async.DeferredList(
253 /** @type {!Array.<!goog.async.Deferred> } */ (results));
254};

257/**
258 * Tries to decrypt the session key and then the message (this is just a very
259 * thin wrapper around decryptKeyAndMessage that silences the passphrase).
260 * @param {!e2e.cipher.key.Key} key The key object.
261 * @param {!e2e.openpgp.packet.EncryptedSessionKey} eskPacket The encrypted
262 * session key packet to decrypt with the key object.
263 * @return {!e2e.async.Result.<!e2e.openpgp.block.Message>}
264 * @private
265 */
266e2e.openpgp.block.EncryptedMessage.prototype.testPassphraseKey_ = function(
267 key, eskPacket) {
268 var cipher = eskPacket.createCipher(key);
269 return this.decryptKeyAndMessage_(cipher, eskPacket).addCallback(function() {
270 this.silencePassphraseCallback_ = true;
271 }, this).addErrback(function(e) {
272 // Error types that can be caught on an incorrect decryption:
273 // e2e.openpgp.error.PassphraseError - when the symmetric
274 // ESK decryption is incorrect and the ESK cipher byte is invalid.
275 // e2e.cipher.Error - when we have an incorrectly decrypted
276 // ESK, but the ESK cipher byte happens to be valid and the SEIP
277 // packet decrypts to have an invalid cipher choice.
278 // e2e.openpgp.error.DecryptError - when the ESK incorrectly
279 // decrypts, the ESK cipher byte happens to be valid, the SEIP
280 // cipher byte happens to be valid, but the SEIP decryption fails
281 // the duplicated two bytes and/or MDC check.
282 if ((e instanceof e2e.cipher.Error) ||
283 (e instanceof e2e.openpgp.error.PassphraseError) ||
284 (e instanceof e2e.openpgp.error.DecryptError)) {
285 this.repeatPassphraseCallback_ = true;
286 } else {
287 // We caught an unexpected error, fail the operation.
288 this.silencePassphraseCallback_ = true;
289 }
290 }, this);
291};

294/**
295 * Decrypts the encrypted data packet with the session key.
296 * @param {!e2e.openpgp.packet.EncryptedSessionKey} eskPacket The unlocked
297 * session key packet.
298 * @return {!e2e.openpgp.block.Message}
299 * @private
300 */
301e2e.openpgp.block.EncryptedMessage.prototype.decryptMessage_ = function(
302 eskPacket) {
303 if (!goog.isDef(eskPacket.symmetricAlgorithm)) {
304 throw new e2e.openpgp.error.DecryptError('Invalid session key packet.');
305 }
306 this.encryptedData.decrypt(
307 eskPacket.symmetricAlgorithm, eskPacket.getSessionKey());
308 var decryptedData = this.encryptedData.data;
309 // TODO(user): Can this be refactored to avoid the circular dependency?
310 /** @suppress {missingRequire} We assume the factory is already present. */
311 var message = e2e.openpgp.block.factory.parseByteArrayMessage(
312 decryptedData, this.getCharset());
313 if (message) {
314 return message;
315 }
316 throw new e2e.openpgp.error.ParseError('Invalid decrypted message.');
317};

320/** @inheritDoc */
321e2e.openpgp.block.EncryptedMessage.prototype.parse = function(packets) {
322 var eskPackets = [];
323 while (packets[0] instanceof
324 e2e.openpgp.packet.EncryptedSessionKey) {
325 eskPackets.push(packets.shift());
326 }
327 if (packets[0] instanceof
328 e2e.openpgp.packet.EncryptedData) {
329 var encryptedData = packets.shift();
330 } else {
331 throw new e2e.openpgp.error.ParseError(
332 'Invalid EncryptedMessage. Missing encrypted data block.');
333 }
334
335 this.eskPackets = eskPackets;
336 this.encryptedData = /** @type {e2e.openpgp.packet.EncryptedData} */
337 (encryptedData);
338 return packets;
339};

353/**
354 * Makes an EncryptedMessage containing data. Encrypts the data for
355 * the public keys passed in.
356 * @param {!e2e.openpgp.block.LiteralMessage} literalMessage
357 * Data to encrypt.
358 * @param {!Array.<!e2e.openpgp.block.TransferableKey>=} opt_publicKeys
359 * Keys to encrypt to.
360 * @param {!Array.<string>=} opt_passphrases Symmetrically encrypt
361 * session key with each of these passphrases. Either opt_publicKeys or
362 * opt_passphrases must be provided or
363 * {e2e.openpgp.error.InvalidArgumentsError} will be thrown.
364 * @param {e2e.openpgp.packet.SecretKeyInterface=} opt_signatureKey The key
365 * used to sign the message.
366 * @return {!e2e.async.Result.<!e2e.openpgp.block.EncryptedMessage>}
367 */
368e2e.openpgp.block.EncryptedMessage.construct = function(
369 literalMessage, opt_publicKeys, opt_passphrases, opt_signatureKey) {
370 // Prepare encryption keys.
371 var publicKeys = opt_publicKeys || [];
372 var passphrases = opt_passphrases || [];
373 goog.array.forEach(passphrases, function(passphrase, i, passphrases) {
374 passphrases[i] = e2e.stringToByteArray(passphrase);
375 });
376 publicKeys = goog.array.filter(goog.array.map(
377 publicKeys,
378 function(keyBlock) {
379 return keyBlock.getKeyToEncrypt();
380 }), goog.isDefAndNotNull);
381 if (publicKeys.length == 0 && passphrases.length == 0) {
382 throw new e2e.openpgp.error.InvalidArgumentsError(
383 'No public key nor passphrase was provided, encryption is impossible.');
384 }
385 // Optionally sign the message.
386 /** @type {!e2e.async.Result.<undefined>} */
387 var signResult = new e2e.async.Result;
388 if (opt_signatureKey) {
389 // Creates OnePassSignature + LiteralData + Signature sequence.
390 // That sequence will be later compressed and encrypted.
391 // This allows e.g. GnuPG to verify the signature.
392 signResult = literalMessage.signWithOnePass(opt_signatureKey);
393 } else {
394 signResult.callback();
395 }
396 var cipher = /** @type {!e2e.cipher.SymmetricCipher} */ (
397 e2e.cipher.factory.require(
398 e2e.openpgp.constants.DEFAULT_SYMMETRIC_CIPHER));
399 var sessionKey = e2e.random.getRandomBytes(cipher.keySize);
400 cipher.setKey({key: sessionKey});
401
402 var encryptedResult = signResult.addCallback(function() {
403 var compressedPacket = e2e.openpgp.packet.Compressed.construct(
404 literalMessage.serialize());
405
406 return e2e.openpgp.packet.SymmetricallyEncryptedIntegrity.construct(
407 compressedPacket.serialize(),
408 cipher);
409 });
410
411 var encryptedSessions = [];
412 goog.array.forEach(passphrases, function(passphrase) {
413 var packet = e2e.openpgp.packet.SymmetricKey.construct(
414 passphrase, sessionKey);
415 encryptedSessions.push(packet);
416 });
417
418 var pending = publicKeys.slice();
419 var blockResult = new e2e.async.Result;
420 goog.array.forEach(publicKeys, function(publicKey) {
421 var packetResult = e2e.openpgp.packet.PKEncryptedSessionKey.construct(
422 publicKey, sessionKey);
423 packetResult.addCallback(function(packet) {
424 encryptedSessions.push(packet);
425 pending.splice(pending.indexOf(publicKey), 1);
426 if (pending.length == 0) {
427 encryptedResult.addCallback(function(encrypted) {
428 // Make sure that blockresult won't finish until encryptedResult does.
429 blockResult.callback(encrypted);
430 });
431 }
432 });
433 });
434 if (publicKeys.length == 0) {
435 encryptedResult.addCallback(function(encrypted) {
436 // Make sure that blockresult won't finish until encryptedResult does.
437 blockResult.callback(encrypted);
438 });
439 }
440
441 blockResult.addCallback(function(encryptedData) {
442 var block = new e2e.openpgp.block.EncryptedMessage(
443 encryptedSessions,
444 encryptedData);
445 return block;
446 });
447 return blockResult;
448};
