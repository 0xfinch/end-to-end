29/**
30 * Implements a simple {@link KeyManager}.
31 * This KeyManager uses a single object as a {@link SecretKeyProvier} and
32 * {@link PublicKeyProvider}.
33 * @param {!e2e.openpgp.SecretKeyProvider} dualKeyProvider Object to use as both
34 * a public, and a secret key provider.
35 * @constructor
36 * @implements {e2e.openpgp.KeyManager}
37 */
38e2e.openpgp.SimpleKeyManager = function(dualKeyProvider) {
39 /** @private {!e2e.openpgp.SecretKeyProvider} */
40 this.keyProvider_ = dualKeyProvider;
41};

44/**
45 * Deferred constructor.
46 * @param {!goog.Thenable.<!e2e.openpgp.SecretKeyProvider>} keyProviderPromise
47 * The promise of both a public, and a secret key provider.
48 * @return {!goog.Thenable<!e2e.openpgp.SimpleKeyManager>} The SimpleKeyManager
49 * promise, fulfilled when the object has initialized.
50 */
51e2e.openpgp.SimpleKeyManager.launch = function(keyProviderPromise) {
52 return keyProviderPromise.then(function(keyProvider) {
53 return new e2e.openpgp.SimpleKeyManager(keyProvider);
54 });
55};

73/** @override */
74e2e.openpgp.SimpleKeyManager.prototype.getKeysByKeyId = function(purpose, id) {
75 var isSecret;
76 switch (purpose) {
77 case e2e.openpgp.KeyPurposeType.VERIFICATION:
78 isSecret = false;
79 break;
80 case e2e.openpgp.KeyPurposeType.DECRYPTION:
81 isSecret = true;
82 break;
83 default:
84 return goog.Promise.reject(
85 new e2e.openpgp.error.InvalidArgumentsError('Invalid key purpose.'));
86 }
87 return this.keyProvider_.getKeysByKeyId(purpose, id);
88};

185/** @override */
186e2e.openpgp.SimpleKeyManager.prototype.initializeKeyProviders = function(
187 config) {
188 return this.keyProvider_.getId().then(function(myProviderId) {
189 return this.keyProvider_.configure(config[myProviderId] || {})
190 .then(goog.bind(this.wrapState_, this, myProviderId));
191 }, null, this);

195/** @override */
196e2e.openpgp.SimpleKeyManager.prototype.reconfigureKeyProvider = function(
197 providerId, config) {
198 return this.keyProvider_.getId().then(function(myProviderId) {
199 if (providerId !== myProviderId) {
200 throw new e2e.openpgp.error.InvalidArgumentsError('Invalid provider');
201 }
202 return this.keyProvider_.configure(config);
203 }, null, this);
204};

216/**
217 * Wraps the state into an object keyed by KeyProviderId to satisfy the
218 * KeyManager contract.
219 * @param {!e2e.openpgp.KeyProviderId} providerId
220 * @param {e2e.openpgp.KeyProviderState} state
221 * @return {!Object<e2e.openpgp.KeyProviderId,e2e.openpgp.KeyProviderState>}
222 * @private
223 */
224e2e.openpgp.SimpleKeyManager.prototype.wrapState_ = function(providerId,
225 state) {
226 var states = {};
227 states[providerId] = state;
