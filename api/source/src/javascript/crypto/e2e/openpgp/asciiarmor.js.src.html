34/**
35 * Does base64 decoding ignoring extra characters, such as whitespace.
36 * @param {string} ascii The ASCII text to parse.
37 * @return {!e2e.ByteArray} The decoded data.
38 * @private
39 */
40e2e.openpgp.asciiArmor.decodeRadix64_ = function(ascii) {
41 var real = ascii.replace(/[^a-zA-Z0-9+/=]+/g, '');
42 return /** @type {!e2e.ByteArray} */ (
43 goog.crypt.base64.decodeStringToByteArray(real));
44};

47/**
48 * Does base64 encoding, inserting newlines to wrap long text.
49 * @param {!e2e.ByteArray} data The data to encode.
50 * @return {string} The encoded data as ASCII text.
51 * @private
52 */
53e2e.openpgp.asciiArmor.encodeRadix64_ = function(data) {
54 var ascii = goog.crypt.base64.encodeByteArray(data);
55 var lines = ascii.replace(/.{1,64}/g, '$&\r\n');
56 return lines.slice(0, -2); // Remove extra \r\n at end.
57};

60/**
61 * Calculates CRC24.
62 * @param {!e2e.ByteArray} data The data to do a checksum of.
63 * @return {number} The checksum.
64 * @private
65 */
66e2e.openpgp.asciiArmor.crc24_ = function(data) {
67 var CRC24_INIT = 0xB704CE;
68 var CRC24_POLY = 0x1864CFB;
69 var j = 0;
70 var crc = CRC24_INIT;
71 while (j < data.length) {
72 crc ^= data[j++] << 16;
73 for (var i = 0; i < 8; i++) {
74 crc = crc << 1;
75 if (crc & 0x1000000)
76 crc ^= CRC24_POLY;
77 }
78 }
79 return crc & 0xFFFFFF;
80};

92/**
93 * Parses the first ASCII Armor in a string.
94 * Specified in RFC 4880 Section 6.2.
95 * Throws a {@code e2e.openpgp.error.ParseError} if the Armor is invalid.
96 * @param {string} text The text to parse as ASCII Armor.
97 * @return {!e2e.openpgp.ArmoredMessage} The parsed message.
98 */
99e2e.openpgp.asciiArmor.parse = function(text) {
100 var armors = e2e.openpgp.asciiArmor.parseAll(text, 1);
101 if (armors.length !== 1) {
102 throw new e2e.openpgp.error.ParseError('ASCII Armor not found.');
103 }
104 return armors[0];
105};

108/**
109 * Parses all ASCII Armors present in a string.
110 * Specified in RFC 4880 Section 6.2.
111 * Throws a {@code e2e.openpgp.error.ParseError} if the Armor is invalid.
112 * @param {string} text The text to parse as ASCII Armor.
113 * @param {number=} opt_limit Stop parsing once opt_limit armors have been
114 * parsed.
115 * @return {!Array.<!e2e.openpgp.ArmoredMessage>} The parsed message.
116 */
117e2e.openpgp.asciiArmor.parseAll = function(text, opt_limit) {
118 // The 0x80 bit is always set for the Packet Tag for OpenPGP packets.
119 if (text.charCodeAt(0) >= 0x80) {
120 // Not ASCII Armored. Treat as a binary OpenPGP block
121 return [{
122 'data': goog.crypt.stringToByteArray(text),
123 'type': 'BINARY',
124 'startOffset': 0,
125 'endOffset': text.length
126 }];
127 }
128 if (text.indexOf('-----BEGIN PGP ') == -1) {
129 return [];
130 }
131
132 var armor, newLine = e2e.openpgp.asciiArmor.NEW_LINE_;
133 // TODO(adhintz) Switch away from regex to line-by-line parsing.
134 var armorRe = new RegExp(
135 '(^|' + newLine + ')-----BEGIN PGP ([^-]+)-----' + newLine +
136 '((?:[A-Za-z]+:[ ][^\\n]+' + newLine + ')*)' + newLine + // headers
137 '((?:[a-zA-Z0-9/+]+=*' + newLine + ')*)' + // body
138 '(?:=([a-zA-Z0-9/+]+))?' + newLine + // checksum
139 '(?:' + newLine + ')*-----END PGP \\2-----($|' + newLine + ')', 'gm');
140 var validArmors = [], payload, checksum, calculatedChecksum, prefixNewline,
141 suffixNewline, charset, charsetMatch;
142 while ((!goog.isDef(opt_limit) || opt_limit > 0) &&
143 goog.isDefAndNotNull(armor = armorRe.exec(text))) {
144 prefixNewline = armor[1];
145 suffixNewline = armor[6];
146 payload = e2e.openpgp.asciiArmor.decodeRadix64_(armor[4]);
147 checksum = e2e.openpgp.asciiArmor.decodeRadix64_(armor[5]);
148 calculatedChecksum = [e2e.openpgp.asciiArmor.crc24_(payload)];
149 calculatedChecksum = e2e.dwordArrayToByteArray(calculatedChecksum);
150 calculatedChecksum = calculatedChecksum.slice(-3);
151 if (calculatedChecksum.join('') != checksum.join('')) {
152 throw new e2e.openpgp.error.ParseError(
153 'ASCII Armor checksum incorrect.');
154 }
155 charsetMatch = armor[3].match(/^Charset: ([^\r\n]+)\r?\n/im);
156 if (charsetMatch) {
157 charset = charsetMatch[1].toLowerCase().match(/[\w-]+/)[0] || 'utf-8';
158 } else {
159 charset = undefined;
160 }
161 validArmors.push({
162 'data': payload,
163 'charset': charset,
164 'type': armor[2],
165 'startOffset': armor.index + prefixNewline.length,
166 'endOffset': armorRe.lastIndex - suffixNewline.length});
167 if (goog.isDef(opt_limit)) {
168 opt_limit--;
169 }
170 }
171 return validArmors;
172};

175/**
176 * Parses ASCII Armor ClearSign messages.
177 * Specified in RFC 4880 Section 6.2.
178 * Throws a {@code e2e.openpgp.error.ParseError} if the Armor is invalid.
179 * @param {string} text The text to parse as ASCII Armor.
180 * @return {!e2e.openpgp.ClearSignMessage} Parsed message parameters
181 * and the signature ByteArray.
182 */
183e2e.openpgp.asciiArmor.parseClearSign = function(text) {
184 var startMessage = text.indexOf('-----BEGIN PGP SIGNED MESSAGE-----');
185 var startSignature = text.indexOf('-----BEGIN PGP SIGNATURE-----');
186 var armor = text.substr(startMessage, startSignature - startMessage).match(
187 new RegExp('^-----BEGIN PGP SIGNED MESSAGE-----\\r?\\n' +
188 'Hash:[ ]([^\\n\\r]+)\\r?\\n' + // Hash header
189 '(?:[A-Za-z]+:[ ][^\\n\\r]+\\r?\\n)*' + // Other headers
190 '\\r?\\n')); // New line
191 if (!armor) {
192 throw new e2e.openpgp.error.ParseError('invalid clearsign format');
193 }
194 var hashString = armor[1];
195 var startBody = text.indexOf('\n\n') + 2;
196 if (startBody == (-1 + 2)) {
197 startBody = text.indexOf('\r\n\r\n') + 4;
198 }
199 var body = text.substr(startBody,
200 startSignature - startBody - 1); // -1 to remove \n
201 if (goog.string.endsWith(body, '\r')) {
202 body = goog.string.removeAt(body, body.length - 1, 1); // Remove ending \r
203 }
204 body = e2e.openpgp.asciiArmor.dashUnescape(body);
205 body = e2e.openpgp.asciiArmor.convertNewlines(body);
206 var signature = e2e.openpgp.asciiArmor.parse(text.substr(startSignature));
207 return new e2e.openpgp.ClearSignMessage(body, signature.data, hashString);
208};

222/**
223 * Checks if the message has a clearsign message format
224 * @param {string} text
225 * @return {!boolean} true if the message has a clearsign message format.
226 */
227e2e.openpgp.asciiArmor.isClearSign = function(text) {
228 var startMessage = text.indexOf('-----BEGIN PGP SIGNED MESSAGE-----');
229 var startSignature = text.indexOf('-----BEGIN PGP SIGNATURE-----');
230 return Boolean(startMessage !== -1 &&
231 startSignature !== -1 &&
232 startSignature > startMessage);

236/**
237 * Dash-Escapes Text as described in RFC4880 7.1.
238 * @param {string} plaintext The plaintext that has already been through
239 * e2e.openpgp.asciiArmor.convertNewlines().
240 * @protected
241 * @return {string} The dash-escaped text.
242 */
243e2e.openpgp.asciiArmor.dashEscape = function(plaintext) {
244 return (plaintext.replace(/^\-/gm, '\- -') // Dash-escape leading -
245 .replace(/^From /gm, '\- From ') // Dash-escape leading "From "
246 .replace(/[\t ]*$/gm, '')); // Remove trailing tabs and spaces.
247};

250/**
251 * Removes the Dash-Escaping as described in RFC4880 7.1.
252 * @param {string} plaintext Text with optional dash-escapes
253 * @protected
254 * @return {string} The text with removed dash-escapes.
255 */
256e2e.openpgp.asciiArmor.dashUnescape = function(plaintext) {
257 return (plaintext.replace(/^\- /gm, ''));

261/**
262 * Construct a cleartext signature ASCII Armor.
263 * Specified in RFC 4880 Section 7.
264 * @param {e2e.openpgp.ClearSignMessage} message The message.
265 * @param {!Object.<string>=} opt_headers Extra headers to add to signature.
266 * @return {string} The ASCII Armored text.
267 */
268e2e.openpgp.asciiArmor.encodeClearSign = function(message, opt_headers) {
269 return ['-----BEGIN PGP SIGNED MESSAGE-----',
270 'Hash: ' + message.getSignature().hashAlgorithm,
271 '',
272 e2e.openpgp.asciiArmor.dashEscape(
273 e2e.openpgp.asciiArmor.convertNewlines(message.getBody())),
274 e2e.openpgp.asciiArmor.encode('SIGNATURE',
275 message.getSignature().serialize(), opt_headers)
276 ].join('\r\n');
277};

280/**
281 * Encode data as ASCII Armor, with a trailing new line characters (\r\n).
282 * Specified in RFC 4880 Section 6.2.
283 * @param {string} type Descriptive type, such as "MESSAGE".
284 * @param {!e2e.ByteArray} payload The data to encode.
285 * @param {!Object.<string>=} opt_headers Extra headers to add.
286 * @return {string} The ASCII Armored text.
287 */
288e2e.openpgp.asciiArmor.encode = function(type, payload, opt_headers) {
289 var byteChecksum = e2e.dwordArrayToByteArray(
290 [e2e.openpgp.asciiArmor.crc24_(payload)]);
291 var checksum = e2e.openpgp.asciiArmor.encodeRadix64_(
292 byteChecksum.slice(-3));
293 var headers = [];
294 if (type !== 'SIGNATURE') {
295 headers = ['Charset: UTF-8'];
296 }
297 if (opt_headers) {
298 var headerNames = Object.getOwnPropertyNames(opt_headers);
299 for (var i = 0; i < headerNames.length; i++) {
300 var key = headerNames[i];
301 var val = opt_headers[key];
302 if (key.match(/^\w+$/) && val && val.match(/^[^\r\n]+$/)) {
303 headers.push(key + ': ' + val);
304 }
305 }
306 }
307 return goog.array.flatten(
308 '-----BEGIN PGP ' + type + '-----',
309 headers,
310 '',
311 e2e.openpgp.asciiArmor.encodeRadix64_(payload),
312 '=' + checksum,
313 '-----END PGP ' + type + '-----',
314 ''
315 ).join('\r\n');
316};

319/**
320 * Extracts the PGP block from the free-text content. If no PGP block exists,
321 * returns the original content. If multiple PGP blocks are present, only the
322 * first one is returned.
323 * @param {string} content The original content from which the PGP block is to
324 * be extracted.
325 * @return {string} The first PGP block that is found in the original content.
326 */
327e2e.openpgp.asciiArmor.extractPgpBlock = function(content) {
328 var extractRe =
329 /(.*)-----BEGIN\sPGP\s([\w\s]+)-----[\s\S.]*(?:MESSAGE|BLOCK|SIGNATURE)-----/;
330 var result = extractRe.exec(content);
331 if (result) {
332 var pgpBlock = result[0];
333 var linePrefix = result[1];
334 var firstPrefixType = result[2];
335 var expectedSuffixType = firstPrefixType;
336 if (firstPrefixType == 'SIGNED MESSAGE') {
337 expectedSuffixType = 'SIGNATURE';
338 }
339 // Check if more then one blocks are present.
340 if (/-----BEGIN\sPGP/.test(pgpBlock.substring(1))) {
341 // Cutoff at first Armor Suffix
342 pgpBlock = pgpBlock.replace(new RegExp(
343 '(-----END\\sPGP\\s' + expectedSuffixType + '-----)([\\s\\S.]*)$',
344 'g'),
345 '$1');
346 }
347 if (linePrefix.length > 0) {
348 // Make trailing spaces optional in the line prefix.
349 // They get removed for otherwise empty lines.
350 pgpBlock = pgpBlock.replace(new RegExp(
351 '^' + goog.string.regExpEscape(goog.string.trimRight(linePrefix)) +
352 '[\\t ]*',
353 'gm'),
354 '');
355 }
356 return pgpBlock;
357 } else {
358 return content;
359 }
360};

363/**
364 * Marks an ASCII Armor as a draft message.
365 * @param {string} armoredContent The ASCII Armor to mark as draft.
366 * @return {string} The marked ASCII Armor.
367 */
368e2e.openpgp.asciiArmor.markAsDraft = function(armoredContent) {
