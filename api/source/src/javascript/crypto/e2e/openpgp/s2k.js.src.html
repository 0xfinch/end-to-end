109/**
110 * Parses (and extracts) an S2K object from the given ByteArray.
111 * Throws e2e.openpgp.error.Error if it fails.
112 * @param {!e2e.ByteArray} bytes The ByteArray to extract the S2K from.
113 * @return {e2e.openpgp.S2k} The generated S2K instance.
114 */
115e2e.openpgp.S2k.parse = function(bytes) {
116 var type = bytes.shift();
117 if (!goog.object.containsValue(e2e.openpgp.S2k.Type, type)) {
118 throw new e2e.openpgp.error.ParseError('Invalid S2K type.');
119 }
120 type = /** @type {e2e.openpgp.S2k.Type} */ (type);
121 var hid = bytes.shift();
122 var hash;
123 if (type == e2e.openpgp.S2k.Type.DUMMY && !hid) {
124 // Allow empty hash algo for dummy s2k.
125 hash = null;
126 } else {
127 hash = e2e.openpgp.constants.getInstance(
128 e2e.openpgp.constants.Type.HASH, hid);
129 }
130 hash = /** @type {e2e.hash.Hash} */ (hash);
131 var salt, encodedCount;
132 if (type === e2e.openpgp.S2k.Type.SALTED ||
133 type === e2e.openpgp.S2k.Type.ITERATED) {
134 salt = bytes.splice(0, 8);
135 if (salt.length != 8) {
136 throw new e2e.openpgp.error.ParseError('Invalid S2K packet.');
137 }
138 if (type === e2e.openpgp.S2k.Type.ITERATED) {
139 encodedCount = bytes.shift();
140 return new e2e.openpgp.IteratedS2K(hash, salt, encodedCount);
141 } else {
142 return new e2e.openpgp.SaltedS2K(hash, salt);
143 }
144 } else if (type === e2e.openpgp.S2k.Type.SIMPLE) {
145 return new e2e.openpgp.SimpleS2K(hash);
146 } else if (type === e2e.openpgp.S2k.Type.DUMMY) {
147 /* We're basing this on GnuPG's dummy S2k extension. First three bytes are
148 * 'E2E' or 'GPG', last byte is a mode. See the enums defined below for
149 * the values of these modes.
150 */
151 var header = bytes.splice(0, 3);
152 if (header.length === 3 && bytes.length >= 1) {
153 return new e2e.openpgp.DummyS2k(hash, header, bytes.shift());
154 }
155 }
156 // TODO(evn): Implement a scrypt KDF as a new S2K type.
157 throw new e2e.openpgp.error.ParseError('Invalid S2K type.');
158};

162/**
163 * Implements a dummy S2k algorithm for E2E.
164 * @param {e2e.hash.Hash} hash An instance of the hash algorithm to use.
165 * @param {!e2e.ByteArray} header ['E','2','E'] or ['G','N','U']
166 * @param {number} mode A byte indicating S2k mode
167 * @constructor
168 * @extends {e2e.openpgp.S2k}
169 */
170e2e.openpgp.DummyS2k = function(hash, header, mode) {
171 goog.asserts.assert(
172 header.length === e2e.openpgp.DummyS2k.E2E_HEADER.length &&
173 header.length === e2e.openpgp.DummyS2k.GPG_HEADER.length);
174 var is_e2e = goog.array.equals(header, e2e.openpgp.DummyS2k.E2E_HEADER);
175 var is_gpg = goog.array.equals(header, e2e.openpgp.DummyS2k.GPG_HEADER);
176
177 this.dummy = is_e2e ? e2e.openpgp.DummyS2k.DummyTypes.E2E :
178 e2e.openpgp.DummyS2k.DummyTypes.GPG;
179 var mode_enum = is_e2e ? e2e.openpgp.DummyS2k.E2E_modes :
180 e2e.openpgp.DummyS2k.GPG_modes;
181 if (!goog.object.containsValue(mode_enum, mode)) {
182 throw new e2e.openpgp.error.ParseError('Invalid S2k mode.');
183 }
184
185 if (is_e2e) {
186 this.mode = /** @type {e2e.openpgp.DummyS2k.E2E_modes}*/ (mode);
187 } else if (is_gpg) {
188 this.mode = /** @type {e2e.openpgp.DummyS2k.GPG_modes}*/ (mode);
189 } else {
190 throw new e2e.openpgp.error.ParseError('Invalid dummy S2k header!');
191 }
192 goog.base(this, hash);
193};
194goog.inherits(e2e.openpgp.DummyS2k, e2e.openpgp.S2k);

283/** @inheritDoc */
284e2e.openpgp.SimpleS2K.prototype.getKey = function(passphrase, length) {
285 var hashed = [], original_length = length;
286 while (length > 0) {
287 // Hash the passphrase repeatedly by appending 0's at the beginning.
288 var checksum = this.hash.hash(passphrase);
289 passphrase.unshift(0);
290 length -= checksum.length;
291 goog.array.extend(hashed, checksum);
292 }
293 return hashed.slice(0, original_length);
294};

341/**
342 * Implements the Iterated S2K algorithm. Throws e2e.openpgp.error.Error if
343 * it fails.
344 * @param {e2e.hash.Hash} hash An instance of the hash algorithm to use.
345 * @param {!e2e.ByteArray} salt The salt to use for the S2K.
346 * @param {number} encodedCount The encoded number of bytes to be hashed.
347 * @constructor
348 * @extends {e2e.openpgp.SimpleS2K}
349 */
350e2e.openpgp.IteratedS2K = function(hash, salt, encodedCount) {
351 goog.base(this, hash);
352 if (salt.length != 8 || !e2e.isByteArray(salt)) {
353 throw new e2e.openpgp.error.InvalidArgumentsError('Invalid salt.');
354 }
355 if (!e2e.isByte(encodedCount)) {
356 throw new e2e.openpgp.error.InvalidArgumentsError(
357 'Invalid encoded count.');
358 }
359 /**
360 * The salt to use for the S2K. Must be 8 bytes long.
361 * @type {!e2e.ByteArray}
362 * @private
363 */
364 this.salt_ = salt;
365 /**
366 * The encoded number of bytes to be hashed.
367 * @type {number}
368 * @private
369 */
370 this.encodedCount_ = encodedCount;
371 /**
372 * The decoded number of bytes to be hashed.
373 * @type {number}
374 * @private
375 */
376 this.count_ = e2e.openpgp.IteratedS2K.getCount_(encodedCount);
377};
378goog.inherits(e2e.openpgp.IteratedS2K, e2e.openpgp.SimpleS2K);

385/** @inheritDoc */
386e2e.openpgp.IteratedS2K.prototype.getKey = function(passphrase, length) {
387 var salted_passphrase = this.salt_.concat(passphrase);
388 var count = this.count_;
389
390 if (count < salted_passphrase.length) {
391 count = salted_passphrase.length;
392 }
393
394 // Construct an array with multiple copies of salted_passphrase. This enables
395 // us to pass block_size chunks of salted_passphrase into the hash function.
396 // This runs twice as fast as the naive approach.
397 var block_size = this.hash.blockSize;
398 var reps = Math.ceil(block_size / salted_passphrase.length) + 1;
399 var repeated = goog.array.flatten(goog.array.repeat(salted_passphrase, reps));
400 var slices = [];
401 for (var i = 0; i < salted_passphrase.length; i++) {
402 slices.push(repeated.slice(i, i + block_size));
403 }
404
405 var num_zero_prepend = 0;
406 var hashed = [], original_length = length;
407 while (length > 0) { // Loop to handle when checksum len < length requested.
408 this.hash.reset();
409 var remaining = count; // Number of input bytes we still want.
410 if (num_zero_prepend > 0) {
411 var firstRound = goog.array.repeat(0, num_zero_prepend);
412 // Align initial hash input size to block size.
413 var size = (block_size < remaining) ? block_size : remaining;
414 size -= num_zero_prepend;
415 if (size < 0) {
416 size = 0;
417 }
418 goog.array.extend(firstRound, repeated.slice(0, size));
419 this.hash.update(firstRound);
420 remaining -= size;
421 }
422 while (remaining > 0) {
423 var offset = (count - remaining) % salted_passphrase.length;
424 var size = (block_size < remaining) ? block_size : remaining;
425 if (size == block_size) {
426 this.hash.update(slices[offset]);
427 } else {
428 this.hash.update(repeated.slice(offset, offset + size));
429 }
430 remaining -= size;
431 }
432 var checksum = this.hash.digest();
433 length -= checksum.length;
434 goog.array.extend(hashed, checksum);
435 num_zero_prepend += 1;
436 }
437 return hashed.slice(0, original_length);
438};

459/**
460 * Returns the number of iterations for a given count using the algorithm
461 * defined in RFC 4880 Section 3.7.1.3. Must be in the range [0-255].
462 * Throws e2e.openpgp.error.Error if it fails.
463 * @param {number} c The encoded count for.
464 * @return {number} The number of bytes to be hashed.
465 * @private
466 */
467e2e.openpgp.IteratedS2K.getCount_ = function(c) {
468 if (e2e.isByte(c)) {
469 return (0x10 + (c & 0x0F)) << ((c >> 4) +
470 e2e.openpgp.IteratedS2K.EXPBIAS_);
471 } else {
472 throw new e2e.openpgp.error.InvalidArgumentsError(
473 'Invalid encoded count.');
474 }
