49/**
50 * Implements a key ring that exposes basic key management features such as
51 * generating, searching, importing, exporting keys, etc. The key ring shall
52 * be stored in browser's local storage, and shall be encrypted if the user
53 * provides a passphrase.
54 * @param {!e2e.openpgp.LockableStorage} lockableStorage persistent
55 * storage mechanism.
56 * @param {string=} opt_keyServerUrl The optional http key server url. If not
57 * specified then only support key operation locally.
58 * @constructor
59 */
60e2e.openpgp.KeyRing = function(lockableStorage, opt_keyServerUrl) {
61 this.localStorage_ = lockableStorage;
62 if (goog.isDefAndNotNull(opt_keyServerUrl)) {
63 this.keyClient_ = new e2e.openpgp.KeyClient(opt_keyServerUrl);
64 }
65 this.pubKeyRing_ = new goog.structs.Map();
66 this.privKeyRing_ = new goog.structs.Map();
67 this.keyGenerator_ = new e2e.openpgp.KeyGenerator();
68 /** @private {boolean} */
69 this.initialized_ = false;
70};

73/**
74 * Creates and initializes the KeyRing object with an unlocked storage.
75 * @param {!e2e.openpgp.LockableStorage} lockableStorage persistent
76 * storage mechanism. Storage must already be unlocked, otherwise this method
77 * will return a {@link e2e.openpgp.error.MissingPassphraseError}.
78 * @param {string=} opt_keyServerUrl The optional http key server url. If not
79 * specified then only support key operation locally.
80 * @return {!goog.async.Deferred.<!e2e.openpgp.KeyRing>} The initialized
81 * keyring.
82 */
83e2e.openpgp.KeyRing.launch = function(lockableStorage, opt_keyServerUrl) {
84 var keyRing = new e2e.openpgp.KeyRing(lockableStorage, opt_keyServerUrl);
85 var returnKeyRing = goog.functions.constant(keyRing);
86 return /** @type {!goog.async.Deferred.<!e2e.openpgp.KeyRing>} */ (
87 keyRing.initialize().addCallback(returnKeyRing));
88};

181/**
182 * Imports a key block to the key ring.
183 * @param {!e2e.openpgp.block.TransferableKey} keyBlock The key block to
184 * import.
185 * @param {!e2e.ByteArray=} opt_passphrase The passphrase to use to
186 * import the key.
187 * @return {!goog.async.Deferred<boolean>} If the key import was successful.
188 */
189e2e.openpgp.KeyRing.prototype.importKey = function(
190 keyBlock, opt_passphrase) {
191 var result = e2e.async.Result.toResult(undefined);
192 return result.addCallback(function() {
193 var keys = [keyBlock.keyPacket].concat(keyBlock.subKeys);
194 var keyRing;
195 if (keyBlock instanceof e2e.openpgp.block.TransferablePublicKey) {
196 keyRing = this.pubKeyRing_;
197 } else if (keyBlock instanceof e2e.openpgp.block.TransferableSecretKey) {
198 keyRing = this.privKeyRing_;
199 } else {
200 return false;
201 }
202 // This will throw on signature verification failures.
203 keyBlock.processSignatures();
204 var uids = keyBlock.getUserIds();
205 goog.array.removeDuplicates(uids);
206 var importedKeysResults = goog.async.DeferredList.gatherResults(
207 goog.array.map(uids, function(uid) {
208 return this.importKey_(uid, keyBlock, keyRing, opt_passphrase);
209 }, this));
210 return importedKeysResults.addCallback(function(importedKeys) {
211 // Return true only if the key was imported for all the uids.
212 return (importedKeys.indexOf(false) == -1);
213 });
214 }, this);
215};

231/**
232 * Generates and imports to the key ring a master signing key and a subordinate
233 * encryption key.
234 * @param {string} uid User ID to associate the key with.
235 * @param {!e2e.signer.Algorithm} keyAlgo Algorithm of the master key.
236 * It must be one of the digital signature algorithms.
237 * @param {number} keyLength Length in bits of the master key.
238 * @param {e2e.cipher.Algorithm} subkeyAlgo Algorithm of the subkey.
239 * It must be one of the cipher algorithms.
240 * @param {number} subkeyLength Length in bits of the subkey.
241 * @param {!e2e.algorithm.KeyLocations=} opt_keyLocation Where should the key be
242 * stored? (default to JS)
243 * @return {e2e.async.Result.<!Array.<!e2e.openpgp.block.TransferableKey>>}
244 * The generated public key and secret key in an array.
245 */
246e2e.openpgp.KeyRing.prototype.generateKey = function(uid,
247 keyAlgo,
248 keyLength,
249 subkeyAlgo,
250 subkeyLength,
251 opt_keyLocation) {
252 return this.keyGenerator_.generateKey(uid, keyAlgo, keyLength, subkeyAlgo,
253 subkeyLength, opt_keyLocation).addCallback(function(keys) {
254 goog.array.forEach(keys, function(key) {
255 if (key instanceof e2e.openpgp.block.TransferableSecretKey) {
256 this.importKey_(uid, key, this.privKeyRing_);
257 } else if (key instanceof e2e.openpgp.block.TransferablePublicKey) {
258 this.importKey_(uid, key, this.pubKeyRing_);
259 if (this.keyClient_ !== null) {
260 this.keyClient_.importPublicKey(key);
261 }
262 }
263 }, this);
264 }, this);
265};

296/**
297 * Obtains the key with a given keyId. If opt_secret is set, it only returns
298 * secret keys.
299 * @param {!e2e.ByteArray} keyId The key id to search for.
300 * @param {boolean=} opt_secret Whether to search the private key ring.
301 * @param {!e2e.openpgp.KeyFingerprint=} opt_fingerprint The key packet has
302 * to come from a key with this fingerprint.
303 * @return {?e2e.openpgp.packet.Key} The key packet with that key id or null.
304 * @private
305 */
306e2e.openpgp.KeyRing.prototype.getKey_ = function(keyId, opt_secret,
307 opt_fingerprint) {
308 var keyRing = opt_secret ? this.privKeyRing_ : this.pubKeyRing_;
309 var result = null;
310 // Using goog.array.some to break on first match.
311 goog.array.some(goog.array.flatten(keyRing.getValues()), function(key) {
312 if (goog.isDefAndNotNull(opt_fingerprint) &&
313 !e2e.compareByteArray(key.keyPacket.fingerprint, opt_fingerprint)) {
314 return false;
315 }
316 // Check main key packet, then the subkeys.
317 if (e2e.compareByteArray(keyId, key.keyPacket.keyId)) {
318 result = key.keyPacket;
319 return true;
320 }
321 return goog.array.some(key.subKeys, function(subKey) {
322 if (e2e.compareByteArray(keyId, subKey.keyId)) {
323 result = subKey;
324 return true;
325 }
326 return false;
327 });
328 });
329 return result;
330};

333/**
334 * Obtains a key block corresponding to the given key object or null.
335 * @param {!e2e.openpgp.Key} keyObject
336 * @return {?e2e.openpgp.block.TransferableKey}

338e2e.openpgp.KeyRing.prototype.getKeyBlock = function(keyObject) {
339 if (!keyObject.key.secret) {

369/**
370 * Locks a TransferableSecretKey with keyring passphrase. Used to prevent
371 * exporting unencrypted secret keys. Operates on a copy of key argument.
372 * @param {!e2e.openpgp.block.TransferableKey} key
373 * @return {?e2e.openpgp.block.TransferableKey}
374 * @private
375 */
376e2e.openpgp.KeyRing.prototype.lockSecretKey_ = function(key) {
377 if (!key) {
378 return key;
379 }
380 // Only let SecretKeys out if they are encrypted.
381 if (key instanceof e2e.openpgp.block.TransferableSecretKey) {
382 var serialized = key.serialize();
383 var parsed = /** @type {!e2e.openpgp.block.TransferableSecretKey} */ (
384 e2e.openpgp.block.factory.parseByteArrayTransferableKey(serialized));
385 parsed.processSignatures();
386 parsed.unlock();
387 var success = false;
388 if (this.localStorage_.getPassphrase()) {
389 success = parsed.lock(e2e.stringToByteArray(
390 this.localStorage_.getPassphrase()));
391 } else {
392 success = parsed.lock();
393 }
394 return success ? parsed : null;
395 }
396 return key;
397};

441/**
442 * Internal implementation of {@link #searchKey}.
443 * @param {(string|function(string): boolean)} uidOrMatcher User ID or a User ID
444 * matching function - see {@link #searchKeyByUidMatcher}.
445 * @param {e2e.openpgp.KeyRing.Type=} opt_type Key type to search for.
446 * @return {?Array.<!e2e.openpgp.block.TransferableKey>} An array of keys
447 * matching the criteria.
448 * @private
449 */
450e2e.openpgp.KeyRing.prototype.searchKeyInternal_ = function(uidOrMatcher,
451 opt_type) {
452 if (!opt_type || opt_type == e2e.openpgp.KeyRing.Type.PUBLIC) {
453 return this.searchKey_(this.pubKeyRing_, uidOrMatcher);
454 }
455 if (opt_type == e2e.openpgp.KeyRing.Type.PRIVATE) {
456 return this.searchKey_(this.privKeyRing_, uidOrMatcher);
457 }
458 if (opt_type == e2e.openpgp.KeyRing.Type.ALL) {
459 var keys = [];
460 var priv = this.searchKey_(this.privKeyRing_, uidOrMatcher);
461 if (priv) { // Do this to avoid having a null element.
462 goog.array.extend(keys, priv);
463 }
464 var pub = this.searchKey_(this.pubKeyRing_, uidOrMatcher);
465 if (pub) {
466 goog.array.extend(keys, pub);
467 }
468 return keys;
469 }
470 return null;
471};

493/**
494 * Searches a public or private key for a User ID asynchronously. The search is
495 * first performed locally. If the key is not found locally and we're searching
496 * for public key, then searches the public key in the http key server and
497 * imports the the found key to keyring.
498 * @param {string} uid User ID to search for, or empty to search all.
499 * @param {e2e.openpgp.KeyRing.Type=} opt_type Key type to search for.
500 * @return {!e2e.async.Result.<!Array.<!e2e.openpgp.block.TransferableKey>>}
501 * An array of keys for the given User ID or [] if not found.
502 */
503e2e.openpgp.KeyRing.prototype.searchKeyLocalAndRemote = function(uid,
504 opt_type) {
505 var resultKeys = new e2e.async.Result();
506 var localKeys = this.searchKey(uid, opt_type);
507 if (localKeys != null) {
508 resultKeys.callback(localKeys);
509 } else if (opt_type == e2e.openpgp.KeyRing.Type.PUBLIC) {
510 this.searchPublicKeyRemote_(uid).addCallback(function(pubKeys) {
511 resultKeys.callback(pubKeys);
512 });
513 } else {
514 resultKeys.callback([]);
515 }
516 return resultKeys;
517};

520/**
521 * Gets all of the keys in the keyring.
522 * @param {boolean=} opt_privOnly If true, fetch only private keys.
523 * @return {!e2e.openpgp.TransferableKeyMap} A clone of the key ring maps.
524 */
525e2e.openpgp.KeyRing.prototype.getAllKeys = function(opt_privOnly) {
526 if (opt_privOnly) {
527 return this.privKeyRing_.clone();
528 }
529 // Return a map of user IDs to an Array of both public and secret keys.
530 var keys = this.pubKeyRing_.clone();
531 var ids = this.privKeyRing_.getKeys();
532 var values = this.privKeyRing_.getValues();
533 for (var i = 0; i < ids.length; i++) {
534 keys.set(ids[i], goog.array.concat(values[i], keys.get(ids[i], [])));
535 }
536 return keys;
537};

551/**
552 * Deletes a private or public key that has a given key fingerprint from chosen
553 * keyring. Use e2e.openpgp.KeyRing.Type.ALL to delete the whole keypair.
554 * @param {!e2e.openpgp.KeyFingerprint} fingerprint The fingerprint.
555 * @param {!e2e.openpgp.KeyRing.Type} keyRingType The keyring to delete the key
556 * from.
557 */
558e2e.openpgp.KeyRing.prototype.deleteKeyByFingerprint = function(fingerprint,
559 keyRingType) {
560 var keyRings = [];
561 switch (keyRingType) {
562 case e2e.openpgp.KeyRing.Type.PUBLIC:
563 keyRings = [this.pubKeyRing_];
564 break;
565 case e2e.openpgp.KeyRing.Type.PRIVATE:
566 keyRings = [this.privKeyRing_];
567 break;
568 case e2e.openpgp.KeyRing.Type.ALL:
569 keyRings = [this.pubKeyRing_, this.privKeyRing_];
570 break;
571 }
572 goog.array.forEach(keyRings, function(keyring) {
573 var uidsToRemove = [];
574 keyring.forEach(function(keys, uid) {
575 var hadMatchingKeys = goog.array.removeIf(keys, function(key) {
576 return e2e.compareByteArray(key.keyPacket.fingerprint,
577 fingerprint);
578 });
579 if (hadMatchingKeys && keys.length == 0) {
580 uidsToRemove.push(uid);
581 }
582 }, this);
583 // Remove User ID entries with no keys left.
584 goog.array.forEach(uidsToRemove, function(uid) {
585 keyring.remove(uid);
586 });
587 }, this);
588};

626/**
627 * Searches a key in a key ring associated with a given User ID.
628 * @param {!e2e.openpgp.TransferableKeyMap} keyRing The key ring to search.
629 * @param {(string|function(string): boolean)} uidOrMatcher User ID or a User ID
630 * matching function - see {@link #searchKeyByUidMatcher}.
631 * @return {Array.<!e2e.openpgp.block.TransferableKey>} An array of keys for
632 * that User ID or null if no key is found.
633 * @private
634 */
635e2e.openpgp.KeyRing.prototype.searchKey_ = function(keyRing, uidOrMatcher) {
636 if (goog.isString(uidOrMatcher)) {
637 var uid = uidOrMatcher;
638 return keyRing.get(uid) ?
639 goog.array.clone(keyRing.get(uid)) : null;
640 } else if (goog.isFunction(uidOrMatcher)) {
641 var matcher = uidOrMatcher;
642 var keys = goog.array.flatten(goog.array.filter(

688/**
689 * Imports a new key associated with a User ID to the key ring. Does not add the
690 * key if there is already a matching Key ID.
691 * @param {string} uid User ID associated with the key.
692 * @param {!e2e.openpgp.block.TransferableKey} keyBlock The key to import.
693 * @param {!e2e.openpgp.TransferableKeyMap} keyRing The keyring to add the keys
694 * to.
695 * @param {!e2e.ByteArray=} opt_passphrase The passphrase used to
696 * protect the key.
697 * @return {e2e.async.Result<boolean>} If the key import was successful.
698 * False if the key for a given User ID with the same Key ID already
699 * exists in a keyring.
700 * @private
701 */
702e2e.openpgp.KeyRing.prototype.importKey_ = function(
703 uid, keyBlock, keyRing, opt_passphrase) {
704 var uidKeyBlocks = [], addKey = false;
705 var keys = [keyBlock.keyPacket].concat(keyBlock.subKeys);
706 goog.array.forEach(keys, function(key) {
707 if (key instanceof e2e.openpgp.packet.SecretKey) {
708 key.cipher.unlockKey(opt_passphrase);
709 // Re-lock the key in plaintext.
710 key.cipher.lockKey();
711 }
712 });
713 if (keyRing.containsKey(uid)) {
714 uidKeyBlocks = keyRing.get(uid);
715 addKey = !goog.array.some(uidKeyBlocks, function(uidKeyBlock) {
716 return goog.isDef(keyBlock.keyPacket.keyId) && e2e.compareByteArray(
717 uidKeyBlock.keyPacket.keyId, keyBlock.keyPacket.keyId);
718 });
719 // TODO(evn): Merge information when the key block is already in keyring.
720 } else {
721 addKey = true;
722 }
723 if (addKey) {
724 keyRing.set(uid, uidKeyBlocks.concat([keyBlock]));
725 return this.persist_().addCallback(function() {
726 return true;
727 });
728 }
729 return e2e.async.Result.toResult(false);
730};

773 * from lockable storage to memory.

785 return result;

787
788 // Try to unlock the storage.
789 if (this.localStorage_.isLocked()) {
790 result.addCallback(function() {

794 throw new e2e.openpgp.error.WrongPassphraseError();
