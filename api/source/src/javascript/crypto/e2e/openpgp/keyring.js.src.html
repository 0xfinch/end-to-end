68/**
69 * Implements a key ring that exposes basic key management features such as
70 * generating, searching, importing, exporting keys, etc. The key ring shall
71 * be stored in browser's local storage, and shall be encrypted if the user
72 * provides a passphrase.
73 * @param {string} passphrase The passphrase used to encrypt the keyring.
74 * @param {!goog.storage.mechanism.Mechanism} storageMechanism persistent
75 * storage mechanism.
76 * @param {string=} opt_keyServerUrl The optional http key server url. If not
77 * specified then only support key operation locally.
78 * @constructor
79 */
80e2e.openpgp.KeyRing = function(passphrase, storageMechanism, opt_keyServerUrl) {
81 this.localStorage_ = storageMechanism;
82 if (goog.isDefAndNotNull(opt_keyServerUrl)) {
83 this.keyClient_ = new e2e.openpgp.KeyClient(opt_keyServerUrl);
84 }
85 this.pubKeyRing_ = new goog.structs.Map();
86 this.privKeyRing_ = new goog.structs.Map();
87 this.passphrase_ = passphrase;
88 this.readKeyData_();
89};

240/**
241 * Imports a key block to the key ring.
242 * @param {!e2e.openpgp.block.TransferableKey} keyBlock The key block to
243 * import.
244 * @param {!e2e.ByteArray=} opt_passphrase The passphrase to use to
245 * import the key.
246 * @return {boolean} If the key import was successful.
247 */
248e2e.openpgp.KeyRing.prototype.importKey = function(
249 keyBlock, opt_passphrase) {
250 var keys = [keyBlock.keyPacket].concat(keyBlock.subKeys);
251 var keyRing;
252 if (keyBlock instanceof e2e.openpgp.block.TransferablePublicKey) {
253 keyRing = this.pubKeyRing_;
254 } else if (keyBlock instanceof e2e.openpgp.block.TransferableSecretKey) {
255 keyRing = this.privKeyRing_;
256 } else {
257 return false;
258 }
259 // This will throw on signature verification failures.
260 keyBlock.processSignatures();
261 var uids = keyBlock.getUserIds();
262 goog.array.removeDuplicates(uids);
263 var importedKeys = goog.array.map(uids, function(uid) {
264 return this.importKey_(uid, keyBlock, keyRing, opt_passphrase);
265 }, this);
266 // Return false if any key failed to import (e.g. because it already existed).
267 return importedKeys.indexOf(false) > -1;
268};

284/**
285 * Generates the next key in the sequence based on the ECC generation seed
286 * @param {number} keyLength Length in bits of the key.
287 * @private
288 * @return {!e2e.ByteArray} Deterministic privKey based on ECC seed.
289 */
290e2e.openpgp.KeyRing.prototype.getNextKey_ = function(keyLength) {
291 if (!this.eccSeed_) {
292 this.eccSeed_ = e2e.random.getRandomBytes(
293 e2e.openpgp.KeyRing.ECC_SEED_SIZE);
294 this.eccCount_ = 0;
295 }
296
297 if (++this.eccCount_ > 0x7F) {
298 throw new e2e.openpgp.error.UnsupportedError('Too many ECC keys generated');
299 }
300
301 if (keyLength % 8) {
302 throw new e2e.openpgp.error.UnsupportedError(
303 'Key length is not a multiple of 8');
304 }
305
306 return new e2e.Hkdf(new e2e.hash.Sha256()).getHKDF(this.eccSeed_,
307 e2e.dwordArrayToByteArray([this.eccCount_]), keyLength / 8);
308
309};

312/**
313 * Generates and imports to the key ring a master signing key and a subordinate
314 * encryption key.
315 * @param {string} email The email to associate the key with.
316 * @param {!e2e.signer.Algorithm} keyAlgo Algorithm of the master key.
317 * It must be one of the digital signature algorithms.
318 * @param {number} keyLength Length in bits of the master key.
319 * @param {e2e.cipher.Algorithm} subkeyAlgo Algorithm of the subkey.
320 * It must be one of the cipher algorithms.
321 * @param {number} subkeyLength Length in bits of the subkey.
322 * @param {!e2e.algorithm.KeyLocations=} opt_keyLocation Where should the key be
323 * stored? (default to JS)
324 * @return {e2e.async.Result.<!Array.<!e2e.openpgp.block.TransferableKey>>}
325 * The generated public key and secret key in an array.
326 */
327e2e.openpgp.KeyRing.prototype.generateKey = function(email,
328 keyAlgo,
329 keyLength,
330 subkeyAlgo,
331 subkeyLength,
332 opt_keyLocation) {
333 var keyData = {
334 'pubKey': new Array(),
335 'privKey': new Array()
336 };
337 if (!goog.isDef(opt_keyLocation)) {
338 opt_keyLocation = e2e.algorithm.KeyLocations.JAVASCRIPT;
339 }
340
341 if (opt_keyLocation == e2e.algorithm.KeyLocations.JAVASCRIPT) {
342 var fingerprint;
343 if (keyAlgo == e2e.signer.Algorithm.ECDSA &&
344 keyLength == 256) {
345 var ecdsa = e2e.openpgp.keygenerator.newEcdsaWithP256(
346 this.getNextKey_(keyLength));
347 this.extractKeyData_(keyData, ecdsa);
348 fingerprint = keyData.pubKey[0].fingerprint;
349 }
350 if (subkeyAlgo == e2e.cipher.Algorithm.ECDH &&
351 subkeyLength == 256) {
352 var ecdh = e2e.openpgp.keygenerator.newEcdhWithP256(
353 this.getNextKey_(subkeyLength));
354 this.extractKeyData_(keyData, ecdh, true);
355 }
356 return e2e.async.Result.toResult(this.certifyKeys_(email, keyData));
357 } else if (opt_keyLocation == e2e.algorithm.KeyLocations.WEB_CRYPTO) {
358 if (keyAlgo == e2e.signer.Algorithm.RSA) {
359 if ((keyLength != 4096 && keyLength != 8192) ||
360 subkeyAlgo != e2e.cipher.Algorithm.RSA || subkeyLength != keyLength) {
361 throw new e2e.openpgp.error.UnsupportedError(
362 'WebCrypto RSA keyLength must be 4096 or 8192');
363 }
364 return e2e.openpgp.keygenerator.newWebCryptoRsaKeys(
365 keyLength).addCallback(
366 function(ciphers) {
367 this.extractKeyData_(keyData, ciphers[0]);
368 this.extractKeyData_(keyData, ciphers[1]);
369 return this.certifyKeys_(email, keyData);
370 });
371 }
372 } else if (opt_keyLocation == e2e.algorithm.KeyLocations.HARDWARE) {
373 // TODO(user): https://code.google.com/p/end-to-end/issues/detail?id=130
374 throw new e2e.openpgp.error.UnsupportedError(
375 'Hardware keygen not supported yet');
376 }
377 // Should never happen.
378 throw new e2e.openpgp.error.UnsupportedError(
379 'Unsupported key type or length.');
380};

383/**
384 * @param {string} email The email to associate the key with.
385 * @param {{privKey: Array, pubKey: Array}} keyData
386 * @return {!Array.<!e2e.openpgp.block.TransferableKey>}
387 * @private
388 */
389e2e.openpgp.KeyRing.prototype.certifyKeys_ = function(email, keyData) {
390 if (keyData['pubKey'].length == 2 && keyData['privKey'].length == 2) {
391 // TODO(evn): Move this code to a .construct.
392 var primaryKey = keyData['privKey'][0];
393 var uid = new e2e.openpgp.packet.UserId(email);
394 uid.certifyBy(primaryKey);
395 keyData['privKey'][1].bindTo(
396 primaryKey,
397 e2e.openpgp.packet.Signature.SignatureType.SUBKEY,
398 e2e.openpgp.packet.SignatureSub.KeyFlags.ENCRYPT_COMMUNICATION |
399 e2e.openpgp.packet.SignatureSub.KeyFlags.ENCRYPT_STORAGE);
400 keyData['pubKey'][1].bindTo(
401 primaryKey,
402 e2e.openpgp.packet.Signature.SignatureType.SUBKEY,
403 e2e.openpgp.packet.SignatureSub.KeyFlags.ENCRYPT_COMMUNICATION |
404 e2e.openpgp.packet.SignatureSub.KeyFlags.ENCRYPT_STORAGE
405 );
406
407 var privKeyBlock = new e2e.openpgp.block.TransferableSecretKey();
408 privKeyBlock.keyPacket = primaryKey;
409 privKeyBlock.subKeys.push(keyData['privKey'][1]);
410 privKeyBlock.userIds.push(uid);
411
412 var pubKeyBlock = new e2e.openpgp.block.TransferablePublicKey();
413 pubKeyBlock.keyPacket = keyData['pubKey'][0];
414 pubKeyBlock.subKeys.push(keyData['pubKey'][1]);
415 pubKeyBlock.userIds.push(uid);
416
417 this.importKey_(email, pubKeyBlock, this.pubKeyRing_);
418 this.importKey_(email, privKeyBlock, this.privKeyRing_);
419
420 // Imports the generated public key to the key server.
421 if (this.keyClient_ != null) {
422 this.keyClient_.importPublicKey(pubKeyBlock);
423 }
424 return [pubKeyBlock, privKeyBlock];
425 }
426 // Should never happen.
427 throw new e2e.openpgp.error.UnsupportedError(
428 'Unsupported key type or length.');
429};

455/**
456 * Obtains the key with a given keyId. If opt_secret is set, it only returns
457 * secret keys.
458 * @param {!e2e.ByteArray} keyId The key id to search for.
459 * @param {boolean=} opt_secret Whether to search the private key ring.
460 * @return {?e2e.openpgp.packet.Key} The key packet with that key id or null.
461 * @private
462 */
463e2e.openpgp.KeyRing.prototype.getKey_ = function(keyId, opt_secret) {
464 var keyRing = opt_secret ? this.privKeyRing_ : this.pubKeyRing_;
465 var result;
466 // Using goog.array.find to break on first match.
467 goog.array.find(goog.array.flatten(keyRing.getValues()), function(key) {
468 if (e2e.compareByteArray(keyId, key.keyPacket.keyId)) {
469 result = key.keyPacket;
470 return true;
471 }
472 return Boolean(goog.array.find(key.subKeys, function(subKey) {
473 if (e2e.compareByteArray(keyId, subKey.keyId)) {
474 result = subKey;
475 return true;
476 }
477 return false;
478 }));
479 });
480 return result || null;
481};

484/**
485 * Obtains a key block corresponding to the given key object or null.
486 * @param {!e2e.openpgp.Key} keyObject
487 * @return {?e2e.openpgp.block.TransferableKey}
488 */
489e2e.openpgp.KeyRing.prototype.getKeyBlock = function(keyObject) {
490 var fingerprint = keyObject.key.fingerprint;
491 var secret = keyObject.key.secret;
492 var keyRing = secret ? this.privKeyRing_ : this.pubKeyRing_;
493 var ret = goog.array.find(
494 goog.array.flatten(keyRing.getValues()),
495 function(keyBlock) {
496 return e2e.compareByteArray(keyBlock.keyPacket.fingerprint,
497 fingerprint);
498 });
499 return this.lockSecretKey_(ret);
500};

503/**
504 * Locks a TransferableSecretKey with keyring passphrase. Used to prevent
505 * exporting unencrypted secret keys. Operates on a copy of key argument.
506 * @param {!e2e.openpgp.block.TransferableKey} key
507 * @return {?e2e.openpgp.block.TransferableKey}
508 * @private
509 */
510e2e.openpgp.KeyRing.prototype.lockSecretKey_ = function(key) {
511 if (!key) {
512 return key;
513 }
514 // Only let SecretKeys out if they are encrypted.
515 if (key instanceof e2e.openpgp.block.TransferableSecretKey) {
516 var serialized = key.serialize();
517 var parsed = /** @type {!e2e.openpgp.block.TransferableSecretKey} */ (
518 e2e.openpgp.block.factory.parseByteArrayTransferableKey(serialized));
519 parsed.processSignatures();
520 parsed.unlock();
521 var success = false;
522 if (this.passphrase_) {
523 success = parsed.lock(e2e.stringToByteArray(this.passphrase_));
524 } else {
525 success = parsed.lock();
526 }
527 return success ? parsed : null;
528 }
529 return key;
530};

562/**
563 * Searches a public or private key from an email.
564 * @param {string} email The email to search for, or empty to search all.
565 * @param {e2e.openpgp.KeyRing.Type=} opt_type Key type to search for.
566 * @return {?Array.<!e2e.openpgp.block.TransferableKey>} An array of keys for
567 * the given email or null if not found.
568 */
569e2e.openpgp.KeyRing.prototype.searchKey = function(email, opt_type) {
570 if (!opt_type || opt_type == e2e.openpgp.KeyRing.Type.PUBLIC) {
571 return this.searchKey_(this.pubKeyRing_, email);
572 }
573 if (opt_type == e2e.openpgp.KeyRing.Type.PRIVATE) {
574 return this.searchKey_(this.privKeyRing_, email);
575 }
576 if (opt_type == e2e.openpgp.KeyRing.Type.ALL) {
577 var keys = [];
578 var priv = this.searchKey_(this.privKeyRing_, email);
579 if (priv) { // Do this to avoid having a null element.
580 goog.array.extend(keys, priv);
581 }
582 var pub = this.searchKey_(this.pubKeyRing_, email);
583 if (pub) {
584 goog.array.extend(keys, pub);
585 }
586 return keys;
587 }
588 return null;
589};

592/**
593 * Searches a public or private key from email asynchronously. The search is
594 * first performed locally. If the key is not found locally and we're
595 * searching for public key, then searches the public key in the http key
596 * server and imports the the found key to keyring.
597 * @param {string} email The email to search for, or empty to search all.
598 * @param {e2e.openpgp.KeyRing.Type=} opt_type Key type to search for.
599 * @return {!e2e.async.Result.<!Array.<!e2e.openpgp.block.TransferableKey>>}
600 * An array of keys for the given email or [] if not found.
601 */
602e2e.openpgp.KeyRing.prototype.searchKeyLocalAndRemote = function(email,
603 opt_type) {
604 var resultKeys = new e2e.async.Result();
605 var localKeys = this.searchKey(email, opt_type);
606 if (localKeys != null) {
607 resultKeys.callback(localKeys);
608 } else if (opt_type == e2e.openpgp.KeyRing.Type.PUBLIC) {
609 this.searchPublicKeyRemote_(email).addCallback(function(pubKeys) {
610 resultKeys.callback(pubKeys);
611 });
612 } else {
613 resultKeys.callback([]);
614 }
615 return resultKeys;
616};

619/**
620 * Gets all of the keys in the keyring.
621 * @param {boolean=} opt_priv If true, fetch only private keys.
622 * @return {!e2e.openpgp.TransferableKeyMap} A clone of the key ring maps.
623 */
624e2e.openpgp.KeyRing.prototype.getAllKeys = function(opt_priv) {
625 if (opt_priv) {
626 return this.privKeyRing_.clone();
627 }
628 var keys = this.pubKeyRing_.clone();
629 var ids = this.privKeyRing_.getKeys();
630 var values = this.privKeyRing_.getValues();
631 for (var i = 0; i < ids.length; i++) {
632 keys.set(ids[i], goog.array.concat(values[i], keys.get(ids[i], [])));
633 }
634 return keys;
635};

719/**
720 * Imports a new key associated with an email to the key ring. Does not add the
721 * key if there is already a matching key ID.
722 * @param {string} email The email associated with the key.
723 * @param {!e2e.openpgp.block.TransferableKey} keyBlock The key to import.
724 * @param {!e2e.openpgp.TransferableKeyMap} keyRing The keyring to add the keys
725 * to.
726 * @param {!e2e.ByteArray=} opt_passphrase The passphrase used to
727 * protect the key.
728 * @return {boolean} If the key import was successful. False if the key for a
729 * given email address with the same ID already exists in a keyring.
730 * @private
731 */
732e2e.openpgp.KeyRing.prototype.importKey_ = function(
733 email, keyBlock, keyRing, opt_passphrase) {
734 var emailKeyBlocks = [], addKey = false;
735 var keys = [keyBlock.keyPacket].concat(keyBlock.subKeys);
736 goog.array.forEach(keys, function(key) {
737 if (key instanceof e2e.openpgp.packet.SecretKey) {
738 key.cipher.unlockKey(opt_passphrase);
739 // Re-lock the key in plaintext.
740 key.cipher.lockKey();
741 }
742 });
743 if (keyRing.containsKey(email)) {
744 emailKeyBlocks = keyRing.get(email);
745 addKey = !goog.array.some(emailKeyBlocks, function(emailKeyBlock) {
746 return goog.isDef(keyBlock.keyPacket.keyId) && e2e.compareByteArray(
747 emailKeyBlock.keyPacket.keyId, keyBlock.keyPacket.keyId);
748 });
749 // TODO(evn): Merge information when the key block is already in keyring.
750 } else {
751 addKey = true;
752 }
753 if (addKey) {
754 keyRing.set(email, emailKeyBlocks.concat([keyBlock]));
755 this.persist_();
756 }
757 return addKey;
758};

796/**
797 * Encrypts a string with a passphrase.
798 * @param {string} plaintext String to encrypt.
799 * @return {string}
800 * @private
801 */
802e2e.openpgp.KeyRing.prototype.encrypt_ = function(plaintext) {
803 goog.asserts.assert(this.passphrase_, 'passphrase not set');
804 // TODO(adhintz) Cache the s2k result instead of the passphrase.
805 var salt = this.getOrCreateSalt_();
806 var s2k = new e2e.openpgp.IteratedS2K(
807 new e2e.hash.Sha1, salt, 96); // 96 is 65536 iterations.
808 var aes = new e2e.cipher.Aes(e2e.cipher.Algorithm.AES128);
809 var doubleKey = s2k.getKey(
810 e2e.stringToByteArray(this.passphrase_),
811 (aes.keySize + e2e.openpgp.KeyRing.HMAC_KEY_SIZE_));
812 var key = {};
813 key.key = doubleKey.splice(0, aes.keySize);
814 var hmacKey = doubleKey; // Remaining bytes in doubleKey are the hmacKey.
815 aes.setKey(key);
816 var aescfb = new e2e.ciphermode.Cfb(aes);
817 var iv = e2e.random.getRandomBytes(aes.blockSize);
818 var ciphertext = e2e.async.Result.getValue(aescfb.encrypt(
819 e2e.stringToByteArray(plaintext), iv));
820 var formatted = goog.array.concat(
821 e2e.openpgp.KeyRing.VERSION_,
822 iv,
823 ciphertext);
824 var hmac = new goog.crypt.Hmac(
825 new goog.crypt.Sha256(),
826 hmacKey,
827 e2e.openpgp.KeyRing.HASH_BLOCK_SIZE_);
828 var digest = hmac.getHmac(formatted);
829 formatted = goog.array.concat(digest, formatted);
830 return goog.crypt.base64.encodeByteArray(formatted);
831};

879e2e.openpgp.KeyRing.prototype.readKeyData_ = function() {

896 this.deserialize_(serialized);

925/**
926 * Decrypts a string with a passphrase.
927 * @param {string} ciphertext String to decrypt.
928 * @return {string}
929 * @private
930 */
931e2e.openpgp.KeyRing.prototype.decrypt_ = function(ciphertext) {
932 var decoded = goog.crypt.base64.decodeStringToByteArray(ciphertext);
933 var digestSaved = decoded.splice(0, e2e.openpgp.KeyRing.HMAC_SIZE_);
934 goog.asserts.assert(this.passphrase_, 'passphrase not set');
935 var salt = this.getOrCreateSalt_();
936 var s2k = new e2e.openpgp.IteratedS2K(
937 new e2e.hash.Sha1, salt, 96);
938 var aes = new e2e.cipher.Aes(e2e.cipher.Algorithm.AES128);
939 var doubleKey = s2k.getKey(
940 e2e.stringToByteArray(this.passphrase_),
941 (aes.keySize + e2e.openpgp.KeyRing.HMAC_KEY_SIZE_));
942 var key = {};
943 key.key = doubleKey.splice(0, aes.keySize);
944 var hmacKey = doubleKey;
945 aes.setKey(key);
946 var aescfb = new e2e.ciphermode.Cfb(aes);
947 var hmac = new goog.crypt.Hmac(
948 new goog.crypt.Sha256(),
949 hmacKey,
950 e2e.openpgp.KeyRing.HASH_BLOCK_SIZE_);
951 var digest = hmac.getHmac(decoded);
952 if (!e2e.compareByteArray(digest, digestSaved)) {
953 throw new Error('HMAC does not match! LocalStorage modified?');
954 }
955 var version = decoded.shift();
956 if (version != e2e.openpgp.KeyRing.VERSION_) {
957 throw new Error('Unknown keyring version');
958 }
959 var iv = decoded.splice(0, aes.blockSize);
960 var plaintext = e2e.async.Result.getValue(aescfb.decrypt(decoded, iv));
961 return e2e.byteArrayToString(plaintext);
962};

1042/**
1043 * Extracts serialized key data contained in a crypto object.
1044 * @param {{privKey: Array, pubKey: Array}} keyData The map
1045 * to store the extracted data.
1046 * @param {!e2e.cipher.Cipher|!e2e.signer.Signer} cryptor
1047 * The crypto object to extract key material.
1048 * @param {boolean=} opt_subKey Whether the key is a subkey. Defaults to false.
1049 * @param {boolean=} opt_isJS Whether the key material is stored in JS.
1050 * Default to true.
1051 * @private
1052 */
1053e2e.openpgp.KeyRing.prototype.extractKeyData_ = function(
1054 keyData, cryptor, opt_subKey, opt_isJS) {
1055 var version = 0x04;
1056 var timestamp = 0;
1057 var publicConstructor = opt_subKey ?
1058 e2e.openpgp.packet.PublicSubkey : e2e.openpgp.packet.PublicKey;
1059 var secretConstructor = opt_subKey ?
1060 e2e.openpgp.packet.SecretSubkey : e2e.openpgp.packet.SecretKey;
1061 var pubKey = new publicConstructor(
1062 version, timestamp, cryptor);
1063 var serializedPubKey = pubKey.serializePacketBody();
1064
1065 if (!goog.isDef(opt_isJS)) {
1066 opt_isJS = true;
1067 }
1068 var serializedPrivKey;
1069 if (opt_isJS) {
1070 // privKey is MPI, needs to serialize to get the right byte array.
1071 var privKey = e2e.openpgp.Mpi.serialize(cryptor.getKey()['privKey']);
1072 serializedPrivKey = goog.array.flatten(
1073 serializedPubKey,
1074 /* key is not encrypted individually. */
1075 e2e.openpgp.EncryptedCipher.KeyDerivationType.PLAINTEXT,
1076 privKey,
1077 e2e.openpgp.calculateNumericChecksum(privKey));
1078 } else {
1079 // Use dummy s2k
1080 var s2k = new e2e.openpgp.DummyS2k(new e2e.hash.Sha256,
1081 [0x45, 0x32, 0x45],
1082 e2e.openpgp.DummyS2k.E2E_modes.WEB_CRYPTO);
1083 serializedPrivKey = goog.array.flatten(
1084 serializedPubKey,
1085 e2e.openpgp.EncryptedCipher.KeyDerivationType.S2K_CHECKSUM,
1086 s2k.serialize(), [0], [0]);
1087 }
1088
1089 privKey = secretConstructor.parse(serializedPrivKey);
1090 privKey.cipher.unlockKey();
1091 keyData['privKey'].push(privKey);
1092 keyData['pubKey'].push(
1093 publicConstructor.parse(serializedPubKey));
1094};
