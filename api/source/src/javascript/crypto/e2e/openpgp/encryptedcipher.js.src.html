54/**
55 * An EncryptedCipher wraps a e2e.cipher.Cipher so that some of the
56 * operations (sign/decrypt) are only available after unlocking the encrypted
57 * key material.
58 * @param {!e2e.ByteArray} encryptedKeyData The encrypted key data.
59 * @param {e2e.openpgp.EncryptedCipher.KeyDerivationType} keyDerivation The
60 * key derivation style to use.
61 * @param {e2e.cipher.Cipher|e2e.signer.Signer} cipher Instance
62 * of the cipher with the public key.
63 * @param {e2e.cipher.Algorithm=} opt_algorithm The algorithm to
64 * use for decrypting the encrypted key data.
65 * @param {!e2e.ByteArray=} opt_iv The initialization vector for the key
66 * data.
67 * @param {e2e.openpgp.S2k=} opt_s2k The S2K instance to use to generate
68 * the key from a passphrase.
69 * @constructor
70 * @extends {e2e.AlgorithmImpl}
71 * @implements {e2e.cipher.AsymmetricCipher}
72 * @implements {e2e.signer.Signer}
73 */
74e2e.openpgp.EncryptedCipher = function(
75 encryptedKeyData,
76 keyDerivation,
77 cipher,
78 opt_algorithm,
79 opt_iv,
80 opt_s2k) {
81 /**
82 * Used to mark whether the key is locked.
83 * @type {boolean}
84 * @private
85 */
86 this.locked_ = true;
87 /**
88 * The key derivation type to be used for decrypting the data.
89 * @type {e2e.openpgp.EncryptedCipher.KeyDerivationType}
90 * @private
91 */
92 this.keyDerivation_ = keyDerivation;
93 switch (this.keyDerivation_) {
94 case e2e.openpgp.EncryptedCipher.KeyDerivationType.S2K_SHA1:
95 case e2e.openpgp.EncryptedCipher.KeyDerivationType.S2K_CHECKSUM:
96 if (!goog.isDefAndNotNull(opt_s2k)) {
97 throw new e2e.openpgp.error.InvalidArgumentsError(
98 'Invalid S2K in encrypted cipher.');
99 }
100 /**
101 * The S2K specifier for generating the key.
102 * @type {e2e.openpgp.S2k|undefined}
103 * @private
104 */
105 this.s2k_ = opt_s2k;
106 // Falling through, no break;
107 case e2e.openpgp.EncryptedCipher.KeyDerivationType.MD5:
108 if (!goog.isDefAndNotNull(opt_iv) ||
109 !e2e.isByteArray(opt_iv) ||
110 !goog.isDefAndNotNull(opt_algorithm)) {
111 throw new e2e.openpgp.error.InvalidArgumentsError(
112 'Invalid IV for encrypted cipher');
113 }
114 /**
115 * The IV to be used for decrypting the data.
116 * @type {!e2e.ByteArray|undefined}
117 * @private
118 */
119 this.iv_ = opt_iv;
120 /**
121 * The algorithm in which the key is encrypted.
122 * @type {e2e.cipher.Algorithm|undefined}
123 * @private
124 */
125 this.symmetricAlgorithm_ = opt_algorithm;
126 // Falling through, no break;
127 case e2e.openpgp.EncryptedCipher.KeyDerivationType.PLAINTEXT:
128 /**
129 * The cipher to use.
130 * @type {e2e.cipher.Cipher|e2e.signer.Signer}
131 * @private
132 */
133 this.cipher_ = cipher;
134 if (!e2e.isByteArray(encryptedKeyData)) {
135 throw new e2e.openpgp.error.InvalidArgumentsError(
136 'Invalid Encrypted Key Data');
137 }
138 /**
139 * The potentially encrypted key data.
140 * @type {!e2e.ByteArray}
141 */
142 this.encryptedKeyData = encryptedKeyData;
143 break;
144 default:
145 throw new e2e.openpgp.error.InvalidArgumentsError(
146 'Invalid Key Derivation Type.');
147 }
148 goog.base(this, cipher.algorithm);
149};
150goog.inherits(e2e.openpgp.EncryptedCipher, e2e.AlgorithmImpl);

212/**
213 * Locks the key with the given passphrase.
214 * @param {!e2e.ByteArray=} opt_passphrase
215 */
216e2e.openpgp.EncryptedCipher.prototype.lockKey = function(
217 opt_passphrase) {
218 if (this.locked_) {
219 throw new e2e.openpgp.EncryptedCipher.LockedKeyError(this);
220 }
221 if (!goog.isDef(opt_passphrase)) {
222 this.keyDerivation_ =
223 e2e.openpgp.EncryptedCipher.KeyDerivationType.PLAINTEXT;
224 this.iv_ = [];
225 this.symmetricAlgorithm_ = undefined;
226 this.s2k_ = undefined;
227 this.encryptedKeyData = this.keyBytes_.concat(
228 e2e.openpgp.calculateNumericChecksum(this.keyBytes_));
229 } else {
230 // Encrypt all ciphers with passphrase on SHA1 Salted Iterated S2K.
231 this.keyDerivation_ =
232 e2e.openpgp.EncryptedCipher.KeyDerivationType.S2K_SHA1;
233 var salt = e2e.random.getRandomBytes(8);
234 this.symmetricAlgorithm_ =
235 e2e.openpgp.EncryptedCipher.DEFAULT_CIPHER;
236 var cipher = /** @type {e2e.cipher.SymmetricCipher} */ (
237 e2e.openpgp.constants.getInstance(
238 e2e.openpgp.constants.Type.SYMMETRIC_KEY,
239 this.symmetricAlgorithm_));
240 this.iv_ = e2e.random.getRandomBytes(
241 cipher.blockSize);
242 var count = e2e.openpgp.EncryptedCipher.DEFAULT_COUNT;
243 // TODO(evn): Maybe we can use a cheaper function here instead of sha1.
244 this.s2k_ = new e2e.openpgp.IteratedS2K(
245 new e2e.hash.Sha1, salt, count);
246 var symCipher = /** @type {e2e.cipher.SymmetricCipher} */ (
247 e2e.openpgp.constants.getInstance(
248 e2e.openpgp.constants.Type.SYMMETRIC_KEY,
249 e2e.openpgp.EncryptedCipher.DEFAULT_CIPHER));
250 var cfbSymCipher = new e2e.ciphermode.Cfb(symCipher);
251 var key = this.s2k_.getKey(opt_passphrase, symCipher.keySize);
252 symCipher.setKey({key: key});
253 var sha1 = new e2e.hash.Sha1;
254 var hash = sha1.hash(this.keyBytes_);
255 // TODO(evn): Make this call asynchronous.
256 this.encryptedKeyData = e2e.async.Result.getValue(
257 cfbSymCipher.encrypt(this.keyBytes_.concat(hash), this.iv_));
258 }
259};

262/**
263 * Checks if the key data is valid and sets the key. Can throw an error
264 * if the key type is unrecognized.
265 * @param {!e2e.ByteArray} decryptedKeyData The key data.
266 * @private
267 */
268e2e.openpgp.EncryptedCipher.prototype.unlockAndVerifyKey_ = function(
269 decryptedKeyData) {
270 switch (this.keyDerivation_) {
271 case e2e.openpgp.EncryptedCipher.KeyDerivationType.S2K_SHA1:
272 // S2K_SHA1 uses sha1 for the checksum.
273 this.unlockKeyWithSha1Checksum_(decryptedKeyData);
274 break;
275 case e2e.openpgp.EncryptedCipher.KeyDerivationType.PLAINTEXT:
276 case e2e.openpgp.EncryptedCipher.KeyDerivationType.S2K_CHECKSUM:
277 case e2e.openpgp.EncryptedCipher.KeyDerivationType.MD5:
278 // The MD5, plaintext and S2K checksum use numeric checksum.
279 this.unlockKeyWithNumericChecksum_(decryptedKeyData);
280 break;
281 default:
282 // This should never happen, but..
283 throw new e2e.openpgp.error.InvalidArgumentsError(
284 'Invalid Key Derivation Type.');
285 }
286};

289/**
290 * Unlocks the key so that this key can be used for private operations. If this
291 * key isn't encrypted verify that its checksum is correct; otherwise decrypt it
292 * and verify checksum. There are several ways of decrypting an encrypted
293 * cipher, and there are different checksums that can be used to verify the key
294 * was correct. Each has different decryption modes of operation as well, and
295 * the encoding of the data inside the code is formatted differently depending
296 * on which key derivation type.
297 * @param {!e2e.ByteArray=} opt_passphrase The passphrase to use to
298 * decrypt it.
299 */
300e2e.openpgp.EncryptedCipher.prototype.unlockKey = function(
301 opt_passphrase) {
302 if (!this.locked_) {
303 return;
304 }
305 e2e.openpgp.EncryptedCipher.console_.info(
306 'Unlocking key with derivation type', this.keyDerivation_);
307 // Special casing for unencrypted keys.
308 if (this.keyDerivation_ ==
309 e2e.openpgp.EncryptedCipher.KeyDerivationType.PLAINTEXT) {
310 // Plaintext key derivation type uses numeric checksum.
311 this.unlockAndVerifyKey_(this.encryptedKeyData);
312 return;
313 } else if (!goog.isDef(this.iv_) || !goog.isDef(this.symmetricAlgorithm_)) {
314 throw new e2e.openpgp.error.DecryptError(
315 'Missing encrypted key metadata.');
316 }
317
318 var key;
319 var symCipher = /** @type {e2e.cipher.SymmetricCipher} */ (
320 e2e.openpgp.constants.getInstance(
321 e2e.openpgp.constants.Type.SYMMETRIC_KEY,
322 this.symmetricAlgorithm_));
323 var cfbSymCipher = new e2e.ciphermode.Cfb(symCipher);
324 switch (this.keyDerivation_) {
325 case e2e.openpgp.EncryptedCipher.KeyDerivationType.S2K_SHA1:
326 case e2e.openpgp.EncryptedCipher.KeyDerivationType.S2K_CHECKSUM:
327 // For SHA1 and numeric checksum a passphrase is required.
328 if (!goog.isDef(opt_passphrase)) {
329 throw new e2e.openpgp.error.MissingPassphraseError();
330 }
331 key = this.s2k_.getKey(opt_passphrase, symCipher.keySize);
332 break;
333 case e2e.openpgp.EncryptedCipher.KeyDerivationType.MD5:
334 // For MD5 there's no S2K, and the key is simply the MD5 of the secret.
335 var md5 = new e2e.hash.Md5;
336 if (!goog.isDef(opt_passphrase)) {
337 throw new e2e.openpgp.error.MissingPassphraseError();
338 }
339 key = md5.hash(opt_passphrase);
340 break;
341 default:
342 // This should never happen, but..
343 throw new e2e.openpgp.error.InvalidArgumentsError(
344 'Invalid Key Derivation Type.');
345 }
346 symCipher.setKey({key: key});
347 // TODO(evn): Make this call asynchronous.
348 var decryptedData = e2e.async.Result.getValue(
349 cfbSymCipher.decrypt(this.encryptedKeyData, this.iv_));
350 this.unlockAndVerifyKey_(decryptedData);
351};

354/**
355 * Verifies the data is valid, and unlocks the key to do private operations.
356 * @param {!e2e.ByteArray} data The data to verify and import.
357 * @private
358 */
359e2e.openpgp.EncryptedCipher.prototype.unlockKeyWithSha1Checksum_ =
360 function(data) {
361 var key = data.slice(0, -20);
362 var chk = data.slice(-20);
363 var sha1 = new e2e.hash.Sha1;
364 var hash = /** @type {!e2e.ByteArray} */ (sha1.hash(key));
365 if (!e2e.compareByteArray(chk, hash)) {
366 e2e.openpgp.EncryptedCipher.console_.info(
367 'Shasum mismatch, assuming wrong passphrase.');
368 throw new e2e.openpgp.error.WrongPassphraseError();
369 }
370 this.unlockKey_(key);
371};

374/**
375 * Verifies the data is valid, and unlocks the key to do private operations.
376 * @param {!e2e.ByteArray} data The data to verify and import.
377 * @private
378 */
379e2e.openpgp.EncryptedCipher.prototype.unlockKeyWithNumericChecksum_ =
380 function(data) {
381 var key = data.slice(0, -2);
382 var checksum = data.slice(-2);
383 if (!e2e.compareByteArray(checksum,
384 e2e.openpgp.calculateNumericChecksum(key))) {
385 e2e.openpgp.EncryptedCipher.console_.info(
386 'Numeric checksum mismatch, assuming wrong passphrase.');
387 throw new e2e.openpgp.error.WrongPassphraseError();
388 }
389 this.unlockKey_(key);
390};

393/**
394 * Unlocks the encrypted cipher so that it can do private operations.
395 * @param {!e2e.ByteArray} keyBytes The key data to use to get the MPIs.
396 * @private
397 */
398e2e.openpgp.EncryptedCipher.prototype.unlockKey_ = function(keyBytes) {
399 // Make a copy of the key to avoid changing it.
400 var key = keyBytes.slice(0);
401 var keyData = this.cipher_.getKey();
402 switch (this.cipher_.algorithm) {
403 case e2e.cipher.Algorithm.RSA:
404 case e2e.cipher.Algorithm.RSA_ENCRYPT:
405 case e2e.signer.Algorithm.RSA_SIGN:
406 keyData.d = e2e.openpgp.Mpi.parse(key);
407 keyData.p = e2e.openpgp.Mpi.parse(key);
408 keyData.q = e2e.openpgp.Mpi.parse(key);
409 keyData.u = e2e.openpgp.Mpi.parse(key);
410 break;
411 case e2e.signer.Algorithm.DSA:
412 case e2e.cipher.Algorithm.ELGAMAL:
413 keyData.x = e2e.openpgp.Mpi.parse(key);
414 break;
415 case e2e.cipher.Algorithm.ECDH:
416 case e2e.signer.Algorithm.ECDSA:
417 keyData.privKey = e2e.openpgp.Mpi.parse(key);
418 break;
419 default:
420 e2e.openpgp.EncryptedCipher.console_.warn(
421 'Unknown cipher algorithm', this.cipher_.algorithm);
422 throw new e2e.openpgp.error.InvalidArgumentsError('Unknown algorithm');
423 }
424 // TODO(user): Figure out what loc this is once multiple locs are supported.
425 keyData.loc = e2e.algorithm.KeyLocations.JAVASCRIPT;
426 this.cipher_.setKey(keyData);
427 this.locked_ = false;
428 this.keyBytes_ = keyBytes;
429};
