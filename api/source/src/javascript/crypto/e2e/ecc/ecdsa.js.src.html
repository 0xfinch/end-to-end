40/**
41 * Representation of an instance of the ECDSA protocol.
42 * @param {!e2e.ecc.PrimeCurve} curveName The curve used for
43 * this protocol.
44 * @param {{pubKey: !e2e.ByteArray, privKey: (!e2e.ByteArray|undefined)}=}
45 * opt_key The public and/or private key used in this protocol.
46 * @constructor
47 * @extends {e2e.ecc.Protocol}
48 */
49e2e.ecc.Ecdsa = function(curveName, opt_key) {
50 e2e.ecc.Ecdsa.base(this, 'constructor', curveName, opt_key);
51 switch (curveName) {
52 case e2e.ecc.PrimeCurve.P_256:
53 this.hash_ = new e2e.hash.Sha256();
54 break;
55 case e2e.ecc.PrimeCurve.P_384:
56 this.hash_ = new e2e.hash.Sha384();
57 break;
58 case e2e.ecc.PrimeCurve.P_521:
59 this.hash_ = new e2e.hash.Sha512();
60 break;
61 default:
62 throw new e2e.error.InvalidArgumentsError(
63 'Unknown algorithm for ECDSA: ' + curveName);
64 }
65};
66goog.inherits(e2e.ecc.Ecdsa, e2e.ecc.Protocol);

82/**
83 * Applies the signing algorithm to the data.
84 * @param {!Uint8Array|!e2e.ByteArray|string} message The data to sign.
85 * @return {!e2e.signer.signature.Signature}
86 */
87e2e.ecc.Ecdsa.prototype.sign = function(message) {
88 var sig;
89 var digest = this.hash_.hash(message);
90 do {
91 var k = this.generatePerMessageNonce_(digest);
92 sig = this.signWithNonce_(digest, k);
93 } while (sig == null);
94 return sig;
95};

110/**
111 * Generates the ECDSA signature using the provided per-message secret.
112 * Returns null in the very rare case that r or s evaluates to 0, and we need
113 * to try a different nonce.
114 *
115 * @param {!e2e.ByteArray} digest The digest of the message being signed.
116 * @param {!e2e.BigNum} k The per-message secret.
117 * @return {?e2e.signer.signature.Signature}
118 * @private
119 */
120e2e.ecc.Ecdsa.prototype.signWithNonce_ = function(digest, k) {
121 goog.asserts.assertObject(this.params, 'Domain params should be defined.');
122 goog.asserts.assertObject(this.getPrivateKey(),
123 'Private key value should be defined.');
124 var N = this.params.n;
125 // Sanity check on the per-message nonce that it's in [1, N-1].
126 if (k.isGreaterOrEqual(N) ||
127 k.isEqual(e2e.BigNum.ZERO)) {
128 throw new e2e.error.InvalidArgumentsError(
129 'Failed to sign message: invalid per-message nonce.');
130 }
131 // kG = (x1, y1)
132 // r = x1 mod n.
133 var r = N.residue(this.params.g.multiply(k).getX().toBigNum());
134 if (r.isEqual(e2e.BigNum.ZERO)) {
135 return null;
136 } // e = digest as bignum
137 var e = new e2e.BigNum(digest);
138 // s = k^{-1}(e + dr) (mod n)
139 var privateKeyBytes = this.getPrivateKey();
140 var privateKey =
141 N.residue(this.params.bigNumFromPrivateKey(privateKeyBytes));
142 var s = N.modAdd(N.residue(e), N.modMultiply(privateKey, r));
143 if (s.isEqual(e2e.BigNum.ZERO)) {
144 return null;
145 }
146 s = N.modMultiply(s, N.modInverse(k));
147 var sig = {
148 'r': r.toByteArray(),
149 's': s.toByteArray(),
150 'hashValue': digest
151 };
152 return sig;
153};

156/**
157 * Applies the verification algorithm to the data.
158 * @param {!Uint8Array|!e2e.ByteArray|string} message The data to verify.
159 * @param {{s: !e2e.ByteArray, r: !e2e.ByteArray}} sig The
160 * signature to verify.
161 * @return {boolean}
162 */
163e2e.ecc.Ecdsa.prototype.verify = function(message, sig) {
164 goog.asserts.assertObject(this.params, 'Domain params should be defined.');
165 goog.asserts.assertObject(this.getPublicKey(),
166 'Public key value should be defined.');
167
168 var N = this.params.n;
169 var r = new e2e.BigNum(sig['r']);
170 var s = new e2e.BigNum(sig['s']);
171 // r and s should be in [1, N-1].
172 if (r.isGreaterOrEqual(N) ||
173 r.isEqual(e2e.BigNum.ZERO) ||
174 s.isGreaterOrEqual(N) ||
175 s.isEqual(e2e.BigNum.ZERO)) {
176 return false;
177 }
178 // e = H(m)
179 var e = new e2e.BigNum(this.hash_.hash(message));
180 // w = s^{-1} mod n
181 var w = N.modInverse(s);
182 // u1 = ew mod n
183 var u1 = N.modMultiply(N.residue(e), w);
184 // u2 = rw mod n
185 var u2 = N.modMultiply(r, w);
186 // X = u1 * G + u2 * Q = s^{-1} * (e + dr) * G = k * G
187 var X = this.params.g.multiply(u1).add(
188 this.getPublicKeyAsPoint().multiply(u2));
189 if (X.isInfinity()) {
190 return false;
191 }
192 var x = N.residue(X.getX().toBigNum());
193 return x.isEqual(r);
194};

197/**
198 * Generates a random number used as per-message secret in ECDSA.
199 * This code includes a hash of the secret key and the message digest of the
200 * message in its calculation in order to minimize the damage that could be
201 * done by a bad RNG.
202 *
203 * This implementation is intended to comply with section B.5.2 of
204 * FIPS-186-4 for generating per-message secret nonces.
205 *
206 * @param {!e2e.ByteArray} digest The digest of the message.
207 * @return {!e2e.BigNum}
208 * @private
209 */
210e2e.ecc.Ecdsa.prototype.generatePerMessageNonce_ = function(digest) {
211 var N = this.params.n;
212 var nonceLength = Math.ceil(this.params.curve.keySizeInBits() / 8);
213 var hasher = new e2e.hash.Sha512();
214 var privateKey = this.getPrivateKey();
215 while (privateKey.length < nonceLength) {
216 // Avoid accidentally leaking the length of the private key.
217 privateKey.unshift(0);
218 }
219 var privateKeyDigest = hasher.hash(privateKey);
220 do {
221 var randomBytes = e2e.random.getRandomBytes(nonceLength);
222 var nonceBytes = [];
223 do {
224 hasher.reset();
225 // When generating more than 512 bytes, the following line ensures that
226 // the multiple pieces are different
227 hasher.update(e2e.wordToByteArray(nonceBytes.length));
228 hasher.update(privateKeyDigest);
229 hasher.update(digest);
230 hasher.update(randomBytes);
231 nonceBytes = goog.array.concat(nonceBytes, hasher.digest());
232 } while (nonceBytes.length < nonceLength);
233 nonceBytes = nonceBytes.slice(0, nonceLength);
234 nonceBytes[0] >>= (8 * nonceLength - N.getBitLength());
235 var nonce = new e2e.BigNum(nonceBytes);
236 // nonce must be in the range [1..N-1]
237 } while (nonce.isEqual(e2e.BigNum.ZERO) || nonce.compare(N) >= 0);
238 return nonce;
239};
