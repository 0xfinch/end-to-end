32/**
33 * Benchmarks for a given curve.
34 * @param {!e2e.ecc.PrimeCurve.<string>} curve The curve
35 * @return {!Object.<string,Object>} Performance stats (object is filled only
36 * when tests are run outside browser).
37 */
38e2e.ecc.eccTester.runBenchmarkForCurve = function(curve) {
39 var message = 'Whisky bueno: ¡excitad mi frágil pequeña vejez!';
40 var params = e2e.ecc.DomainParam.fromCurve(curve);
41 var keypair = e2e.ecc.Protocol.generateKeyPair(curve);
42 var ecdsa = new e2e.ecc.Ecdsa(curve, keypair);
43 var signature = ecdsa.sign(message);
44 var tests = [];
45 e2e.testing.Util.addBenchmark(tests,
46 function() { params.g.multiply(params.n); },
47 curve + ' Fast Multiply');
48 e2e.testing.Util.addBenchmark(tests,
49 function() { params.g.negate().multiply(params.n); },
50 curve + ' Slow Multiply');
51 e2e.testing.Util.addBenchmark(tests,
52 function() { params.n.modInverse(params.curve.B.toBigNum()); },
53 curve + ' Bignum modInverse');
54 e2e.testing.Util.addBenchmark(tests,
55 function() { ecdsa.sign(message); },
56 curve + ' Sign');
57 e2e.testing.Util.addBenchmark(tests,
58 function() { ecdsa.verify(message, signature); },
59 curve + ' Verify');
60 return e2e.testing.Util.runPerfTests(tests, 10, 10000);
61};
