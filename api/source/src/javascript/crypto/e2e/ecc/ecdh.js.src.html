17/**
18 * @fileoverview An implementation of the one-pass ECDH key exchange protocol,
19 * which is the ECC version of HAC 12.51.
20 * In this protocol, Alice knows Bob's public key B and she wants to derive
21 * a shared secret with him. She generates an ephemeral private key a,
22 * computes the ephemeral public key A = aG (G is the base point of the
23 * curve), and derives the shared secret S = aB. She then sends A to Bob,
24 * who shall derive the shared secret with his private key b by computing
25 * S = bA = baG = aB.
26 * Note: the actual shared secret is the x-coordinate of S.
27 * @author thaidn@google.com (Thai Duong)
28 */

37/**
38 * Representation of an instance of the one-pass ECDH key agreement protocol,
39 * which is the ECC version of HAC 12.51.
40 * In this protocol, Alice knows Bob's public key B and she wants to derive
41 * a shared secret with him. She generates an ephemeral private key a,
42 * computes the ephemeral public key A = aG (G is the base point of the
43 * curve), and derives the shared secret S = aB. She then sends A to Bob,
44 * who shall derive the shared secret with his private key b by computing
45 * S = bA = baG = aB.
46 * Note: the actual shared secret is the x-coordinate of S.
47 * @param {!e2e.ecc.PrimeCurve} curve The curve used for this protocol.
48 * @param {{pubKey: !e2e.ByteArray, privKey: (!e2e.ByteArray|undefined)}=}
49 * opt_key The public and/or private key used in this protocol.
50 * @constructor
51 * @extends {e2e.ecc.Protocol}
52 */

54 e2e.ecc.Ecdh.base(this, 'constructor', curve, opt_key);
55};

59/**
60 * Acts as Alice in the one-pass ECDH protocol. Alice knows Bob's public key B
61 * and she wants to derive a shared secret with him. She generates an
62 * ephemeral private key a, computes the ephemeral public key A = aG (G is
63 * the base point of the curve), and derives the shared secret S = aB.
64 * The public key used here should belong to Bob. The returned message
65 * consists of Alice's ephemeral public key and the shared key, which is
66 * the x-coordinate of the shared secret value.
67 * @param {!e2e.ByteArray=} opt_bobPubKey Bob's public key.
68 * @return {{secret: !e2e.ByteArray,
69 * pubKey: !e2e.ByteArray}}
70 */
71e2e.ecc.Ecdh.prototype.alice = function(opt_bobPubKey) {
72 goog.asserts.assertObject(this.params, 'Domain params should be defined.');
73 var publicKey;
74 if (goog.isDefAndNotNull(opt_bobPubKey)) {
75 publicKey = this.params.curve.pointFromByteArray(opt_bobPubKey);
76 } else {
77 publicKey = this.getPublicKeyAsPoint();
78 }
79 goog.asserts.assertObject(publicKey, 'Public key should be defined.');
80 // The per-message secret is also the ephemeral private key in ECDH.
81 var ephemeralKeyPair = this.params.generateKeyPair();
82 var ephemeralPrivateBigNum = ephemeralKeyPair['privateKeyBigNum'];
83 var ephemeralPublicKey = ephemeralKeyPair['publicKey'];
84 return {
85 'secret': this.params.calculateSharedSecret(publicKey,
86 ephemeralPrivateBigNum),
87 'pubKey': ephemeralPublicKey
88 };
89};

92/**
93 * Acts as Bob in the one-pass ECDH protocol. Bob receives Alice's ephemeral
94 * public key from which he derives the shared secret using his private key
95 * b by computing S = bA = baG = aB.
96 * The public key used here is Alice's ephemeral public key. The private
97 * key should belong to Bob. The returned message is the shared key, which
98 * is the x-coordinate of the shared secret value.
99 * @param {!e2e.ByteArray} alicePubKey Alice's ephemeral public key,
100 * which is sent by Alice to Bob as part of the protocol.
101 * @param {!e2e.ByteArray=} opt_bobPrivKey Bob's private key.
102 * @return {{secret: !e2e.ByteArray}}
103 */
104e2e.ecc.Ecdh.prototype.bob = function(alicePubKey, opt_bobPrivKey) {
105 goog.asserts.assertObject(this.params, 'Domain params should be defined.');
106
107 var privateKeyBytes = opt_bobPrivKey || this.getPrivateKey();
108 var privateKey = this.params.bigNumFromPrivateKey(privateKeyBytes);
109 goog.asserts.assertObject(privateKey, 'Private key should be defined.');
110 var publicKey = this.params.curve.pointFromByteArray(alicePubKey);
111 return {
112 'secret': this.params.calculateSharedSecret(publicKey, privateKey)
113 };
114};
