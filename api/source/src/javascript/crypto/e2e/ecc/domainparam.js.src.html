75/**
76 * Representation of domain parameters for prime curves.
77 * @param {!e2e.ecc.curve.Curve} curve The elliptic curve.
78 * @param {!e2e.ecc.point.Point} g The base point.
79 * @param {!e2e.BigPrimeNum} n The order of the base point.
80 * @constructor
81 */
82e2e.ecc.DomainParam = function(curve, g, n) {
83 this.curve = curve;
84 this.g = g;
85 this.n = n;
86 goog.asserts.assert(g);
87 // Expensive sanity check. Documentation claims that asserts are
88 // elided if code is compiled. But this is such an expensive
89 // test that we're being paranoid.
90 goog.asserts.assert(!goog.asserts.ENABLE_ASSERTS ||
91 g.multiply(n).isIdentity());
92};

95/**
96 * Gets curve's name from curve's OID.
97 * @param {e2e.ecc.PrimeCurveOid} curveOid Curve's OID (including
98 * the one-byte length prefix), as defined in section 11 in RFC 6637.
99 * @return {e2e.ecc.PrimeCurve}
100 */
101e2e.ecc.DomainParam.curveNameFromCurveOid = function(curveOid) {
102 if (goog.array.equals(curveOid, e2e.ecc.PrimeCurveOid.P_256)) {
103 return e2e.ecc.PrimeCurve.P_256;
104 } else if (goog.array.equals(curveOid, e2e.ecc.PrimeCurveOid.P_384)) {
105 return e2e.ecc.PrimeCurve.P_384;
106 } else if (goog.array.equals(curveOid, e2e.ecc.PrimeCurveOid.P_521)) {
107 return e2e.ecc.PrimeCurve.P_521;
108 }
109 throw new e2e.error.UnsupportedError('Invalid curve OID');
110};

152/**
153 * Obtains the set of domain parameters for a particular curve.
154 * @param {!e2e.ecc.PrimeCurve} curveName The curve to retrieve.
155 * @return {!e2e.ecc.DomainParam}
156 */
157e2e.ecc.DomainParam.fromCurve = function(curveName) {
158 if (curveName in e2e.ecc.DomainParam) {
159 // already generated this curve.
160 return e2e.ecc.DomainParam[curveName];
161 }
162 var result;
163 switch (curveName) {
164 case e2e.ecc.PrimeCurve.P_256:
165 case e2e.ecc.PrimeCurve.P_384:
166 case e2e.ecc.PrimeCurve.P_521:
167 result = e2e.ecc.DomainParam.NIST.fromCurve(curveName);
168 break;
169 case e2e.ecc.PrimeCurve.CURVE_25519:
170 result = e2e.ecc.DomainParam.Curve25519.fromCurve(curveName);
171 break;
172 case e2e.ecc.PrimeCurve.ED_25519:
173 result = e2e.ecc.DomainParam.Ed25519.fromCurve(curveName);
174 break;
175 default:
176 throw new e2e.error.UnsupportedError(
177 'Curve is not known or not supported');
178 }
179 e2e.ecc.DomainParam[curveName] = result;
180 return result;
181};

239/**
240 * Representation of domain parameters for NIST prime curves. These values
241 * are defined in
242 * http://csrc.nist.gov/groups/ST/toolkit/documents/dss/NISTReCur.pdf
243 * @param {!e2e.ecc.PrimeCurve} curveName The domain params to retrieve.
244 * @return {!e2e.ecc.DomainParam.NIST}
245 */
246e2e.ecc.DomainParam.NIST.fromCurve = function(curveName) {
247 var constants, fastModulus, fastMultiplyTable;
248 if (curveName == e2e.ecc.PrimeCurve.P_256) {
249 constants = e2e.ecc.constant.P_256;
250 fastModulus = e2e.ecc.fastModulus.Nist.P_256;
251 fastMultiplyTable = e2e.ecc.constant.p_256.G_FAST_MULTIPLY_TABLE;
252 } else if (curveName == e2e.ecc.PrimeCurve.P_384) {
253 constants = e2e.ecc.constant.P_384;
254 fastModulus = e2e.ecc.fastModulus.Nist.P_384;
255 fastMultiplyTable = e2e.ecc.constant.p_384.G_FAST_MULTIPLY_TABLE;
256 } else if (curveName == e2e.ecc.PrimeCurve.P_521) {
257 constants = e2e.ecc.constant.P_521;
258 fastMultiplyTable = e2e.ecc.constant.p_521.G_FAST_MULTIPLY_TABLE;
259 } else {
260 throw new e2e.error.InvalidArgumentsError('Unknown curve.');
261 }
262 var q = new e2e.BigPrimeNum(constants.Q); // prime field
263 var b = new e2e.BigPrimeNum(constants.B); // parameter of curve
264 if (fastModulus) {
265 q.setFastModulusType(fastModulus);
266 }
267 var curve = new e2e.ecc.curve.Nist(q, b);
268
269 var g = curve.pointFromByteArray(constants.G);
270 if (fastMultiplyTable) {
271 g.setFastMultiplyTable(fastMultiplyTable);
272 }
273 var n = new e2e.BigPrimeNum(constants.N); // order of group
274 n.setFastModulusType(e2e.FastModulus.FFFFFF);
275 return new e2e.ecc.DomainParam.NIST(curve, g, n);
276};

279/**
280 * Generates a random private key in the range [1, N-1] as described
281 * in section B.4.2 of FIPS-186-4.
282 *
283 * @override
284 */
285e2e.ecc.DomainParam.NIST.prototype.generateKeyPair = function(
286 opt_privateKey) {
287 var privateKey;
288 var expectedKeyLength = Math.ceil(this.curve.keySizeInBits() / 8);
289 var count = 0;
290 do {
291 if (goog.isDefAndNotNull(opt_privateKey)) {
292 if (count++ != 0) {
293 throw new e2e.error.InvalidArgumentsError(
294 'Bad private key');
295 }
296 goog.asserts.assert(opt_privateKey.length == expectedKeyLength,
297 'Private key length must be ' + expectedKeyLength + ' bytes');
298 privateKey = opt_privateKey;
299 } else {
300 privateKey = e2e.random.getRandomBytes(expectedKeyLength);
301 // remove extra bits (if any) in the high-order byte
302 privateKey[0] >>= (8 * expectedKeyLength - this.curve.keySizeInBits());
303 }
304 var multiplier = new e2e.BigNum(privateKey);
305 } while (multiplier.isEqual(e2e.BigNum.ZERO) ||
306 multiplier.compare(this.n) >= 0);
307 var publicKey = this.g.multiply(multiplier);
308 return {
309 'privateKey': privateKey,
310 'publicKey': publicKey.toByteArray(),
311 'privateKeyBigNum': multiplier,
312 'publicKeyPoint': publicKey
313 };
314};

317/** @override */
318e2e.ecc.DomainParam.NIST.prototype.calculateSharedSecret = function(
319 peerPublicKey, myPrivateKey) {
320 var S = peerPublicKey.multiply(myPrivateKey);
321 if (S.isInfinity()) {
322 throw new e2e.error.InvalidArgumentsError(
323 'ECDH: Cannot derive shared secret.');
324 }
325 var Xbytes = S.getX().toBigNum().toByteArray();
326 var fieldSize = Math.ceil(this.curve.keySizeInBits() / 8);
327 // Pads X if needed.
328 while (Xbytes.length < fieldSize) {
329 goog.array.insertAt(Xbytes, 0x00, 0);
330 }
331 return Xbytes;
332};

377/** @override */
378e2e.ecc.DomainParam.Curve25519.prototype.generateKeyPair = function(
379 opt_privateKey) {
380 // A private key is any sequence of 32 bytes
381 var privateKey;
382 if (goog.isDefAndNotNull(opt_privateKey)) {
383 goog.asserts.assert(opt_privateKey.length == 32,
384 'Private key length must be 32 bytes');
385 privateKey = opt_privateKey;
386 } else {
387 privateKey = e2e.random.getRandomBytes(32);
388 }
389 // Clamp a copy of the private key
390 var p = privateKey.slice();
391 p[0] &= ~7; // Must be a multiple of 8
392 p[31] = (p[31] & 63) | 64; // high bit clear, next higher bit set.
393 // Use that clamped private key as a little-endian bignum, and scalar
394 // multiply the base.
395 var multiplier = new e2e.BigNum(p.reverse());
396 var publicKey = this.g.multiply(multiplier);
397 return {
398 'privateKey': privateKey,
399 'publicKey': publicKey.toByteArray(),
400 'privateKeyBigNum': multiplier,
401 'publicKeyPoint': publicKey
402 };
403};

411 p[31] = (p[31] & 63) | 64; // high bit clear, next higher bit set.

500/**
501 * Expands a 32-byte protokey into a 64 byte EdDSA private key.
502 *
503 * p = bytes[0:32] are used as a private scalar; p*B = the public key
504 *
505 * Several bits are clamped to zero or one in order to clear the cofactor
506 * of 8 (equivalently, clear the torsion), and to make it more difficult
507 * to implement scalar multiplication incorrectly.
508 *
509 * k = bytes[32:64] is used as the key input to the nonce-derivation PRF.
510 *
511 * @param {!e2e.hash.Hash} hash The hash function.
512 * @param {!e2e.ByteArray} privateKey The private (proto-)key.
513 * @return {{multiplier: !e2e.BigNum, extra: !e2e.ByteArray}}
514 */
515e2e.ecc.DomainParam.Ed25519.prototype.expandPrivateKey = function(hash,
516 privateKey) {
517 var digest = hash.hash(privateKey);
518 goog.asserts.assert(digest.length == 64, 'Digest length must be 64 bytes');
519
520 // Clamp the low 32 bytes of the expanded private key to use as a multiplier
521 var bytes = digest.slice(0, 32);
522 bytes[0] &= ~7;
523 bytes[31] = (bytes[31] & 63) | 64;
524 var a = this.bigNumFromPrivateKey(bytes);
525
526 return {
527 multiplier: a,
528 extra: digest.slice(32)
529 };
