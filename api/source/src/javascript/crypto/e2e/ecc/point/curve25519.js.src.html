33/**
34 * Constructs a point on the elliptic curve Curve25519 defined over a
35 * prime field.
36 * @param {!e2e.ecc.curve.Curve} curve The curve.
37 * @param {!e2e.ecc.Element} x The x Jacobian coordinate.
38 * @param {!e2e.ecc.Element=} opt_z The optional z Jacobian coordinate.
39 * @constructor
40 * @extends {e2e.ecc.point.Point}
41 */
42e2e.ecc.point.Curve25519 = function(curve, x, opt_z) {
43 e2e.ecc.point.Curve25519.base(
44 this, 'constructor', curve);
45 var z = opt_z || curve.ONE;
46 goog.asserts.assert(!x.isEqual(curve.ZERO) || !z.isEqual(curve.ZERO),
47 '(0,0) point is undefined');
48 /**
49 * The x Jacobian projective coordinate of this.
50 * @type {!e2e.ecc.Element}
51 */
52 this.x = x;
53
54 /**
55 * The z Jacobian projective coordinate of this.
56 * @type {!e2e.ecc.Element}
57 */
58 this.z = z;
59
60 /**
61 * The equivalent affine Point.
62 * @type {e2e.ecc.point.Curve25519}
63 */
64 this.affine = this.z.isEqual(curve.ONE) ? this : null;
65};
66goog.inherits(e2e.ecc.point.Curve25519, e2e.ecc.point.Point);

90/**
91 * Returns the equivalent affine point.
92 * @return {!e2e.ecc.point.Curve25519}
93 * @private
94 */
95e2e.ecc.point.Curve25519.prototype.getAffine_ = function() {
96 if (this.affine_ == null) {
97 var x = this.x.multiply(this.z.inverse());
98 this.affine_ = new e2e.ecc.point.Curve25519(this.curve, x);
99 }
100 return this.affine_;
101};

121/**
122 * Compares another point with this. Return true if they are the same.
123 * @param {!e2e.ecc.point.Point} that The point to compare.
124 * @return {boolean}
125 */
126e2e.ecc.point.Curve25519.prototype.isEqual = function(that) {
127 if (this.isInfinity()) {
128 return that.isInfinity();
129 }
130 if (that.isInfinity()) {
131 return this.isInfinity();
132 }
133 // x and y coordinates must be equal
134 return this.x.multiply(that.z).isEqual(that.x.multiply(this.z));
135};

159/**
160 * Doubles this, and return the result as a new point.
161 * @return {!e2e.ecc.point.Curve25519}
162 * @private
163 */
164e2e.ecc.point.Curve25519.prototype.twice_ = function() {
165 var t1 = this.x.add(this.z).square();
166 var t2 = this.x.subtract(this.z).square();
167 var t3 = t1.subtract(t2);
168 var xOut = t1.multiply(t2);
169 var zOut = t3.multiply(t1.add(t3.multiply(this.curve.A4)));
170 return new e2e.ecc.point.Curve25519(this.curve, xOut, zOut);
171};

174/**
175 * Returns the value of this + that as a new point. This method must not
176 * be called if delta is infinity (i.e. this == that) or if delta is the zero
177 * point.
178 *
179 * @param {!e2e.ecc.point.Curve25519} that
180 * @param {!e2e.ecc.point.Curve25519} delta this - that
181 * @return {!e2e.ecc.point.Curve25519}
182 * @private
183 */
184e2e.ecc.point.Curve25519.prototype.add_ = function(that, delta) {
185 // This formula work only if that - this is neither zero nor infinity
186 goog.asserts.assert(!delta.x.isEqual(this.curve.ZERO));
187 goog.asserts.assert(!delta.z.isEqual(this.curve.ZERO));
188 var t1 = (this.x.subtract(this.z)).multiply(that.x.add(that.z));
189 var t2 = (this.x.add(this.z)).multiply(that.x.subtract(that.z));
190 var xOut = (t1.add(t2)).square().multiply(delta.z);
191 var zOut = (t1.subtract(t2)).square().multiply(delta.x);
192 return new e2e.ecc.point.Curve25519(this.curve, xOut, zOut);
193};

196/**
197 * @override
198 */
199e2e.ecc.point.Curve25519.prototype.multiply = function(k) {
200 // Normally, k > 0 and this isn't infinity or the zero point. They are
201 // included here for completeness.
202 if (this.isInfinity() || k.isEqual(e2e.BigNum.ZERO)) {
203 return this.curve.INFINITY;
204 }
205 if (this.x.isEqual(this.curve.ZERO)) {
206 // The zero point has order 2. It either returns itself or infinity.
207 return k.isOdd() ? this : this.curve.INFINITY;
208 }
209 var m = [this, this.twice_()];
210 for (var i = k.getBitLength() - 2; i >= 0; --i) {
211 // Assertion. Let j = k >>> i; Then
212 // m = j * this and m1 = (j + 1) * this;
213 var bit = k.isBitSet(i) | 0; // force boolean to 0 or 1.
214 // Following code is cryptic because it has no branches.
215 // If bit = 0, we want
216 // m[0] = old m[0] + old m[0];
217 // m[1] = old m[0] + old m[1];
218 // If bit = 1, we want
219 // m[0] = old m[0] + old m[1];
220 // m[1] = old m[1] + old m[1]
221 m[1 - bit] = m[0].add_(m[1], this);
222 m[bit] = m[bit].twice_();
223 }
224 return m[0];
