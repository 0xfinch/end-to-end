40 * The OAEP encoding method as described in RFC 3447 section 7.1 with SHA-1,
41 * MGF1 and an empty label. This is compatible with OpenSSL's
42 * RSA_PKCS1_OAEP_PADDING padding mode.

48 * Is equivalent to the output of new e2e.hash.Sha1().hash([]).
49 * @private {!e2e.ByteArray}

65/**
66 * Encodes the given message according to the OAEP method as described in
67 * RFC 3447 section 7.1.1 with SHA-1, MGF1 and an empty label.
68 * @param {number} k The size of the key in bytes.
69 * @param {!e2e.ByteArray} m The message to encode.
70 * @return {!e2e.ByteArray} The encoded message.
71 */
72e2e.pkcs.eme.Oaep.prototype.encode = function(k, m) {
73 // OAEP can operate only on messages of length up to k - 2hLen - 2 octets,
74 // where hlen is the length of the output from the underlying hash function,
75 // which is SHA-1 in this case, and k is the length in octets of the modulus.
76 if (m.length > k - this.labelHash_.length - 2) {
77 throw new e2e.pkcs.Error('Message too long.');
78 }
79 var ps = goog.array.repeat(
80 0x00, k - m.length - this.labelHash_.length * 2 - 2);
81 var db = goog.array.flatten(this.labelHash_, ps, 0x01, m);
82 goog.asserts.assert(db.length == k - this.labelHash_.length - 1);
83
84 var seed = e2e.random.getRandomBytes(this.labelHash_.length);
85 var dbMask = this.maskGenerationFunction_(
86 seed, k - this.labelHash_.length - 1);
87 var maskedDb = goog.crypt.xorByteArray(db, dbMask);
88 var seedMask = this.maskGenerationFunction_(
89 maskedDb, this.labelHash_.length);
90 var maskedSeed = goog.crypt.xorByteArray(seed, seedMask);
91
92 return [0x00].concat(maskedSeed).concat(maskedDb);
93};

96/**
97 * Decodes the given message according to the OAEP method as defined in
98 * RFC 3447 section 7.1.2 with SHA-1, MGF1 and an empty label.
99 * Note: this function does not make length check as length is checked in
100 * the decryption function as specified in the RFC. Throwing additional
101 * errors may lead to timing attack. As a result, it must not be used
102 * anywhere except in the context of RSA-OAEP decryption or interoperative
103 * testing.
104 * @param {number} k The size of the key in bytes.
105 * @param {!e2e.ByteArray} em The encoded message.
106 * @return {?e2e.ByteArray} The decoded message or null if decoding has
107 * error.
108 */
109e2e.pkcs.eme.Oaep.prototype.decode = function(k, em) {
110 var maskedSeed = em.slice(1, this.labelHash_.length + 1);
111 var maskedDb = em.slice(this.labelHash_.length + 1);
112 var seedMask = this.maskGenerationFunction_(
113 maskedDb, this.labelHash_.length);
114 var seed = goog.crypt.xorByteArray(maskedSeed, seedMask);
115 var dbMask = this.maskGenerationFunction_(
116 seed, k - this.labelHash_.length - 1);
117 var db = goog.crypt.xorByteArray(maskedDb, dbMask);
118 var labelHash = db.slice(0, this.labelHash_.length);
119 var paddedMsg = db.slice(this.labelHash_.length);
120 // In fixed-timing up to the length of paddedMsg, find the first possition of
121 // 0x1 in paddedMsg and checks that all bytes before it are zero.
122 var foundOne = 0;
123 var indexOne = -1;
124 var allZerosBeforeOne = 1;
125 for (var i = 0; i < paddedMsg.length; i++) {
126 // Best effort fixed-timing number comparisons.
127 var isOne = (paddedMsg[i] === 1) | 0;
128 var isZero = (paddedMsg[i] === 0) | 0;
129 indexOne = e2e.fixedtiming.select(i, indexOne, isOne & (foundOne ^ 1));
130 foundOne |= isOne;
131 allZerosBeforeOne = e2e.fixedtiming.select(0, allZerosBeforeOne,
132 (isZero ^ 1) & (foundOne ^ 1));
133 }
134 // Make sure that the errors are indistinguishable, otherwise it's vulnerable
135 // to timing attack.
136 var error = 0;
137 error |= em[0];
138 error |= (!e2e.compareByteArray(this.labelHash_, labelHash) | 0);
139 error |= (foundOne ^ 1);
140 error |= (allZerosBeforeOne ^ 1);
141 if (error) {
142 return null;
143 }
144 return paddedMsg.slice(indexOne + 1);
145};

148/**
149 * Mask generation function (MGF1) used in the OAEP encoding method as
150 * described in RFC 3447 section B.2.1 and RFC 2437 section 10.2.1.
151 * This is the mask function used in OpenSSL.
152 * @param {!e2e.ByteArray} seed The seed from which mask is generated.
153 * @param {number} maskLen intended length in octets of the mask, at most
154 * 2^32 * HashLength.
155 * @return {!e2e.ByteArray}
156 * @private
157 */
158e2e.pkcs.eme.Oaep.prototype.maskGenerationFunction_ =
159 function(seed, maskLen) {
160 if (maskLen > (this.labelHash_.length * 0x100000000)) {
161 throw new e2e.pkcs.Error('Mask too long.');
162 }
163 var mask = [];
164 var n = Math.ceil(maskLen / this.labelHash_.length);
165 for (var i = 0; i < n; ++i) {
166 mask = mask.concat(new e2e.hash.Sha1().hash(
167 seed.concat(e2e.dwordArrayToByteArray([i]))));
168 }
169 goog.asserts.assert(mask.length >= maskLen);
170 return mask.slice(0, maskLen);
171};

184/**
185 * Encodes the given message according to EME PKCS1-v1.5.
186 * @param {number} k The size of the key in bytes.
187 * @param {!e2e.ByteArray} m The message to encode.
188 * @return {!e2e.ByteArray} The encoded message.
189 */
190e2e.pkcs.eme.Pkcs1.prototype.encode = function(k, m) {
191 var em, ps;
192 if (m.length > k - 11) {
193 throw new e2e.pkcs.Error('Message too long.');
194 }
195 ps = e2e.random.getRandomBytes(k - m.length - 3, [0]);
196 em = [0x00, 0x02].concat(ps).concat([0x00]).concat(m);
197 return em;
198};

284/**
285 * Converts a decodeStateTransitions to an Array to accelerate access time.
286 * @param {Object.<string|number, e2e.pkcs.eme.Pkcs1.Transition>}
287 * trans The transitions to set in the Array.
288 * @param {e2e.pkcs.eme.Pkcs1.DecodeState} state The state that the

291e2e.pkcs.eme.Pkcs1.decodeStateToArray = function(trans, state) {
292 var arr = [];
293 for (var i = 0; i < 256; i++) {
294 arr.push(trans[i] || trans['default']);
295 }
296 e2e.pkcs.eme.Pkcs1.decodeStateTransitionsArray[state] = arr;

303/**
304 * Decodes the given encoded message according to EME PKCS1-v1.5. This MUST be
305 * done close to constant time, that's why we have a state machine.
306 * @param {!e2e.ByteArray} em The data to decode.
307 * @param {number=} opt_messageLength The expected length of the message.
308 * @param {number=} opt_keySize The size of the key (which should be em size).
309 * @return {!e2e.ByteArray} The decoded message.
310 */
311e2e.pkcs.eme.Pkcs1.prototype.decode = function(
312 em, opt_messageLength, opt_keySize) {
313 var error = 0;
314 var msg = [];
315 var em2 = em.splice(0);
316 if (goog.isDef(opt_messageLength) && goog.isDef(opt_keySize)) {
317 if (opt_keySize == em.length + 1) {
318 // Some implementations remove leading 0's, so we add them back.
319 em.unshift(0);
320 }
321 // We know the length of the key, so we don't need to scan.
322 msg = em2.slice(-opt_messageLength);
323 var paddingLength = em2.length - opt_messageLength;
324 // Should be 0.
325 error |= em[0];
326 // Should be 2.
327 error |= em[1] - 2;
328 for (var i = 2; i < paddingLength; i++) {
329 // Should be anything but 0.
330 error |= em[i] ^ 0xFF;
331 }
332 // Should be 0.
333 error |= em[paddingLength];
334 } else {
335 var state = e2e.pkcs.eme.Pkcs1.DecodeState.START;
336
337 var nextState = e2e.pkcs.eme.Pkcs1.decodeStateTransitionsArray;
338 var undef = nextState[
339 e2e.pkcs.eme.Pkcs1.DecodeState.UNDEFINED];
340 var c = goog.object.getCount(
341 e2e.pkcs.eme.Pkcs1.DecodeState) + 1;
342 var accumulators = [];
343 for (var i = 0; i < c; i++) {
344 accumulators[i] = [];
345 }
346 msg = accumulators[e2e.pkcs.eme.Pkcs1.DecodeState.KEY];
347 var rnd = accumulators[e2e.pkcs.eme.Pkcs1.DecodeState.RANDOM];
348
349 if (em2.length < 10) {
350 em2 = goog.array.repeat(0, 10);
351 error = 1;
352 }
353
354 goog.array.forEach(em2, function(b, i) {
355 // === WARNING ===
356 // If we can't know the expected length of the key, we need to visit all
357 // bytes in constant time to verify they are valid. Don't add any code
358 // that might fork in this function.
359 var hasError, newState;
360 var newStateObj = nextState[state][b];
361 newState = newStateObj.state;
362 hasError = newStateObj.error;
363 accumulators[state].push(b);
364 error |= hasError;
365 state = newState;
366 });
367 error |= 1 * (msg.length == 0);
368 error |= 1 * (rnd.length <= 8);
369 }
370
371 if (error) {
372 throw new e2e.pkcs.Error('Decryption error.');
373 }
374 return msg;
375};
