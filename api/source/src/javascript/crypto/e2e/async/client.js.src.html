30/**
31 * Class for the client-side part of the Port-based asynchronous service.
32 * @param {MessagePort} port The port to register the client on.
33 * @param {function(*)=} opt_errback The default function to call on errors.
34 * @constructor
35 */
36e2e.async.Client = function(port, opt_errback) {
37 /**
38 * The port to use to communicate with the service.
39 * @type {MessagePort}
40 * @private
41 */
42 this.port_ = port;
43
44 /**
45 * The default function to call on errors.
46 * @type {function(*)}
47 * @private
48 */
49 this.errback_ = opt_errback || goog.nullFunction;
50};
51
52
53/**
54 * Invokes a service method and returns its value to callback.
55 * @param {string} method The method to invoke in the service.
56 * @param {Array.<*>} args The arguments to send to the method.
57 * @param {function(*)} callback The callback for this service.
58 * @param {function(*)=} opt_errback The callback to call for errors.
59 */
60e2e.async.Client.prototype.call = function(
61 method, args, callback, opt_errback) {
62 var mc = new MessageChannel();
63 var otherPorts = [];
64 this.registerCallback_(mc.port1, callback, opt_errback || this.errback_);
65 for (var i = 0; i < args.length; i++) {
66 if (typeof args[i] == 'function') {
67 var portWrap = e2e.async.util.wrapFunction(
68 /** @type {function(...*):*} */(args[i]));
69 otherPorts.push(portWrap);
70 // There's no off-by-1, as the first port will be mc.port2.
71 args[i] = {'__port__': otherPorts.length};
72 }
73 }
74 this.port_.postMessage({
75 'method': method,
76 'arguments': args}, [mc.port2].concat(otherPorts));
77};

80/**
81 * Invokes a service method and returns a Deferred that will be resolved with
82 * its return value.
83 * @param {string} func The function name.
84 * @param {!Array.<*>} callArguments The arguments of the function.
85 * @return {!e2e.async.Result} Deferred result.
86 */
87e2e.async.Client.prototype.deferredCall = function(
88 func, callArguments) {
89 var result = new e2e.async.Result();
90 this.call(func, callArguments, goog.bind(result.callback, result),
91 goog.bind(result.errback, result));
92 return result;
93};

96/**
97 * Registers a callback for a specific method.
98 * @param {MessagePort} port The port to receive the return value on.
99 * @param {function(*)} callback The callback to use.
100 * @param {function(*)} errback The errback to use.
101 * @private
102 */
103e2e.async.Client.prototype.registerCallback_ = function(
104 port, callback, errback) {
105 port.onmessage = function(event) {
106 if (event.data.error) {
107 errback(event.data.error);
108 } else {
109 callback(event.data.returnValue);
110 }
111 };
