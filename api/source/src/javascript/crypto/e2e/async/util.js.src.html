26/**
27 * Wraps a function within a port.
28 * @param {function(...*):*} callback The callback to use.
29 * @return {MessagePort} The port that wraps the callback.
30 */
31e2e.async.util.wrapFunction = function(callback) {
32 var mc = new MessageChannel();
33 mc.port1.onmessage = function(event) {
34 var args = [];
35 for (var i = 0; i < event.data.arguments.length; i++) {
36 var arg = event.data.arguments[i];
37 if (goog.isObject(arg) && goog.isNumber(arg.__port__)) {
38 args.push(e2e.async.util.unwrapFunction(event.ports[arg.__port__]));
39 } else {
40 args.push(arg);
41 }
42 }
43 try {
44 var returnValue = callback.apply(null, args);
45 if (goog.async.Deferred && returnValue instanceof goog.async.Deferred) {
46 returnValue.addCallback(function(ret) {
47 e2e.async.util.return_(event.target, ret, '');
48 }).addErrback(function(err) {
49 e2e.async.util.return_(event.target, undefined, String(err));
50 });
51 } else {
52 e2e.async.util.return_(event.target, returnValue, '');
53 }
54 } catch (e) {
55 if (e instanceof Error) {
56 e2e.async.util.return_(event.target, undefined, String(e.message));
57 } else {
58 e2e.async.util.return_(event.target, undefined, 'Unknown error');
59 }
60 }
61 };
62 return mc.port2;
63};

66/**
67 * Sends a return message to the port.
68 * @param {MessagePort} port The port to respond to.
69 * @param {*} returnValue The return value of the function.
70 * @param {string} error The error to send.
71 * @private
72 */
73e2e.async.util.return_ = function(port, returnValue, error) {
74 port.postMessage({
75 'returnValue': returnValue,
76 'error': error
77 });
78};

81/**
82 * Unwraps a function from a port.
83 * @param {MessagePort} port The port that is wrapping the function.
84 * @return {function(...*):!e2e.async.Result} A function that calls the wrapped
85 * function and returns a deferred result object.
86 */
87e2e.async.util.unwrapFunction = function(port) {
88 return function() {
89 var result = new e2e.async.Result();
90 port.onmessage = function(event) {
91 if (event.data.error) {
92 result.errback(event.data.error);
93 } else {
94 result.callback(event.data.returnValue);
95 }
96 };
97 var args = [];
98 var ports = [];
99 for (var i = 0; i < arguments.length; i++) {
100 if (typeof arguments[i] == 'function') {
101 var wrappedPort = e2e.async.util.wrapFunction(arguments[i]);
102 ports.push(wrappedPort);
103 args.push({
104 '__port__': ports.length - 1
105 });
106 } else {
107 args.push(arguments[i]);
108 }
109 }
110 port.postMessage({
111 'arguments': args
112 }, ports);
113 return result;
114 };
